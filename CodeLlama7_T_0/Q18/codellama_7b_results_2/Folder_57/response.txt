def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    sum = 0
    start_index = 51
    end_index = 76

    # Iterate through the list and calculate the sum of all integers divisible by either -90 or -59
    for i in range(start_index, end_index + 1):
