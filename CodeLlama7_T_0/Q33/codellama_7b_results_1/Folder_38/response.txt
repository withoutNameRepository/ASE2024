def return_vowels(my_string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 17
    end_index = 65
    result = []
    for char in my_string[start_index:end_index]:
        if char.isalpha() and char.lower() in vowels and char > 'U' and char <= '{':
            result.append(char)
    return result
