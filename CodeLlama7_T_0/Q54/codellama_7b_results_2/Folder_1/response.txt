def is_prime(n):
    # Check if n is less than 2
    if n < 2:
        # If n is less than 2, it is not a prime number
        return False

    # Check if n is divisible by any number between 2 and the square root of n
    for i in range(2, int(n ** 0.5) + 1):
        # If n is divisible by any number between 2 and the square root of n, it is not a prime number
        if n % i == 0:
            return False

    # If n is greater than 1 and it is not divisible by any number between 2 and the square root of n, it is a prime number
    return True
