def if_perfect_num(my_list):
    # Check if the list has at least 21 elements
    if len(my_list) < 21:
        return False

    # Get the integer at index 21
    num = my_list[21]

    # Check if the number is a perfect number
    if num == sum(range(1, num))) {

    } else {
