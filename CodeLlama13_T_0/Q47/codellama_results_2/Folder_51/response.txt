def palindromes_of_specific_lengths(string):
    palindromes = set()
    for i in range(2, 8):
        for j in range(i+3, i+5):
            substr = string[i:j]
            if len(substr) >= 3 and len(substr) <= 4 and substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
