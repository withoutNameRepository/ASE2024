def find_second_largest_num(my_list):
    # Check if list is empty
    if len(my_list) == 0:
        return None
    
    # Get the second largest element
    second_largest = sorted(my_list)[-2]
    
    # Check if second largest element is in the given range
    if my_list.index(second_largest) >= 56 and my_list.index(second_largest) <= 98:
        return second_largest
    
    # If not, return None
    return None
