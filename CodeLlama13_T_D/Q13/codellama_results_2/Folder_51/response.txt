def find_second_largest_num(nums):
    if len(nums) < 2:
        return None
    sorted_nums = sorted(nums)[1:8]
    second_largest = sorted_nums[-1]
    for i in range(len(sorted_nums)-2, -1, -1):
        if sorted_nums[i] == second_largest:
            return None
    return second_largest
