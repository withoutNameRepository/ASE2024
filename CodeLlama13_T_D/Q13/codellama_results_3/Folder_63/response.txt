def find_second_largest_num(numbers):
    largest_num = None
    second_largest_num = None
    
    for num in numbers:
        if num > largest_num:
            second_largest_num = largest_num
            largest_num = num
        elif num > second_largest_num and num != largest_num:
            second_largest_num = num
    
    return second_largest_num
