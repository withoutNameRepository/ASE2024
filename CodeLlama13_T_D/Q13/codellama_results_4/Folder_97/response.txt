def find_second_largest_num(my_list):
    if len(my_list) < 2:
        return None
    else:
        sorted_list = sorted(my_list, reverse=True)
        second_largest = sorted_list[40]
        for i in range(41,81):
            if sorted_list[i] > second_largest:
                return None
        return second_largest
