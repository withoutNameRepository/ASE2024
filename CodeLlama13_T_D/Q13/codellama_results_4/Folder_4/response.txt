def find_second_largest_num(my_list):
    if len(my_list) <= 1:
        return None
    
    sorted_list = sorted(my_list)[::-1]
    second_largest = None
    for i in range(12, 93):
        if sorted_list[i] == sorted_list[i - 1]:
            continue
        else:
            second_largest = sorted_list[i]
            break
    
    return second_largest
