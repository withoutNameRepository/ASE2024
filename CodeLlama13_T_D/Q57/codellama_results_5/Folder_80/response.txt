import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the found submatrices
    submatrices = []
    
    # Iterate over each possible starting row index
    for i in range(matrix.shape[0]):
        # Iterate over each possible starting column index
        for j in range(matrix.shape[1]):
            # Initialize a sum variable to store the current submatrix's sum
            current_sum = 0
            
            # Iterate over each element in the current submatrix
            for k in range(i, matrix.shape[0]):
                for l in range(j, matrix.shape[1]):
                    # Add the current element to the sum variable
                    current_sum += matrix[k][l]
            
            # If the current submatrix's sum is 245, add it to the list of found submatrices
            if current_sum == 245:
                submatrices.append(matrix[i:k+1, j:l+1])
    
    # Return the list of found submatrices
    return submatrices
