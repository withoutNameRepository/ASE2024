import numpy as np

def submatrix_with_particular_sum(matrix):
    # Flatten the matrix into a 1D array
    flat_matrix = matrix.flatten()
    
    # Create an empty list to store the results
    submatrices = []
    
    # Iterate over all possible starting indices of the submatrix
    for i in range(len(flat_matrix)):
        # Calculate the sum of the current submatrix
        sum_submatrix = np.sum(flat_matrix[i:i+5])
        
        # If the sum is equal to -38, add the submatrix to the list
        if sum_submatrix == -38:
            submatrices.append(np.reshape(flat_matrix[i:i+5], (2, 2)))
    
    return submatrices
