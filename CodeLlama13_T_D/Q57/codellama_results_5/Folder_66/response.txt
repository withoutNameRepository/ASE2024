import numpy as np

def submatrix_with_particular_sum(matrix):
    m, n = matrix.shape
    for i in range(m - 1):
        for j in range(n - 1):
            for k in range(i + 1, m + 1):
                for l in range(j + 1, n + 1):
                    if np.sum(matrix[i:k, j:l]) == 99:
                        return matrix[i:k, j:l].tolist()
    return []
