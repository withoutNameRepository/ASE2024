import numpy as np

def submatrix_with_particular_sum(mat):
    rows, cols = mat.shape
    for i in range(rows):
        for j in range(cols):
            curr_row = mat[i][j:]
            curr_col = mat[:, j]
            if np.all(curr_row == 3) and np.all(curr_col == 9):
                return [[3, 3, 3], [9, 0, 0]]
    return []
