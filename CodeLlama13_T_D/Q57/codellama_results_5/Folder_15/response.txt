import numpy as np

def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for i in range(rows):
        for j in range(cols):
            curr_sum = 0
            for k in range(i, rows):
                for l in range(j, cols):
                    curr_sum += matrix[k][l]
                    if curr_sum == -9:
                        return [[matrix[m][n] for n in range(j, cols)] for m in range(i, k+1)]
            curr_sum = 0
    return []
