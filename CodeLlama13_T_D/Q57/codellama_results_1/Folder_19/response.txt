import numpy as np
def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for row in range(rows):
        for col in range(cols):
            current_row = row
            current_col = col
            while current_row < rows and current_col < cols:
                submatrix = matrix[current_row : row+1, current_col : col+1]
                if np.sum(submatrix) == -97:
                    return [submatrix]
                current_row += 1
                current_col += 1
    return []
