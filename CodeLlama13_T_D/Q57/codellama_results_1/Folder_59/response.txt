import numpy as np

def submatrix_with_particular_sum(matrix):
    result = []
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            sum = 0
            for k in range(i, len(matrix)):
                for l in range(j, len(matrix[0])):
                    sum += matrix[k][l]
                if sum == 2:
                    result.append((i, j, k, l))
    return result
