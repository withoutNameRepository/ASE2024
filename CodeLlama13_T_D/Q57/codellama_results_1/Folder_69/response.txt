import numpy as np

def submatrix_with_particular_sum(matrix):
    # Convert the matrix to a list of lists for easier manipulation
    matrix = matrix.tolist()
    
    # Initialize an empty list to store the submatrices with sum 752
    submatrices = []
    
    # Iterate over each row and column of the matrix
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            # Initialize a temporary list to store the submatrix elements
            temp_list = []
            
            # Iterate over each element in the current submatrix
            for k in range(i, len(matrix)):
                for l in range(j, len(matrix[0])):
                    # Add the current element to the temporary list
                    temp_list.append(matrix[k][l])
                    
                    # Check if the sum of the elements in the temporary list is equal to 752
                    if np.sum(temp_list) == 752:
                        # If it is, add the submatrix to the list of submatrices with sum 752
                        submatrices.append([[k, l] for k, l in temp_list])
                    
    return submatrices
