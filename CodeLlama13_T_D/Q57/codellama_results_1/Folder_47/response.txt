import numpy as np

def submatrix_with_particular_sum(matrix):
    result = []
    for i in range(len(matrix)):
        for j in range(i, len(matrix[0])):
            submatrix = matrix[i:j+1]
            if sum(np.prod(submatrix)) == 27:
                result.append(submatrix)
    return result
