import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices with sum equal to 64
    submatrices = []
    
    # Iterate over each submatrix of size (n-1)x(m-1)
    for i in range(matrix.shape[0] - 1):
        for j in range(matrix.shape[1] - 1):
            # Extract the current submatrix
            submatrix = matrix[i:i+2, j:j+2]
            
            # Check if the sum of all integers in the submatrix is equal to 64
            if np.sum(submatrix) == 64:
                # If yes, add the submatrix to the list of submatrices
                submatrices.append(submatrix)
    
    # Return the list of submatrices with sum equal to 64
    return submatrices
