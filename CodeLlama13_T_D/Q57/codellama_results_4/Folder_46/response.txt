import numpy as np

def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for row in range(rows):
        for col in range(cols):
            # Initialize a new submatrix with the current position as the top-left corner
            submatrix = np.zeros((rows - row, cols - col), dtype=np.int64)
            # Fill the submatrix with values from the original matrix
            for i in range(submatrix.shape[0]):
                for j in range(submatrix.shape[1]):
                    submatrix[i, j] = matrix[row + i, col + j]
            # Check if the sum of all integers in the submatrix equals 416
            if np.sum(submatrix) == 416:
                return submatrix
    return []
