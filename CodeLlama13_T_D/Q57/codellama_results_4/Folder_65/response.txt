import numpy as np
def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for i in range(rows-1):
        for j in range(cols-1):
            if matrix[i:i+2,j:j+2].sum() == 78:
                yield np.array([[matrix[i][j], matrix[i][j+1]], [matrix[i+1][j], matrix[i+1][j+1]]])
