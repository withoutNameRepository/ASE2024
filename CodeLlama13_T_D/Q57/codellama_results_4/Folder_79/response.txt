import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices with sum -47
    submatrices = []

    # Loop through each row of the matrix
    for i in range(matrix.shape[0]):
        # Loop through each column of the matrix
        for j in range(matrix.shape[1]):
            # Calculate the sum of all integers in the submatrix starting from (i, j) and having size (3, 3)
            sum_submatrix = np.sum(matrix[i:i+3, j:j+3])

            # If the sum is -47, add the submatrix to the list of submatrices
            if sum_submatrix == -47:
                submatrices.append((i, j))

    return submatrices
