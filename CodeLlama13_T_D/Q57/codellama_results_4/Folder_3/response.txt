import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices
    submatrices = []
    
    # Iterate over each submatrix of the given matrix
    for i in range(matrix.shape[0] - 1):
        for j in range(matrix.shape[1] - 1):
            # Initialize a temporary list to store the integers in the current submatrix
            tmp = []
            
            # Iterate over each element in the current submatrix
            for k in range(i, i + 2):
                for l in range(j, j + 2):
                    # Add the current integer to the temporary list
                    tmp.append(matrix[k, l])
            
            # Check if the sum of all integers in the current submatrix equals 54
            if sum(tmp) == 54:
                # If so, add the current submatrix to the list of submatrices
                submatrices.append((i, j))
    
    return submatrices
