import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the found submatrices
    submatrices = []
    
    # Iterate over all possible starting rows and columns
    for i in range(matrix.shape[0] - 1):
        for j in range(matrix.shape[1] - 1):
            # Initialize a temporary sum variable to store the current submatrix's sum
            temp_sum = 0
            
            # Iterate over the current submatrix and calculate its sum
            for k in range(i, i + 2):
                for l in range(j, j + 2):
                    temp_sum += matrix[k, l]
            
            # If the current submatrix's sum is equal to -394, add it to the list of found submatrices
            if temp_sum == -394:
                submatrices.append((i, j))
    
    return submatrices
