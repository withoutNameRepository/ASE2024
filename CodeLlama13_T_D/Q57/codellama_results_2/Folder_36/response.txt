import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices with the required sum
    submatrices = []
    
    # Loop through each submatrix of the given matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Check if the current submatrix has the required sum
            if np.sum(matrix[i:, j:]) == -779:
                # If so, add it to the list of submatrices
                submatrices.append(matrix[i:, j:])
    
    return submatrices
