import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the results
    result = []
    
    # Iterate over each row of the matrix
    for i in range(matrix.shape[0]):
        # Iterate over each column of the matrix
        for j in range(matrix.shape[1]):
            # Initialize a variable to store the sum of integers in the current submatrix
            curr_sum = 0
            
            # Iterate over each row and column of the submatrix
            for k in range(i, matrix.shape[0]):
                for l in range(j, matrix.shape[1]):
                    # Add the current integer to the sum
                    curr_sum += matrix[k][l]
            
            # If the sum of integers in the current submatrix is equal to -57, add it to the result list
            if curr_sum == -57:
                result.append(matrix[i:k+1, j:l+1])
    
    # Return the result list
    return result
