import numpy as np

def submatrix_with_particular_sum(mat):
    rows, cols = mat.shape
    for r in range(rows):
        for c in range(cols):
            for i in range(r+1, rows+1):
                for j in range(c+1, cols+1):
                    submat = mat[r:i, c:j]
                    if np.sum(submat) == -13:
                        return list(submat)
    return []
