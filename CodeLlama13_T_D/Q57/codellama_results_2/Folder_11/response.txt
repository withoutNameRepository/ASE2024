import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices with sum 38
    submatrices = []
    
    # Loop through each row of the matrix
    for i in range(matrix.shape[0]):
        # Loop through each column of the matrix
        for j in range(matrix.shape[1]):
            # Initialize a variable to store the sum of integers in the current submatrix
            curr_sum = 0
            
            # Loop through each element in the current submatrix
            for k in range(i, matrix.shape[0]):
                for l in range(j, matrix.shape[1]):
                    # Add the current element to the sum
                    curr_sum += matrix[k][l]
            
            # If the current submatrix has a sum of 38, add it to the list of submatrices
            if curr_sum == 38:
                submatrices.append(matrix[i:k+1, j:l+1])
    
    # Return the list of submatrices with sum 38
    return submatrices
