import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the results
    results = []
    
    # Iterate over all possible starting rows and columns of the submatrix
    for row in range(matrix.shape[0]):
        for col in range(matrix.shape[1]):
            # Check if the current position is a valid starting point for the submatrix
            if matrix[row, col] == -63:
                # Find all possible rows and columns that form a submatrix with the given sum
                rows = []
                cols = []
                for i in range(row, matrix.shape[0]):
                    for j in range(col, matrix.shape[1]):
                        if matrix[i, j] == -63:
                            rows.append(i)
                            cols.append(j)
                # If a submatrix with the given sum is found, add it to the results list
                if len(rows) > 0 and len(cols) > 0:
                    results.append((row, col, rows, cols))
    
    return results
