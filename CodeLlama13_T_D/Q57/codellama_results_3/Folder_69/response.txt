import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices with the particular sum
    submatrices = []
    
    # Iterate over each row of the matrix
    for i in range(matrix.shape[0]):
        # Iterate over each column of the matrix
        for j in range(matrix.shape[1]):
            # Initialize a temporary list to store the elements of the submatrix
            temp_list = []
            
            # Add the element at position (i, j) to the temporary list
            temp_list.append(matrix[i][j])
            
            # Iterate over each row and column below the current position
            for k in range(i+1, matrix.shape[0]):
                for l in range(j+1, matrix.shape[1]):
                    # Add the element at position (k, l) to the temporary list
                    temp_list.append(matrix[k][l])
            
            # Check if the sum of all elements in the temporary list is equal to 752
            if np.sum(temp_list) == 752:
                # Add the submatrix to the list of submatrices with the particular sum
                submatrices.append(temp_list)
    
    return submatrices
