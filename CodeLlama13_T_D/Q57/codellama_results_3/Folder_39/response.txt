```
import numpy as np

def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for i in range(rows):
        for j in range(cols):
            for k in range(i, rows):
                for l in range(j, cols):
                    submatrix = matrix[i:k+1, j:l+1]
                    if submatrix.sum() == 66:
                        return [submatrix]
    return []
```
Note that this code uses the `numpy` library to handle matrices and vectors, and also the `itertools` library to generate all possible combinations of submatrices.
