import numpy as np

def submatrix_with_particular_sum(matrix):
    rows, cols = matrix.shape
    for i in range(rows):
        for j in range(cols):
            sum_of_submatrix = 0
            for k in range(i, rows):
                for l in range(j, cols):
                    sum_of_submatrix += matrix[k][l]
                    if sum_of_submatrix == 126:
                        return [matrix[i:k+1, j:l+1]]
            if sum_of_submatrix > 126:
                break
    return []
