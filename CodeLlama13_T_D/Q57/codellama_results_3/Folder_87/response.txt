import numpy as np
def submatrices_with_particular_sum(matrix):
    result = []
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            for r in range(i, matrix.shape[0]):
                for c in range(j, matrix.shape[1]):
                    sub = matrix[i:r+1, j:c+1]
                    if np.sum(sub) == 35:
                        result.append(sub)
    return result
