import numpy as np
def submatrix_with_particular_sum(matrix):
    if matrix is None:
        return []
    
    n = len(matrix)
    m = len(matrix[0])
    result = []
    
    for i in range(n-1):
        for j in range(m-1):
            for k in range(i+1, n):
                for l in range(j+1, m):
                    if matrix[i][j] + matrix[k][l] == -157:
                        result.append([matrix[i:k+1], matrix[j:l+1]])
    
    return result
