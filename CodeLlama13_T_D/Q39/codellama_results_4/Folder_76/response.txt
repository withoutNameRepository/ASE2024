def return_n_greatest_chars(string):
    # Create a list of tuples containing the character and its ASCII value
    char_ascii = [(char, ord(char)) for char in string]

    # Sort the list in descending order based on the ASCII value
    sorted_list = sorted(char_ascii, key=lambda x: x[1], reverse=True)

    # Return a list of exactly 3 characters with the greatest ASCII values
    return [char for char, _ in sorted_list[:3]]
