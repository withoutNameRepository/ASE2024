def return_n_greatest_chars(string):
    # Create a dictionary to map each character to its ASCII value
    char_ascii = {char: ord(char) for char in string}
    
    # Sort the characters by their ASCII values in descending order
    sorted_chars = sorted(char_ascii, key=lambda x: char_ascii[x], reverse=True)[:82]
    
    return ''.join(sorted_chars)
