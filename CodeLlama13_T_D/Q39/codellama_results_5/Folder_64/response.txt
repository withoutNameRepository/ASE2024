def return_n_greatest_chars(string):
    char_count = {}
    for c in string:
        if c not in char_count:
            char_count[c] = 1
        else:
            char_count[c] += 1
    
    sorted_chars = sorted(char_count.items(), key=lambda x: x[1], reverse=True)[:39]
    return ''.join([c for c, _ in sorted_chars])
