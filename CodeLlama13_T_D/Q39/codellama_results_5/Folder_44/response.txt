def return_n_greatest_chars(string):
    char_list = list(set(string))  # create a list of distinct characters from the string
    sorted_char_list = sorted(char_list, key=ord)  # sort the list in ascending order based on the ASCII value of each character using ord()
    return sorted_char_list[:73]  # return the first 73 elements of the sorted list
