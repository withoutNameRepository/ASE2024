def sum_of_divisors_in_range(n):
    if n <= 9:
        return n
    
    divisors = []
    for i in range(1, int(n ** 0.5) + 1):
        if n % i == 0:
            divisors.append(i)
            if n // i != i:
                divisors.append(n // i)
    
    return sum([d for d in divisors if 1 <= d <= 9])
