def sum_of_divisors_in_range(n):
    if n <= 19 or n >= 69:
        return 0
    
    sum = 0
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0 and 19 <= i <= 69:
            sum += i
            j = n // i
            if j != i and 19 <= j <= 69:
                sum += j
    return sum
