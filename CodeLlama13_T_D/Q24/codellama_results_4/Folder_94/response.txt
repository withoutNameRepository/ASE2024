def sum_of_divisors_in_range(n):
    if n < 35 or n > 37:
        return 0
    else:
        divisors = []
        for i in range(1, int(math.sqrt(n)) + 1):
            if n % i == 0 and i >= 35 and i <= 37:
                divisors.append(i)
                if i * (n // i) != n and i != n // i and i >= 35 and i <= 37:
                    divisors.append(n // i)
        return sum(divisors)
