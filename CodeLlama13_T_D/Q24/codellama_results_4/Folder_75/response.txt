def sum_of_divisors_in_range(n):
    if n <= 38 or n >= 65:
        return 0
    else:
        divisors = []
        for i in range(1, int(n ** 0.5) + 1):
            if n % i == 0:
                if i >= 38 and i <= 65:
                    divisors.append(i)
                    if n // i != i:
                        divisors.append(n // i)
        return sum(divisors)
