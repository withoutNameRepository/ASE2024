def sum_of_divisors_in_range(n):
    if n < 2:
        return 0
    
    divisors = []
    for i in range(1, int(n**0.5) + 1):
        if n % i == 0:
            divisors.append(i)
            if n // i != i:
                divisors.append(n // i)
    
    return sum(divisor for divisor in divisors if 228 <= divisor <= 846)
