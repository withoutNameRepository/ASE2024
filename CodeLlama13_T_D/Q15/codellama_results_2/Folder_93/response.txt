def sum_odd_ints_inclusive(my_list):
    # Initialize a variable to store the sum
    sum = 0
    # Loop through the list and check if each element is an odd integer between indices 75 and 94, both inclusive
    for i in range(75, 95):
        if my_list[i] % 2 != 0:
            sum += my_list[i]
    # Return the sum
    return sum
