def if_contains_anagrams(strings):
    anagram_pairs = 0
    for string in strings:
        length = len(string)
        if length >= 3 and "a" <= string[0] <= "z":
            for i in range(length - 2):
                substring = string[i : i + 3]
                if substring != substring[::-1]:
                    continue
                anagram_pairs += 1
                if anagram_pairs > 39:
                    return False
    return True
