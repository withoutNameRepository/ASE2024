def if_contains_anagrams(strings):
    anagrams = []
    for s in strings:
        anagram_pairs = get_anagram_pairs(s, 3)
        anagrams.extend(anagram_pairs)
    
    return len(set(anagrams)) <= 54

def get_anagram_pairs(string, min_length):
    pairs = []
    for i in range(len(string)):
        if len(string[i]) >= min_length:
            anagram = ''.join(sorted(string[i]))
            for j in range(i+1, len(string)):
                if len(string[j]) >= min_length and sorted(string[j]) == anagram:
                    pairs.append((string[i], string[j]))
    
    return pairs
