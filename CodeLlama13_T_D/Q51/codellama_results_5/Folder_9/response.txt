def if_contains_anagrams(str_list):
    anagram_count = 0
    for s1 in str_list:
        for s2 in str_list:
            if s1 != s2 and len(s1) >= 3 and len(s2) >= 3:
                if ''.join(sorted(s1.lower())) == ''.join(sorted(s2.lower())):
                    anagram_count += 1
    return anagram_count <= 475
