def if_contains_anagrams(strings):
    pairs = []
    for i in range(len(strings)):
        s1 = strings[i].lower()
        for j in range(i+1, len(strings)):
            s2 = strings[j].lower()
            if is_anagram(s1, s2):
                pairs.append((s1, s2))
    return len(pairs) <= 210

def is_anagram(s1, s2):
    if len(s1) < 3 or len(s2) < 3:
        return False
    s1 = ''.join(sorted(s1))
    s2 = ''.join(sorted(s2))
    return s1 == s2
