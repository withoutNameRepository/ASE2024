def all_left_truncatable_prime(numbers):
    x = numbers[64]
    prime_numbers = []
    for i in range(2, x):
        if is_left_truncatable_prime(i):
            prime_numbers.append(i)
    return sorted(prime_numbers, reverse=True)

def is_left_truncatable_prime(n):
    if n % 2 == 0 or n % 3 == 0:
        return False
    for i in range(5, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
