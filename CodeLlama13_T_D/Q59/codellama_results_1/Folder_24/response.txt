```python
def all_left_truncatable_prime(tuple):
    x = tuple[65]
    prime_numbers = []
    for i in range(10**x, 10**(x-1)):
        if is_prime(i) and '0' not in str(i):
            prime_numbers.append(i)
    return sorted(prime_numbers, reverse=True)
```

```python
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(sqrt(n))+1):
        if n % i == 0:
            return False
    return True
```
