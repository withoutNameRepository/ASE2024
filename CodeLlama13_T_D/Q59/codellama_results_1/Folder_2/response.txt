def is_left_truncatable_prime(n, numbers):
    if n in numbers:
        return False
    for i in range(len(str(n))):
        if str(n)[i] == '0':
            return False
    for i in range(1, len(str(n)) + 1):
        if not is_prime(int(str(n)[:-i])):
            return False
    return True

def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
