    # Check if n is divisible by any integer between 2 and sqrt(n)
    for i in range(2, int(sqrt(n)) + 1):
        if n % i == 0:
            return False
    
    # If n is not divisible by any integer between 2 and sqrt(n), it is prime
    return True

def is_left_truncatable(n):
