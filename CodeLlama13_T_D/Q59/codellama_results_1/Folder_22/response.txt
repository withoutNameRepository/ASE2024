def all_left_truncatable_prime(numbers):
    result = []
    for num in numbers:
        if is_left_truncatable_prime(num):
            result.append(num)
    return sorted(result)

def is_left_truncatable_prime(n):
    if n % 2 == 0 or n % 3 == 0 or n % 5 == 0:
        return False
    for i in range(1, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
