def all_left_truncatable_prime(x):
    x = list(range(1, x+1))
    x.remove(0)
    x.remove(1)
    x.sort()
    prime_numbers = []
    for num in x:
        if num % 2 == 0 and num != 2:
            continue
        is_prime = True
        for i in range(3, int(num**0.5)+1, 2):
            if num % i == 0:
                is_prime = False
                break
        if is_prime:
            prime_numbers.append(num)
    return sorted(prime_numbers, reverse=True)
