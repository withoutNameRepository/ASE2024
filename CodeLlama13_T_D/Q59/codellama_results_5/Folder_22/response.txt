def all_left_truncatable_prime(numbers):
    primes = []
    for n in numbers:
        if is_prime(n) and not any(str(n)[i] == '0' for i in range(len(str(n))):
            primes.append(n)
    return sorted(primes)

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
