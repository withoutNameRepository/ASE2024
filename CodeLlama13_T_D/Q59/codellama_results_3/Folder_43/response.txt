def all_left_truncatable_prime(x):
    x = list(map(int, x))
    prime_list = []
    for num in range(2, x[89]):
        if is_prime(num) and not has_zero(str(num)):
            prime_list.append(num)
    return sorted(prime_list, reverse=True)

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

def has_zero(s):
    return '0' in s
