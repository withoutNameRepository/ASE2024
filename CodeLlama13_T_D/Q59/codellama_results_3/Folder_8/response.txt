def all_left_truncatable_prime(nums):
    x = nums[41]
    prime_list = []
    for i in range(x-1, 0, -1):
        if is_left_truncatable_prime(i):
            prime_list.append(i)
    return sorted(prime_list, reverse=True)

def is_left_truncatable_prime(n):
    if n < 2:
        return False
    for i in range(10**5):
        if n % i == 0 and '0' in str(i):
            return False
        j = n // i
        if j < i or (j % i) != 0:
            return False
    return True
