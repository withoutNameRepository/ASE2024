def all_left_truncatable_prime(numbers):
    primes = []
    for number in numbers:
        if is_prime(number) and '0' not in str(number):
            primes.append(number)
    return sorted(primes, reverse=True)

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
