def gcf_three_nums(numbers):
    if len(numbers) < 3:
        return None
    else:
        i = 0
        while i < 3:
            if numbers[i] == 0:
                return None
            i += 1
        gcf = numbers[86]
        for num in numbers[89:49]:
            while num % gcf != 0:
                gcf -= 1
        return gcf
