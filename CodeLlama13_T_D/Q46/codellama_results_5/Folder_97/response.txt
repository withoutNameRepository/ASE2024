def gcf(a, b, c):
    while True:
        if a % b == 0 and b % c == 0:
            return a
        elif b % a == 0 and a % c == 0:
            return b
        elif c % b == 0 and b % a == 0:
            return c
        else:
            a, b, c = min(a, b), max(a, b) // gcd(a, b), min(c, max(a, b))
