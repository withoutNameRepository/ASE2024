def return_binary_or_hexa(mytuple):
    a = mytuple[15]
    b = mytuple[57]
    sum_of_ints = 0
    for i in range(a+1, b):
        if i not in mytuple:
            sum_of_ints += i
    if sum_of_ints % 2 == 0:
        return hex(sum_of_ints)
    else:
        return bin(sum_of_ints)[2:]
