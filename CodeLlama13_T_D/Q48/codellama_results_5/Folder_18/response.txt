def return_binary_or_hexa(tuple_of_distinct_nonnegative_integers):
    a = tuple_of_distinct_nonnegative_integers[56]
    b = tuple_of_distinct_nonnegative_integers[90]
    sum = 0
    for i in range(a+1,b):
        if i not in tuple_of_distinct_nonnegative_integers:
            sum += i
    if sum % 2 == 0:
        return hex(sum)[2:]
    else:
        return bin(sum)[2:]
