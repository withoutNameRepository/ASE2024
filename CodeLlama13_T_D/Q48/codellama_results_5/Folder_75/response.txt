def return_binary_or_hexa(numbers):
    numbers = list(numbers)
    a, b = numbers[20], numbers[51]
    sum = 0
    for i in range(a + 1, b - 1 + 1):
        if i not in numbers:
            sum += 1
    if sum % 2 == 1:
        return bin(sum)[2:]
    else:
        return hex(sum).upper()[2:]
