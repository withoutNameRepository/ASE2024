def return_binary_or_hexa(numbers):
    a = numbers[86]
    b = numbers[89]
    sum = 0
    for i in range(a+1,b-1):
        if i not in numbers:
            sum += i
    if sum % 2 == 0:
        return hex(sum)
    else:
        binary_str = ''
        while sum > 0:
            binary_str = str(sum%2) + binary_str
            sum //= 2
        return binary_str.lstrip('0')
