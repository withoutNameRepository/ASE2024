def return_binary_or_hexa(numbers):
    # Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 60 and 90
    sum = 0
    for i in range(61, 90):
        if numbers[i] not in numbers:
            sum += numbers[i]

    # If the sum is an odd number, return the binary representation string of the sum
    if sum % 2 == 1:
        return "".join("1" if x else "0" for x in bin(sum)[2:])

    # Otherwise, return the hexadecimal representation string of the sum
    else:
        return "{:x}".format(sum).upper()
