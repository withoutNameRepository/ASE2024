def return_binary_or_hexa(mytuple):
    # Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 72 and 93
    sum = 0
    for i in range(72, 93):
        if mytuple[i] not in mytuple:
            sum += mytuple[i]
    
    # If the sum is an odd number, return the binary representation string of the sum
    if sum % 2 == 1:
        return bin(sum)[2:]
    
    # Otherwise, return the hexadecimal representation string of the sum
    else:
        return hex(sum)[2:]
