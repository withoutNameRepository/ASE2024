def perfect_numbers(my_list):
    perfect_nums = []
    for i in range(len(my_list)):
        num = my_list[i]

def is_perfect_number(n):
    root = int(math.sqrt(n))
    sum_of_divisors = 1
    for i in range(2, root+1):
        if n % i == 0:
            sum_of_divisors += i + n//i
    return sum_of_divisors == 2*n
