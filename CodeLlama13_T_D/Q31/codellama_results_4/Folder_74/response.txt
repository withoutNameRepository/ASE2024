def perfect_numbers():
    perfect_nums = []
    for i in range(1, 10**6):
        if sum(divisors(i)) == i:
            perfect_nums.append(i)
    return perfect_nums

def divisors(n):
    return [d for d in range(1, int(n**0.5)+1) if n % d == 0]
