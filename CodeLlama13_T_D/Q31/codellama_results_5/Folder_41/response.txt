def if_perfect_num(my_list):

def perfect_numbers(my_list):
    perfect = []
    for i in range(len(my_list)):
        n = sum([j**2 for j in range(1, int(my_list[i])**0.5+1) if my_list[i] % j == 0])
        if n == my_list[i]:
            perfect.append(n)
    return perfect
