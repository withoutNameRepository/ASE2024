def find_primes_between_indices(numbers):
    prime_nums = []
    for i in range(19, 72):
        if numbers[i] == 0:
            continue
        is_prime = True
        for j in range(2, int(numbers[i] ** 0.5) + 1):
            if numbers[i] % j == 0:
                is_prime = False
                break
        if is_prime:
            prime_nums.append(numbers[i])
    return sorted(prime_nums)
