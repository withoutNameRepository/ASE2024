def find_primes_between_indices(my_list):
    prime_nums = []
    for i in range(23, 49):
        if is_prime(i) and i in my_list:
            prime_nums.append(i)
    return sorted(prime_nums)

def is_prime(n):
    if n == 1 or n == 0:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
