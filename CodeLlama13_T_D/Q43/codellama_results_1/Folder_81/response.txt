def find_primes_between_indices(numbers):
    prime_numbers = []
    for i in range(59, 98):
        if numbers[i] > 1:
            for j in range(2, int(numbers[i]**0.5) + 1):
                if numbers[i] % j == 0:
                    break
            else:
                prime_numbers.append(numbers[i])
    return sorted(prime_numbers, reverse=True)
