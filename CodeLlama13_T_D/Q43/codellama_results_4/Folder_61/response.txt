def find_primes_between_indices(nums):
    prime_numbers = []
    for i in range(3, 9):
        if is_prime(nums[i]):
            prime_numbers.append(nums[i])
    return sorted(prime_numbers)

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
