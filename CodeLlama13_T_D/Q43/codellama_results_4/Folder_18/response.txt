def find_primes_between_indices(numbers):
    primes = []
    for i in range(430, 806):
        if is_prime(numbers[i]):
            primes.append(numbers[i])
    return sorted(primes)

def is_prime(n):
    if n == 1 or n % 2 == 0 and n != 2:
        return False
    for i in range(3, int(n ** 0.5) + 1, 2):
        if n % i == 0:
            return False
    return True
