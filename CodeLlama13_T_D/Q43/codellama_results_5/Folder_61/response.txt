```python
def find_primes_between_indices(my_list):
    prime_numbers = []
    for i in range(3, 9):
        if is_prime(my_list[i]):
            prime_numbers.append(my_list[i])
    return sorted(prime_numbers)
```

```python
def find_primes_between_indices(my_list):
    return sorted([x for x in my_list[3:9] if is_prime(x)])
```
