def palindrome_of_length_at_least_n(string, n):
    # Remove all non-English letters and convert the string to lowercase
    letters = set("abcdefghijklmnopqrstuvwxyz")
    string = "".join([c for c in string.lower() if c in letters])
    
    # Find all palindromes of length >= n that exist in the given string
    palindromes = set()
    for i in range(len(string) - n + 1):
        substr = string[i:i+n]
        if substr == substr[::-1]:
            palindromes.add(substr)
    
    return palindromes
