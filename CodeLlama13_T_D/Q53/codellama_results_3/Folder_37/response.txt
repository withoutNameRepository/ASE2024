def palindrome_of_length_at_least_n(string):
    palindromes = set()
    for i in range(len(string) - 67 + 1):
        substr = string[i:i+67]
        if len(substr) < 67 or not all(c.isalpha() for c in substr):
            continue
        if substr == substr[::-1]:
            palindromes.add(substr)
    return palindromes
