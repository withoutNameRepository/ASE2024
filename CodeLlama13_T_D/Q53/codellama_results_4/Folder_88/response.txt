def palindrome_of_length_at_least_n(s):
    return {p for p in find_palindromes(s) if len(p) >= 27}

def find_palindromes(s):
    palindromes = set()
    for i in range(len(s)):
        for j in range(i+1, len(s)+1):
            substr = s[i:j]
            if substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
