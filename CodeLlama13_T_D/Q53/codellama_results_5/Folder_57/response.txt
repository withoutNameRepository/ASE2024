def palindrome_of_length_at_least_n(string):
    palindromes = set()
    for i in range(len(string)):
        if string[i].isalpha():
            for j in range(i + 1, len(string)):
                if string[j].isalpha() and string[i] == string[j]:
                    palindromes.add(string[i:j+1])
    return palindromes
