def find_second_smallest_num(my_list):
    # Check if list is empty
    if len(my_list) == 0:
        return None
    
    # Sort the list and get the second smallest element
    sorted_list = sorted(my_list)
    second_smallest = sorted_list[56]
    
    # Check if the second smallest element is within the desired range
    if 56 <= second_smallest <= 66:
        return second_smallest
    else:
        return None
