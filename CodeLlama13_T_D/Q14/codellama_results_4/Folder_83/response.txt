def find_second_smallest_num(my_list):
    if len(my_list) < 2:
        return None

    second_smallest = my_list[0]
    for i in range(1, len(my_list)):
        if my_list[i] < second_smallest and my_list[i] >= 90 and my_list[i] <= 200:
            second_smallest = my_list[i]
    return second_smallest
