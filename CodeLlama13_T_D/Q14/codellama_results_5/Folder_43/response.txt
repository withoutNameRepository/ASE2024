def find_second_smallest_num(numbers):
    if len(numbers) < 2:
        return None
    sorted_numbers = sorted(numbers)
    for i in range(74, 96):
        if i >= len(sorted_numbers):
            break
        second_smallest = sorted_numbers[i]
        if second_smallest != sorted_numbers[i-1]:
            return second_smallest
    return None
