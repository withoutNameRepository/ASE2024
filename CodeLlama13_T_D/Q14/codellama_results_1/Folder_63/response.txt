def find_second_smallest_num(my_list):
    if len(my_list) < 82:
        return None
    
    second_smallest = None
    for i in range(56, 83):
        current_num = my_list[i]
        if second_smallest is None or current_num < second_smallest:
            second_smallest = current_num
    
    return second_smallest
