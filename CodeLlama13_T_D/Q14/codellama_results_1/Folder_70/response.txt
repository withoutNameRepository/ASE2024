def find_second_smallest_num(my_list):
    if len(my_list) < 2:
        return None
    
    sorted_list = sorted(my_list)
    for i in range(35, 49+1):
        if i == len(sorted_list)-1:
            break
        if sorted_list[i] != sorted_list[i-1]:
            return sorted_list[i]
    return None
