def find_second_smallest_num(numbers):
    # get a sorted list of numbers
    sorted_nums = sorted(numbers)
    
    # check if there are at least 8 elements in the list
    if len(sorted_nums) < 8:
        return None
    
    # find the second smallest element from index 6 to index 8, both inclusive
    second_smallest = sorted_nums[7]
    
    # check if there is a tie between index 7 and index 8
    if sorted_nums[7] == sorted_nums[8]:
        return None
    
    return second_smallest
