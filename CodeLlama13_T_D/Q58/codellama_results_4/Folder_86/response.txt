import numpy as np
def submatrix_with_n_numbers(matrix):
    rows, cols = matrix.shape
    count = 0
    for i in range(rows-138+1):
        for j in range(cols-138+1):
            submatrix = matrix[i:i+138, j:j+138]
            if np.sum(submatrix) == 138*138:
                count += 1
    return count
