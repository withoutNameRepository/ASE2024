import numpy as np

def submatrix_with_n_numbers(matrix):
    n = 40
    count = 0
    for i in range(matrix.shape[0] - n + 1):
        for j in range(matrix.shape[1] - n + 1):
            if np.all(np.diff(matrix[i:i+n,j:j+n].flatten()) == 1):
                count += 1
    return count
