import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize a counter for the number of submatrices found
    count = 0

    # Loop through each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Check if the current element is 144
            if matrix[i, j] == 144:
                # Find all submatrices that start from the current position
                submatrices = []
                for k in range(i, matrix.shape[0]):
                    for l in range(j, matrix.shape[1]):
                        if matrix[k, l] == 144:
                            submatrices.append((k, l))

                # Check if any of the found submatrices have 144 integers each
                for submatrix in submatrices:
                    if len(submatrix) == 144:
                        count += 1

    return count
