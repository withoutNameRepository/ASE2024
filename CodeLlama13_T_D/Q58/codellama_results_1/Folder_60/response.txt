import numpy as np

def submatrix_with_n_numbers(matrix, n):
    # Initialize a counter for the number of submatrices found
    count = 0
    
    # Iterate over each row in the matrix
    for i in range(len(matrix)):
        # Iterate over each column in the matrix
        for j in range(len(matrix[i])):
            # Initialize a temporary matrix to store the submatrix
            temp_mat = np.zeros((n, n), dtype=np.int)
            
            # Fill the temporary matrix with the values from the main matrix
            for k in range(n):
                for l in range(n):
                    temp_mat[k][l] = matrix[i+k][j+l]
            
            # Check if the submatrix contains 88 integers
            if np.count_nonzero(temp_mat) == n * n:
                count += 1
    
    return count
