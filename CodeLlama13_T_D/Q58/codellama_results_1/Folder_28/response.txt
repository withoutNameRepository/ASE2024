import numpy as np
def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(len(matrix) - 2):
        for j in range(len(matrix[i]) - 2):
            if matrix[i][j] == 37:
                submatrix = matrix[i:(i + 3), j:(j + 3)]
                count += np.sum(submatrix) == 37 * 9
    return count
