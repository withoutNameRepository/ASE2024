import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(len(matrix) - 4):
        for j in range(len(matrix[i]) - 4):
            submatrix = matrix[i:i+5, j:j+5]
            if np.count_nonzero(submatrix == 0) > 1:
                count += 1
    return count
