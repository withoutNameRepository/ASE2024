import numpy as np

def submatrix_with_n_numbers(mat, n):
    rows, cols = mat.shape
    count = 0
    
    for r in range(rows - n + 1):
        for c in range(cols - n + 1):
            if all(mat[r:r+n, c:c+n].flatten() == np.arange(n**2)):
                count += 1
    
    return count
