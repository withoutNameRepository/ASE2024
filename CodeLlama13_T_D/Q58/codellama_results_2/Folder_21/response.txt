import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(len(matrix) - 139 + 1):
        for j in range(len(matrix[0]) - 139 + 1):
            submatrix = matrix[i:i+139, j:j+139]
            if np.sum(submatrix == 139) == 139:
                count += 1
    return count
