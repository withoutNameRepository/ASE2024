import numpy as np

def submatrix_with_n_numbers(matrix):
    num_rows, num_cols = matrix.shape
    count = 0
    for i in range(num_rows - 5):
        for j in range(num_cols - 5):
            submatrix = matrix[i:i+6, j:j+6]
            if np.count_nonzero(submatrix == 63) == 63:
                count += 1
    return count
