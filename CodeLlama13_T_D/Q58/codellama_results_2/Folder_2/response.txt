import numpy as np

def submatrix_with_n_numbers(mat):
    # Initialize a variable to store the count of submatrices with 107 integers
    count = 0
    
    # Iterate over each row of the matrix
    for i in range(len(mat)):
        # Iterate over each column of the matrix
        for j in range(len(mat[0])):
            # If the current element is equal to 1, increment the count by 1
            if mat[i][j] == 1:
                count += 1
                
    # Return the count of submatrices with 107 integers
    return count
