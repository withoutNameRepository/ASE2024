import numpy as np

def submatrix_with_n_numbers(mat, n):
    rows = len(mat)
    cols = len(mat[0])
    count = 0
    
    for i in range(rows - n + 1):
        for j in range(cols - n + 1):
            sub = mat[i:i+n, j:j+n]
            if np.all(sub == 121):
                count += 1
                
    return count
