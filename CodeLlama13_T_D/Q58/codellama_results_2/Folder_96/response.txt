import numpy as np

def submatrix_with_n_numbers(matrix, n):
    count = 0
    for i in range(len(matrix) - n + 1):
        for j in range(len(matrix[i]) - n + 1):
            submatrix = matrix[i:i+n, j:j+n]
            if np.sum(submatrix == n) == (n * n):
                count += 1
    return count
