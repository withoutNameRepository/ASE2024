import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(len(matrix) - 107 + 1):
        for j in range(len(matrix[i]) - 107 + 1):
            submatrix = matrix[i:i+107, j:j+107]
            if np.count_nonzero(submatrix) == 107:
                count += 1
    return count
