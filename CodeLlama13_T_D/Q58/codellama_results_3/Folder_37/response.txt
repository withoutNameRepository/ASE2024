import numpy as np
def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(matrix.shape[0] - 63):
        for j in range(matrix.shape[1] - 63):
            submatrix = matrix[i:i + 64, j:j + 64]
            if np.all(submatrix == 64):
                count += 1
    return count
