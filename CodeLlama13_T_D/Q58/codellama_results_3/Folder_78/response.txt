import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize a count variable to keep track of the number of submatrices with 91 integers
    count = 0
    
    # Iterate over each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # If the current element is not 0, skip it
            if matrix[i][j] == 0:
                continue
            
            # Initialize a temporary matrix to store the submatrix with 91 integers
            temp_matrix = np.zeros((91, 91), dtype=np.int64)
            
            # Iterate over each element in the submatrix and add it to the temporary matrix if it is not 0
            for k in range(i, i+91):
                for l in range(j, j+91):
                    if matrix[k][l] != 0:
                        temp_matrix[k-i][l-j] = matrix[k][l]
            
            # If the temporary matrix is not empty, increment the count variable
            if np.count_nonzero(temp_matrix) == 91*91:
                count += 1
    
    return count
