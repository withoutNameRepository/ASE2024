import numpy as np

def submatrix_with_n_numbers(matrix, n):
    rows = matrix.shape[0]
    cols = matrix.shape[1]
    count = 0
    
    for i in range(rows - n + 1):
        for j in range(cols - n + 1):
            sub_mat = np.array([matrix[i:i+n,j:j+n]])
            if (sub_mat == 69).all():
                count += 1
    
    return count
