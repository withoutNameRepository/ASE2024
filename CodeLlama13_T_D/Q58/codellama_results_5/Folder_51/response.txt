import numpy as np

def submatrix_with_n_numbers(mat):
    n = 95
    rows, cols = mat.shape
    count = 0
    for i in range(rows - n + 1):
        for j in range(cols - n + 1):
            if np.count_nonzero(np.diagonal(mat[i:i+n,j:j+n])) == n * n:
                count += 1
    return count
