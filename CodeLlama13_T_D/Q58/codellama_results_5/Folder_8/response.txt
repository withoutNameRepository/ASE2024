import numpy as np
def submatrix_with_n_numbers(matrix):
    rows, cols = matrix.shape
    count = 0
    for i in range(rows-28):
        for j in range(cols-28):
            submat = matrix[i:i+29,j:j+29]
            if np.sum(submat) == 29*29:
                count += 1
    return count
