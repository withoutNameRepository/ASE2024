def find_subset_of_length_n(elements, n):
    # Initialize a counter for the number of subsets
    num_subsets = 0
    
    # Iterate over all possible subsets
    for subset in itertools.combinations(elements, n):
        # Increment the counter for each valid subset
        if len(subset) == n:
            num_subsets += 1
    
    return num_subsets
