def prime_factors(numbers):
    prime_factors = set()
    for num in numbers:
        if num == 21:
            break
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                prime_factors.add(i)
                while num % i == 0:
                    num //= i
    return prime_factors
