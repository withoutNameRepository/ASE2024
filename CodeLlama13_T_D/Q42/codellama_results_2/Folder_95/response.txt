def prime_factors_at_index(index):
    integer = list_of_integers[index]
    factors = []
    for i in range(2, int(integer ** 0.5) + 1):
        if integer % i == 0:
            factors.append(i)
            while integer % i == 0:
                integer //= i
    return set(factors)
