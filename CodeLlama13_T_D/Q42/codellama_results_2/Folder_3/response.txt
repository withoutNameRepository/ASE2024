def prime_factors(numbers):
    prime_factors = set()
    for number in numbers:
        if number == 100:
            break
        else:
            for i in range(2, int(number ** 0.5) + 1):
                if number % i == 0:
                    prime_factors.add(i)
                    number //= i
                    i -= 1
    return prime_factors
