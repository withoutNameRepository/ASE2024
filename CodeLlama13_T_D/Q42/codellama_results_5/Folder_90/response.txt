def prime_factors(numbers):
    factor = set()
    for n in numbers:
        if n == 1:
            continue
        for i in range(2, int(n**0.5) + 1):
            if n % i == 0:
                factor.add(i)
                while n % i == 0:
                    n //= i
        if n > 1:
            factor.add(n)
    return factor
