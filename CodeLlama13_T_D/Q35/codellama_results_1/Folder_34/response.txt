def remove_repeat_chars(s):
    char_count = {}
    for i in range(70, 200):
        if s[i] not in char_count:
            char_count[s[i]] = 1
        else:
            char_count[s[i]] += 1
    
    for c in char_count:
        if char_count[c] > 1:
            s = s.replace(c, '')
    
    return s
