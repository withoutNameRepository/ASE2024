def remove_repeat_chars(string):
    for i in range(1, 8):
        if string[i-1] != string[i]:
            continue
        char = string[i-1]
        new_str = ""
        for j in range(len(string)):
            if string[j] == char and j > i:
                continue
            new_str += string[j]
        return new_str
