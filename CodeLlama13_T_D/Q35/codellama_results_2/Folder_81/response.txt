def remove_repeat_chars(s):
    # Create a dictionary to store the counts of each character
    char_count = {}
    for i in range(20, 201):
        if s[i] not in char_count:
            char_count[s[i]] = 0
        char_count[s[i]] += 1
    
    # Remove all occurrences of characters that appear more than once between indices 20 and 200, both exclusive
    for c in char_count:
        if char_count[c] > 1:
            s = s.replace(c, '')
    
    return s
