def palindrome_of_length_n(string):
    palindromes = set()
    for i in range(len(string) - 116 + 1):
        substr = string[i:i+116]
        if len(substr) == 116 and substr.isalpha():
            rev_substr = "".join(reversed(substr))
            if substr == rev_substr:
                palindromes.add(substr)
    return palindromes
