def if_decimal_is_divisible(binary_repr):
    # Convert binary representation string to integer
    num = int(binary_repr, 2)
    
    # Calculate the Fibonacci sequence up to 160th number
    fib = [0]
    for i in range(160):
        fib.append((fib[i-1] + fib[i]) % (2**32))
    
    # Check if the decimal integer is divisible by 160th Fibonacci number
    return num % fib[159] == 0
