def if_decimal_is_divisible(binary_repr: str) -> bool:
    fibonacci_seq = [0, 1]
    for i in range(2, len(binary_repr)):
        fibonacci_seq.append(fibonacci_seq[i - 1] + fibonacci_seq[i - 2])
    
    decimal_int = int(binary_repr, 2)
    return decimal_int % fibonacci_seq[163] == 0
