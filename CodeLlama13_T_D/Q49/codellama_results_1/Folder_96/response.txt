def if_decimal_is_divisible(binary):
    # Convert the binary string to an integer
    num = int(binary, 2)
    
    # Calculate the Fibonacci sequence up to 47th number
    fib = [0]
    for i in range(1, 47 + 1):
        if i == 1:
            fib.append(i)
        else:
            fib.append(fib[i - 1] + fib[i - 2])
    
    # Check if the decimal integer is divisible by 47th number in Fibonacci sequence
    return num % fib[46] == 0
