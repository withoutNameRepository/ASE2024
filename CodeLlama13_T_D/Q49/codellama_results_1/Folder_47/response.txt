def if_decimal_is_divisible(binary_string):
    binary_list = list(map(int, binary_string))
    decimal_integer = 0
    for i in range(len(binary_list)):
        decimal_integer += binary_list[i] * (2 ** i)
    fibonacci_sequence = [0, 1]
    while fibonacci_sequence[-1] < decimal_integer:
        fibonacci_sequence.append(fibonacci_sequence[-2] + fibonacci_sequence[-1])
    return (decimal_integer % fibonacci_sequence[93] == 0)
