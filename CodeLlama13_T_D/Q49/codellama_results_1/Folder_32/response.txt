def if_decimal_is_divisible(binary_repr):
    binary_repr = binary_repr[::-1]  # reverse the string
    decimal_int = int(binary_repr, 2)  # convert to decimal integer
    fibonacci_seq = [0, 1]
    for i in range(2, 178):
        fibonacci_seq.append((fibonacci_seq[i-1] + fibonacci_seq[i-2]) % (1 << 32))
    return decimal_int % fibonacci_seq[177] == 0
