def if_decimal_is_divisible(binary_string):
    # Convert the binary string to a decimal number
    decimal = int(binary_string, 2)

    # Initialize a variable to keep track of the Fibonacci sequence
    fib = [0]

    # Loop until the 115th number in the Fibonacci sequence is reached
    for i in range(114):
        fib.append(fib[i] + fib[i-1])

    # Check if the decimal number is divisible by the 115th number in the Fibonacci sequence
    return (decimal % fib[114]) == 0
