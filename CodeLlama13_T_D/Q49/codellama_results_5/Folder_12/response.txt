def if_decimal_is_divisible(binary_string):
    # Convert the binary string to an integer
    decimal = int(binary_string, 2)
    
    # Calculate the 66th number that occurs in the Fibonacci sequence
    fibonacci = [0, 1]
    for i in range(64):
        fibonacci.append(fibonacci[i] + fibonacci[i+1])
    
    # Check if the decimal is divisible by the 66th number that occurs in the Fibonacci sequence
    return decimal % fibonacci[65] == 0
