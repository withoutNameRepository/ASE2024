def if_decimal_is_divisible(binary_string):
    # Convert the binary string to an integer
    decimal = int(binary_string, 2)
    
    # Calculate the 38th Fibonacci number using a recursive formula
    fib38 = lambda n: (n * (n + 1)) // 2 if n > 0 else 0
    
    # Check if the decimal is divisible by the 38th Fibonacci number
    return decimal % fib38(38) == 0
