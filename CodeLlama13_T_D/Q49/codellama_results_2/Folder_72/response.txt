def if_decimal_is_divisible(binary_str):
    binary_num = int(binary_str, 2)
    fibonacci_seq = []
    a, b = 0, 1
    while True:
        fibonacci_seq.append(a)
        if len(fibonacci_seq) == 175:
            break
        a, b = b, a + b
    return binary_num in [x * 2 for x in fibonacci_seq]
