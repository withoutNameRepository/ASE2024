def if_decimal_is_divisible(binary_string):
    # Convert the binary string to an integer
    num = int(binary_string, 2)
    
    # Get the 9th number in the Fibonacci sequence
    fib9 = 21
    
    # Check if the decimal integer corresponding to the given string is divisible by the 9th number
    return num % fib9 == 0
