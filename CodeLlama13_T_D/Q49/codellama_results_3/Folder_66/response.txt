def if_decimal_is_divisible(binary_str):
    # Convert the binary string to a decimal integer
    dec = int(binary_str, 2)

    # Calculate the 120th number in the Fibonacci sequence
    fib = [0, 1]
    for i in range(120):
        fib.append((fib[i - 1] + fib[i]) % 10)

    # Check if the decimal integer is divisible by the 120th Fibonacci number
    return dec % fib[120] == 0
