def if_decimal_is_divisible(binary_repr):
    # Convert binary representation string to integer
    decimal = int(binary_repr, 2)

    # Get the Fibonacci sequence up to 180th number
    fibonacci_seq = [0] + [fibonacci(i) for i in range(179)]

    # Check if the decimal integer is divisible by the 180th number
    return decimal % fibonacci_seq[179] == 0
