def if_decimal_is_divisible(binary_string):
    # Convert the binary string to an integer
    num = int(binary_string, 2)
    
    # Calculate the 179th Fibonacci number
    fib179 = 1346269
    
    # Check if the decimal integer corresponding to the given binary string is divisible by the 179th Fibonacci number
    return num % fib179 == 0
