def if_decimal_is_divisible(binary_repr):
    binary_list = list(map(int, binary_repr))
    fibonacci_sequence = [0]
    current = 1
    while current < len(binary_list):
        fibonacci_sequence.append(current)
        current += fibonacci_sequence[-2]
    
    return any(i == 128 for i in fibonacci_sequence if binary_list[i])
