def sum_even_ints_inclusive(my_list):
    # initialize a variable to store the sum of even integers in the range
    total = 0
    
    # loop through each element in the list
    for i, num in enumerate(my_list):
        # check if the current element is an even integer and is within the specified range
        if num % 2 == 0 and 639 <= i <= 975:
            # add the current element to the total
            total += num
