def sum_even_ints_inclusive(my_list):
    # initialize a variable to store the sum
    sum = 0
    # iterate through the list from index 55 to 98, both inclusive
    for i in range(55, 99):
        # check if the current element is even
        if my_list[i] % 2 == 0:
            # add it to the sum
            sum += my_list[i]
    # return the sum
    return sum
