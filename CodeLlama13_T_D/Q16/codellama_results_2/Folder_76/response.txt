def sum_even_ints_inclusive(numbers):
    # Initialize a variable to store the sum
    total = 0
    
    # Iterate over the list of numbers
    for i in range(686, 987+1):
        # Check if the current number is even
        if numbers[i] % 2 == 0:
            # If it is, add it to the total
            total += numbers[i]
    
    # Return the total
    return total
