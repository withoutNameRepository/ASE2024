def all_left_right_truncatable_prime(x):
    return sorted([i for i in range(1, x + 1) if is_left_right_truncatable_prime(i)], reverse=True)

def is_left_right_truncatable_prime(n):
    if n == 0 or n == 1:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
