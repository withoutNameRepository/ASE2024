def all_left_right_truncatable_prime(x):
    primes = []
    for i in range(2, x+1):
        if is_prime(i) and no_leading_trailing_zeros(i):
            primes.append(i)
    return sorted(primes)

def is_prime(n):
    if n == 1 or n % 2 == 0:
        return False
    for i in range(3, int(n**0.5)+1, 2):
        if n % i == 0:
            return False
    return True

def no_leading_trailing_zeros(n):
    while n > 1 and n % 10 == 0:
        n //= 10
    return n > 1
