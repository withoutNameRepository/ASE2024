def all_left_right_truncatable_prime(x):
    result = []
    for i in range(2, x+1):
        if is_prime(i) and len(str(i)) > 1:
            left = int(str(i)[1:])
            right = int(str(i)[:-1])
            if is_prime(left) and is_prime(right):
                result.append(i)
    return sorted(result)
