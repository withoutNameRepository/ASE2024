def all_left_right_truncatable_prime(numbers):
    left_right_truncatable_primes = []
    for number in numbers:
        if is_left_and_right_truncatable_prime(number):
            left_right_truncatable_primes.append(number)
    return sorted(left_right_truncatable_primes, reverse=True)

def is_left_and_right_truncatable_prime(number):
    if number < 10:
        return False
    for i in range(1, int(len(str(number)) / 2)):
        if str(number)[i] == '0':
            return False
    for i in range(int(len(str(number)) / 2), len(str(number))):
        if str(number)[:-i] % 10 == 0:
            return False
    for i in range(int(len(str(number)) / 2)):
        if str(number)[:-i - 1] % 10 == 0:
            return False
    return True
