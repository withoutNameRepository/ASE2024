def all_left_right_truncatable_prime(nums):
    primes = []
    for num in nums:
        if is_prime(num) and not str(num).startswith('0'):
            primes.append(num)
    return primes

def is_prime(n):
    if n <= 1:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
