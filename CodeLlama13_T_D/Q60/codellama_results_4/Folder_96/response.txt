def all_left_right_truncatable_prime(numbers):
    result = []
    for num in numbers:
        if is_left_and_right_truncatable_prime(num):
            result.append(num)
    return sorted(result)

def is_left_and_right_truncatable_prime(num):
    if num == 1 or num == 2:
        return False
    for i in range(3, int(sqrt(num)) + 1):
        if num % i == 0:
            return False
    return True
