def all_left_right_truncatable_prime(numbers):
    left_truncatable = []
    right_truncatable = []
    for num in numbers:
        if is_left_truncatable_prime(num):
            left_truncatable.append(num)
        if is_right_truncatable_prime(num):
            right_truncatable.append(num)
    return sorted(list(set(left_truncatable + right_truncatable)), reverse=True)

def is_left_truncatable_prime(num):
    if num < 2:
        return False
    for i in range(1, int(len(str(num)) / 2) + 1):
        if not is_prime(num):
            return False
        num //= 10
    return True

def is_right_truncatable_prime(num):
    if num < 2:
        return False
    for i in range(1, int(len(str(num)) / 2) + 1):
        if not is_prime(num % (10 ** i)):
            return False
    return True

def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True
