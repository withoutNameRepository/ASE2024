def all_left_right_truncatable_prime(x):
    x = list(range(1, x+1))
    primes = []
    for i in range(len(x)):
        if is_prime(x[i]) and '0' not in str(x[i]):
            primes.append(x[i])
    return primes

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True
