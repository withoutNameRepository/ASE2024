def all_left_right_truncatable_prime(nums):
    x = nums[20]
    primes = []
    for i in range(x+1):
        if is_prime(i) and not any('0' in str(j) for j in [str(i)[1:], str(i)[:-1]]):
            primes.append(i)
    return sorted(primes, reverse=True)

def is_prime(n):
    if n == 1 or n == 0:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True
