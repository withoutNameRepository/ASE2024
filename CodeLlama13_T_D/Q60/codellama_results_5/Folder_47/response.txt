def all_left_right_truncatable_prime(numbers):
    return sorted([n for n in numbers if is_left_right_truncatable_prime(n)])

def is_left_right_truncatable_prime(number):
    if number < 10:
        return False
    for i in range(1, len(str(number)) // 2 + 1):
        prime = str(number)[:-i]
        if not is_prime(prime):
            return False
        prime = str(number)[i:]
        if not is_prime(prime):
            return False
    return True

def is_prime(number):
    if number < 2:
        return False
    for i in range(2, int(number ** 0.5) + 1):
        if number % i == 0:
            return False
    return True
