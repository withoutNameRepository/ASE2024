def all_left_right_truncatable_prime(x):
    prime_numbers = []
    for i in range(2, x+1):
        if is_prime(i) and not contains_zero(str(i)):
            prime_numbers.append(i)
    
    left_right_truncatable_primes = []
    for p in prime_numbers:
        if is_left_and_right_truncatable_prime(p):
            left_right_truncatable_primes.append(p)
    
    return sorted(left_right_truncatable_primes)

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

def contains_zero(s):
    return '0' in s

def is_left_and_right_truncatable_prime(p):
    str_p = str(p)
    for i in range(1, len(str_p)):
        if not is_prime(int(str_p[:i])) or not is_prime(int(str_p[i:])):
            return False
    return True
