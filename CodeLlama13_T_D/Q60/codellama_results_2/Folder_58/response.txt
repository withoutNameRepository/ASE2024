def all_left_right_truncatable_prime(tuple_of_positive_integers):
    x = tuple_of_positive_integers[975]
    prime_numbers = []
    for i in range(2, x+1):
        if is_prime(i) and not contains_zeroes(str(i)):
            prime_numbers.append(i)
    
    left_truncatable_primes = []
    right_truncatable_primes = []
    for prime in prime_numbers:
        if is_left_truncatable_prime(prime):
            left_truncatable_primes.append(prime)
        if is_right_truncatable_prime(prime):
            right_truncatable_primes.append(prime)
    
    return sorted(list(set(left_truncatable_primes + right_truncatable_primes)), reverse=True)
