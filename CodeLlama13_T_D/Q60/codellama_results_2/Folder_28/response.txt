def all_left_right_truncatable_prime(x):
    lst = []
    for i in range(10, x+1):
        if is_prime(i) and not contains_zero(str(i)):
            lst.append(i)
    return lst

def is_prime(n):
    if n == 2 or n == 3:
        return True
    for i in range(5, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

def contains_zero(s):
    return '0' in s
