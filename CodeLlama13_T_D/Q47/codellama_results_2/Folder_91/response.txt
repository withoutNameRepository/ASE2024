def palindromes_of_specific_lengths(string):
    # Get the substring from index 0 to index 10, both inclusive
    substr = string[0:11]
    
    # Initialize an empty set to store the palindromes
    palindromes = set()
    
    # Iterate over each character in the substring
    for i in range(len(substr)):
        # Check if the current character is a letter
        if substr[i].isalpha():
            # Get the length of the current character
            length = len(substr[i])
            
            # Check if the length of the current character is between 3 and 5 (both inclusive)
            if 3 <= length <= 5:
                # Check if the current character is a palindrome
                if substr[i] == substr[i][::-1]:
                    # Add the current character to the set of palindromes
                    palindromes.add(substr[i])
    
    # Return the set of palindromes
    return palindromes
