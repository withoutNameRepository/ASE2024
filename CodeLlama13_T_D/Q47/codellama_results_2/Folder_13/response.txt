def palindromes_of_specific_lengths(string):
    substring = string[13:96]
    palindromes = set()
    for i in range(57, 61):
        for j in range(len(substring) - i + 1):
            if substring[j:j+i] == substring[j+i-1::-1]:
                palindromes.add(substring[j:j+i])
    return palindromes
