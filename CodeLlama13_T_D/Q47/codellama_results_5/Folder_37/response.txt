def palindromes_of_specific_lengths(string):
    # Find all palindromes of lengths between 3 and 5 (both inclusive)
    # that occur in the substring from index 0 to index 9, both inclusive
    palindromes = set()
    for i in range(3, 6):
        for j in range(len(string) - i + 1):
            substr = string[j:j+i]
            if substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
