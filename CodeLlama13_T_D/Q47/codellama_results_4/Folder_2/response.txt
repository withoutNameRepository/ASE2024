def palindromes_of_specific_lengths(string):
    palindromes = set()
    for i in range(10, 59 + 1):
        for j in range(i, min(len(string), i + 36) + 1):
            substr = string[i:j]
            if len(substr) >= 18 and len(substr) <= 36 and substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
