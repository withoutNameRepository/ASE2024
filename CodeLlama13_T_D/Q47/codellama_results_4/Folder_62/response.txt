def palindromes_of_specific_lengths(string):
    # Initialize empty set for palindromes
    palindromes = set()

    # Iterate over the substring from index 26 to 87, both inclusive
    for i in range(26, 88):
        # Extract substring of length 52-56 (both inclusive) starting at current index
        substr = string[i:i+5]

        # Check if the substring is a palindrome
        if substr == substr[::-1]:
            # Add the palindrome to the set
            palindromes.add(substr)

    # Return the set of palindromes
    return palindromes
