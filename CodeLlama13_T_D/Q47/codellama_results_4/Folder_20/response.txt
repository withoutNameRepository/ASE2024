def palindromes_of_specific_lengths(string):
    substring = string[20:75]
    palindromes = set()
    for i in range(36, 43):
        for j in range(len(substring) - i + 1):
            if substring[j:j+i].lower() == substring[j+i-1::-1]:
                palindromes.add(substring[j:j+i])
    return palindromes
