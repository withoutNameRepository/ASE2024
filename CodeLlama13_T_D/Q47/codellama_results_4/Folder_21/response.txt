def palindromes_of_specific_lengths(string):
    # Create a set to store the palindromes
    palindromes = set()
    
    # Iterate over the substring from index 100 to 295, both inclusive
    for i in range(100, 296):
        # Get the substring starting at index i and ending at index 295
        substring = string[i:295]
        
        # Iterate over the lengths between 136 to 160 (both inclusive)
        for length in range(136, 161):
            # Get the palindrome with the current length
            palindrome = substring[:length]
            
            # Check if the palindrome is a valid palindrome
            if palindrome == palindrome[::-1]:
                # Add the palindrome to the set
                palindromes.add(palindrome)
    
    return palindromes
