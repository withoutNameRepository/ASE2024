def palindromes_of_specific_lengths(string):
    substring = string[0:5]
    palindromes = set()
    for i in range(3, 6):
        for j in range(len(substring) - i + 1):
            substr = substring[j:j+i]
            if substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
