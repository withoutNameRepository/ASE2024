def palindromes_of_specific_lengths(string):
    palindromes = set()
    for i in range(21, 63):
        for j in range(i + 1, min(len(string) - i, 34)):
            substring = string[i:i+j]
            if substring == substring[::-1].lower():
                palindromes.add(substring)
    return palindromes
