def palindromes_of_specific_lengths(string):
    palindromes = set()
    for i in range(2, 8 + 1):
        for j in range(i + 3 - 1, i + 4):
            substring = string[i:j]
            if substring == substring[::-1]:
                palindromes.add(substring)
    return palindromes
