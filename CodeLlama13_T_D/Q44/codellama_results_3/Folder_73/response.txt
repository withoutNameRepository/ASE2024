def composite_nums_between_indices(my_list):
    # find the start and end indices of the range
    start_index = 28
    end_index = 40
    
    # create an empty set to store the composite numbers
    composite_set = set()
    
    # iterate through the list and check if each number is composite
    for i in range(start_index, end_index + 1):
        if my_list[i] % 2 == 0:
            composite_set.add(my_list[i])
            
    return composite_set
