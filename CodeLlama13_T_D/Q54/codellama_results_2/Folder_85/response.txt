def all_right_truncatable_prime(numbers):
    x = numbers[94]
    primes = []
    for i in range(2, x+1):
        if is_prime(i):
            primes.append(i)
    for prime in primes:
        for i in range(len(str(prime))-1, 0, -1):
            if not is_prime(int(str(prime)[:i])):
                break
        else:
            yield prime
