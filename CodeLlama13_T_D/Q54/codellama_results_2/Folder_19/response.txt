    prime_numbers = []

            prime_numbers.append(i)

def is_prime(n):
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

def is_right_truncatable_prime(p):
    while p > 1:
        if not is_prime(p):
            return False
        p //= 10
    return True
