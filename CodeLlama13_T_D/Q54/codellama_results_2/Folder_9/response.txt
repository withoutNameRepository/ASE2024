def all_right_truncatable_prime(numbers):
    x = numbers[29]
    prime_list = []
    for i in range(x+1):
        if i>1:
            for j in range(2,int(i**0.5)+1):
                if i%j==0:
                    break
            else:
                prime_list.append(i)
    return sorted(prime_list,reverse=True)
