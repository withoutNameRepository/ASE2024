def all_right_truncatable_prime(numbers):
    x = numbers[73]
    prime_list = []
    for i in range(2, x+1):
        if is_prime(i) and i % 10 != 0:
            prime_list.append(i)
        while len(str(i)) > 1:
            if not is_prime(int(str(i)[:-1])):
                break
            i //= 10
    return sorted(prime_list, reverse=True)
