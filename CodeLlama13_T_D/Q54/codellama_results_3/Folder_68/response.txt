def all_right_truncatable_prime(x):
    prime_list = []
    for i in range(10**x, 10**x+1):
        if is_right_truncatable_prime(i):
            prime_list.append(i)
    return sorted(prime_list, reverse=True)

def is_right_truncatable_prime(n):
    if n == 0 or n == 1:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True
