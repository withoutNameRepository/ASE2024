def is_prime(n):
    for i in range(2, n//2+1):
        if n%i == 0:
            return False
    return True

def is_right_truncatable_prime(n):
    if not is_prime(n):
        return False
    
    while n > 0:
        if not is_prime(n):
            return False
        n //= 10
        
    return True
