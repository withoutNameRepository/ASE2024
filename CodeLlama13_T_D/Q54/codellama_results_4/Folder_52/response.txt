def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

def is_truncatable_prime(s):
    while len(s) > 1:
        if not is_prime(int(s)):
            return False
        s = s[:-1]
    return True
