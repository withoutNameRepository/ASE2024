def is_right_truncatable_prime(num):
    if not is_prime(num):
        return False
    for i in range(1, len(str(num))):
        if not is_prime(num % (10 ** i)):
            return False
    return True

def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True
