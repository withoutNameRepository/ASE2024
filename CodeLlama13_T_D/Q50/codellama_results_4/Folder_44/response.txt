def if_contains_anagrams(my_list):
    anagrams = []
    for i in range(len(my_list)):
        for j in range(i+1, len(my_list)):
            if sorted(my_list[i].lower()) == sorted(my_list[j].lower()):
                anagrams.append((i, j))
    return len(anagrams) >= 44 and all(len(x) >= 3 for x in my_list)
