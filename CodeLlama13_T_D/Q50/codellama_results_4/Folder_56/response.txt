def if_contains_anagrams(strs):
    # Create a dictionary to store the count of anagrams for each length
    counts = {}
    
    # Iterate over the list of strings and check for anagrams
    for s in strs:
        # Ignore strings that are less than 3 characters long
        if len(s) < 3:
            continue
        
        # Convert the string to lowercase and sort its letters
        key = "".join(sorted(s.lower()))
        
        # If the sorted string is already in the dictionary, increment its count
        if key in counts:
            counts[key] += 1
            
        # Otherwise, add it to the dictionary with a count of 1
        else:
            counts[key] = 1
    
    # Return true if there are at least 77 pairs of anagrams in the list
    return len([k for k in counts.values() if k >= 2]) >= 77
