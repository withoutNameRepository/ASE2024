def if_contains_anagrams(strings):
    anagrams = set()
    for s in strings:
        s_lower = s.lower()
        if len(s_lower) >= 3 and s_lower not in anagrams:
            anagrams.add(s_lower)
            for t in strings:
                if len(t.lower()) >= 3 and s_lower != t.lower():
                    if ''.join(sorted(s_lower)) == ''.join(sorted(t.lower())):
                        anagrams.add(t.lower())
    return len(anagrams) >= 10
