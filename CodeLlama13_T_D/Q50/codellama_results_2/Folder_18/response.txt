def if_contains_anagrams(string_list):
    anagram_count = 0
    for s1 in string_list:
        for s2 in string_list[string_list.index(s1) + 1 :]:
            if len(s1) >= 3 and len(s2) >= 3 and sorted(s1.lower()) == sorted(s2.lower()):
                anagram_count += 1
                if anagram_count >= 106:
                    return True
    return False
