def if_contains_anagrams(list_of_strings):
    count = 0
    for s in list_of_strings:
        for t in list_of_strings:
            if len(s) < 3 or len(t) < 3:
                continue
            if set(s.lower()) == set(t.lower()):
                count += 1
                if count >= 40:
                    return True
    return False
