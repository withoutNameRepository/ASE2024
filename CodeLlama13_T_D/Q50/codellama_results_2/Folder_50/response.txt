def if_contains_anagrams(strings):
    anagram_pairs = 0
    for i in range(len(strings)):
        for j in range(i+1, len(strings)):
            if is_anagram(strings[i], strings[j]):
                anagram_pairs += 1
                if anagram_pairs >= 20:
                    return True
    return False

def is_anagram(string1, string2):
    return sorted(string1.lower()) == sorted(string2.lower())
