def if_contains_anagrams(mylist):
    for word in mylist:
        anagram = []
        for i, letter in enumerate(word):
            if word[i].lower() not in "qwertyuioplkjhgfdsazxcvbnm":
                continue
            else:
                anagram.append(letter)
    return len(anagram) >= 25
