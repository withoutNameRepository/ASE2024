def if_contains_anagrams(string_list):
    anagram_pairs = 0
    for i in range(len(string_list)):
        for j in range(i+1, len(string_list)):
            if (len(set(string_list[i])) >= 3) and (len(set(string_list[j])) >= 3):
                anagram_pairs += 1
    return anagram_pairs > 55
