def return_nth_smallest_ascii(string):
    # Get the characters between indices 3 and 52, both inclusive
    chars = string[3:53]
    
    # Find the 8th smallest ASCII value among those characters
    ascii_values = [ord(char) for char in chars]
    smallest_ascii = sorted(ascii_values)[7]
    
    # Return the character with the 8th smallest ASCII value
    for char in chars:
        if ord(char) == smallest_ascii:
            return char
