def return_nth_smallest_ascii(string):
    # Create a list of the characters in the string between indices 29 and 47, both inclusive
    char_list = [char for char in string[29:48]]
    
    # Use sorted() to sort the list in ascending order based on ASCII value
    sorted_list = sorted(char_list)
    
    # Return the character with the 7th smallest ASCII value from the sorted list
    return sorted_list[6]
