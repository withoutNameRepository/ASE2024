def return_nth_smallest_ascii(string):
    # get the characters in the string between indices 5 and 47, both inclusive
    chars = [char for char in string[5:48]]
    
    # sort the list of characters by their ASCII values
    sorted_chars = sorted(chars, key=lambda x: ord(x))
    
    # return the character with the 13th smallest ASCII value in the list
    return sorted_chars[12]
