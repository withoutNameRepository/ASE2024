def return_nth_smallest_ascii(string):
    ascii_values = {}
    for i, char in enumerate(string[0:18]):
        if char not in ascii_values:
            ascii_values[char] = [i]
        else:
            ascii_values[char].append(i)

    sorted_ascii_values = sorted(ascii_values.items(), key=lambda x: x[1])
    return sorted_ascii_values[17][0]
