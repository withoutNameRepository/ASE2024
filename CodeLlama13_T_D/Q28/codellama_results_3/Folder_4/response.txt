def return_nth_smallest_ascii(string):
    # Initialize an empty list to store the 4th smallest ASCII value
    ascii_values = []
    
    # Iterate over each character in the string
    for char in string:
        # If the current character is between indices 9 and 22, both inclusive
        if 9 <= char.ascii < 23:
            # Append the ASCII value of the current character to the list
            ascii_values.append(char.ascii)
    
    # Sort the list in ascending order
    sorted_ascii_values = sorted(ascii_values)
    
    # Return the 4th smallest ASCII value from the sorted list
    return sorted_ascii_values[3]
