def return_nth_smallest_ascii(string):
    # create a list of all characters in the string between indices 22 and 74, both inclusive
    chars = [char for char in string[22:75]]
    
    # sort the list of characters by their ASCII values
    sorted_chars = sorted(chars, key=ord)
    
    # return the character with the 12th smallest ASCII value
    return sorted_chars[11]
