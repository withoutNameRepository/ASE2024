def return_nth_smallest_ascii(string):
    # Get the 12th smallest ASCII value
    n = 12
    ascii_values = []
    for i in range(22, 75):
        if string[i] not in ascii_values:
            ascii_values.append(string[i])
    # Sort the list and return the element at index n-1
    sorted_ascii_values = sorted(ascii_values)
    return sorted_ascii_values[n-1]
