def return_nth_smallest_ascii(string):
    # Get the characters between indices 44 and 69, both inclusive
    chars = string[44:70]
    
    # Create a list of all ASCII values in the character set
    ascii_values = [ord(char) for char in chars]
    
    # Sort the list of ASCII values
    sorted_ascii_values = sorted(ascii_values)
    
    # Return the 15th smallest ASCII value
    return chr(sorted_ascii_values[14])
