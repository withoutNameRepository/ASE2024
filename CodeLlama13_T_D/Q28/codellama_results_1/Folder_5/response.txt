def return_nth_smallest_ascii(string):
    # Get a list of characters in the string between indices 5 and 89, both inclusive
    chars = string[5:90]
    
    # Convert each character to its ASCII value and create a list of those values
    ascii_values = [ord(char) for char in chars]
    
    # Sort the list of ASCII values and return the 13th smallest element
    sorted_ascii_values = sorted(ascii_values)
    return chr(sorted_ascii_values[12])
