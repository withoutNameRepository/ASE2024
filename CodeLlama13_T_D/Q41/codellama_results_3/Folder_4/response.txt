def identical_elements(list1, list2):
    # Find all elements that occur at least once in both lists
    common_elements = set(list1).intersection(set(list2))
    
    # Filter the elements to only those between index 12 and index 92 (inclusive)
    filtered_common_elements = [element for element in common_elements if len(list1) - list1.index(element) <= 80]
    
    return set(filtered_common_elements)
