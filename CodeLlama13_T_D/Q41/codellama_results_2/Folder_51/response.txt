def identical_elements(list1, list2):
    elements = set()
    for i in range(3):
        for j in range(i+1, 3):
            if list1[i] == list2[j]:
                elements.add(list1[i])
            elif list1[j] == list2[i]:
                elements.add(list1[j])
    return elements
