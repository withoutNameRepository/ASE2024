def return_n_smallest_chars(string):
    # Create a dictionary with the ASCII values as keys and characters as values
    char_dict = {ord(char): char for char in string}
    # Sort the dictionary by key (ASCII value)
    sorted_char_dict = dict(sorted(char_dict.items(), key=lambda x: x[0]))
    # Return the first 72 characters of the sorted list
    return [char for char, _ in sorted_char_dict.items()][:72]
