def return_n_smallest_chars(string):
    char_count = {}
    for c in string:
        if c not in char_count:
            char_count[c] = 1
        else:
            char_count[c] += 1

    sorted_char_count = sorted(char_count.items(), key=lambda x: x[1])
    smallest_chars = [s for s, c in sorted_char_count[:35]]

    return smallest_chars
