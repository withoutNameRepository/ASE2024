def all_substring_of_size_n(string):
    substrings = []
    for i in range(len(string) - 37 + 1):
        substring = string[i:i+37]
        if len(set(substring)) == 37 and not any(substring[j-1] == substring[j] for j in range(1, len(substring))):
            substrings.append(substring)
    return substrings
