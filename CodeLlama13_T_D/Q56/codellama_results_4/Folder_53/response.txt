def all_substring_of_size_n(string):
    substrings = []
    for i in range(len(string)):
        for j in range(i+1, len(string)+1):
            if j-i == 25:
                substring = string[i:j]
                if all(substring.count(c) <= 1 for c in substring):
                    substrings.append(substring)
    return substrings
