def all_substring_of_size_n(string):
    n = 27
    substrings = []
    for i in range(len(string) - n + 1):
        substring = string[i:i+n]
        if len(set(substring)) == n and all([char not in substring for char in substring[:-1]]):
            substrings.append(substring)
    return substrings
