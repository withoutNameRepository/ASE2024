def all_substring_of_size_n(s):
    substrings = []
    for i in range(len(s)):
        for j in range(i+1, len(s)+1):
            if j-i == 96:
                substring = s[i:j]
                if all(c not in substring for c in substrings):
                    substrings.append(substring)
    return substrings
