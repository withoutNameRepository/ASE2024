def lists_with_product_equal_n(my_list):
    result = []
    for i in range(len(my_list)):
        product = 1
        sublist = [my_list[i]]
        for j in range(i+1, len(my_list)):
            product *= my_list[j]
            if product == -18:
                result.append(sublist + [my_list[j]])
                break
        else:
            continue
    return result
