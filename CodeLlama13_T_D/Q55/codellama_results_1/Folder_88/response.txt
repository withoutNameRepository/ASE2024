def lists_with_product_equal_n(circular_list):
    sublists = []
    for i in range(len(circular_list)):
        product = 1
        sublist = [circular_list[i]]
        for j in range(i+1, len(circular_list)):
            product *= circular_list[j]
            if product == -56:
                sublists.append(sublist)
            elif product > -56:
                break
    return sublists
