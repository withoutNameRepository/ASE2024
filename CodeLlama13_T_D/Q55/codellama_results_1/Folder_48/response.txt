def lists_with_product_equal_n(lst, n):
    result = []
    for i in range(len(lst)):
        product = 1
        sublist = [lst[i]]
        for j in range(i+1, len(lst)):
            product *= lst[j]
            if product == n:
                result.append(sublist)
                break
            elif product > n:
                break
            else:
                sublist.append(lst[j])
    return result
