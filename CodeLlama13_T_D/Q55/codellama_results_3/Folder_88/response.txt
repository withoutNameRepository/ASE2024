def lists_with_product_equal_n(mylist):
    result = []
    for i in range(len(mylist)):
        product = 1
        sublist = []
        for j in range(i, len(mylist)):
            product *= mylist[j]
            if product == -56:
                sublist.append(mylist[j])
                result.append(sublist)
    return result
