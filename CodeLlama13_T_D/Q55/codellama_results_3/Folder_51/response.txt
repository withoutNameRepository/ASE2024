def lists_with_product_equal_n(circular_list):
    # Initialize an empty list to store the sublists
    sublists = []
    
    # Iterate over each element in the circular list
    for i in range(len(circular_list)):
        # Initialize a variable to store the product of the elements in the sublist
        product = 1
        
        # Iterate over each element in the sublist starting from the current element
        for j in range(i, len(circular_list)):
            # Update the product by multiplying it with the next element in the sublist
            product *= circular_list[j]
            
            # If the product equals -44, add the sublist to the list of sublists
            if product == -44:
                sublists.append(circular_list[i:j+1])
                
    return sublists
