def lists_with_product_equal_n(list):
    result = []
    for i in range(len(list)):
        product = 1
        sublist = []
        for j in range(i, len(list)):
            product *= list[j]
            if product == -72:
                result.append(sublist)
                break
            elif product > -72:
                break
            else:
                sublist.append(list[j])
    return result
