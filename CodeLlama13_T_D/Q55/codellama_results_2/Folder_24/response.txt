def lists_with_product_equal_n(lst):
    result = []
    for i in range(len(lst)):
        sublist = lst[i:i+1]
        product = 1
        for j in range(i, len(lst)-1):
            product *= lst[j+1]
            if product == -89:
                result.append(sublist)
                break
    return result
