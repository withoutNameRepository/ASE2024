def palindromes_between_indices(string):
    # Create a set to store the palindromes
    palindromes = set()

    # Iterate over each substring of length 6 between indices 2 and 9, both inclusive
    for i in range(2, 10):
        for j in range(i+5, len(string)):
            substr = string[i:j]
            if substr == substr[::-1]:
                palindromes.add(substr)

    # Return the set of palindromes
    return palindromes
