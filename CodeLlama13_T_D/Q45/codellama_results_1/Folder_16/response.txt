def palindromes_between_indices(string):
    letters = set("abcdefghijklmnopqrstuvwxyz")
    palindromes = set()
    for i in range(6, 10):
        for j in range(i+4, len(string)):
            substr = string[i:j]
            if len(substr) >= 5 and all(letter in letters for letter in substr):
                palindromes.add(substr)
    return palindromes
