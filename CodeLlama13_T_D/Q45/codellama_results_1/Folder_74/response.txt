def palindromes_between_indices(string):
    english_letters = "abcdefghijklmnopqrstuvwxyz"
    palindromes = set()
    for i in range(1, 7):
        for j in range(i + 5, len(string) + 1):
            substr = string[i - 1:j]
            if len(substr) >= 6 and substr == substr[::-1]:
                palindromes.add(substr)
    return palindromes
