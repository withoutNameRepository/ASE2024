def palindromes_between_indices(string):
    indices = range(5)
    letters = set("abcdefghijklmnopqrstuvwxyz")
    palindromes = set()
    
    for i in indices:
        for j in indices:
            if j - i < 4:
                continue
            
            substr = string[i:j]
            if "".join(sorted(substr)) in letters and substr == substr[::-1]:
                palindromes.add(substr)
    
    return palindromes
