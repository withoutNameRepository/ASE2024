def palindromes_between_indices(string):
    indices = range(0, 5)
    palindromes = set()
    for i in indices:
        for j in indices[i+1:]:
            substr = string[i:j]
            if len(substr) < 3 or substr == substr[::-1]:
                continue
            palindromes.add(substr)
    return palindromes
