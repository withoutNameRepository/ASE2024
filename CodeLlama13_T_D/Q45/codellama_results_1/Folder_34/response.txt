def palindromes_between_indices(string):
    indices = range(5, 8)
    letters = set("abcdefghijklmnopqrstuvwxyz")
    palindromes = set()
    
    for i in indices:
        for j in indices:
            if i <= j:
                substr = string[i:j+1]
                if len(substr) >= 4 and all(letter in letters for letter in substr):
                    palindromes.add(substr)
    
    return palindromes
