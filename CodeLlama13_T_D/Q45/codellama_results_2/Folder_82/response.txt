def palindromes_between_indices(string):
    # Create a set to store the palindromes
    palindromes = set()
    
    # Iterate over each possible arrangement of letters in the string between indices 1 and 6, both inclusive
    for i in range(1, len(string) - 5):
        # Extract the substring between indices i and i + 6
        substring = string[i:i+7]
        
        # Check if the substring is a palindrome
        if substring == substring[::-1]:
            # Add the palindrome to the set
            palindromes.add(substring)
    
    # Return the set of palindromes
    return palindromes
