def palindromes_between_indices(string):
    english_letters = set("abcdefghijklmnopqrstuvwxyz")
    palindromes = set()
    for i in range(0, 5):
        for j in range(i, 5):
            substring = string[i:j+1]
            if len(substring) >= 3 and all(letter in english_letters for letter in substring.lower()):
                palindromes.add(substring)
    return palindromes
