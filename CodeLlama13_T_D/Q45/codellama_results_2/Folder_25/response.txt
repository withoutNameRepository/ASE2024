def palindromes_between_indices(string):
    english_letters = "abcdefghijklmnopqrstuvwxyz"
    palindromes = set()
    for i in range(1, 5):
        for j in range(i + 2, len(string) - i + 1):
            substr = string[i:j]
            if substr[::-1].lower() == substr.lower():
                palindromes.add(substr)
    return palindromes
