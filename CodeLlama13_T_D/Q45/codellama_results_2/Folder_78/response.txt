def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    
    # Iterate over the characters of the string between indices 6 and 8, both inclusive
    for i in range(6, 9):
        # Check if the character at index i is a letter of the English alphabet
        if string[i].isalpha():
            # Add the character to the set of palindromes
            palindromes.add(string[i])
    
    # Return the set of palindromes
    return palindromes
