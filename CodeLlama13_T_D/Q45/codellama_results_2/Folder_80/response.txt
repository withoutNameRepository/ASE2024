def palindromes_between_indices(string):
    english_letters = "abcdefghijklmnopqrstuvwxyz"
    palindromes = set()
    
    for i in range(3, 9):
        for j in range(i+1, len(string)-7):
            substr = string[i:j]
            if substr == substr[::-1]:
                palindromes.add(substr)
                
    return palindromes
