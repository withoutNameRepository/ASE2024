def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    
    # Loop through each possible arrangement of the English letters in the given string that occur between index 5 to index 9, both inclusive
    for i in range(5, 10):
        for j in range(i + 6, len(string)):
            # Check if the substring from index i to index j is a palindrome
            if string[i:j] == string[i:j][::-1]:
                # Add the palindrome to the set
                palindromes.add(string[i:j])
    
    return palindromes
