def palindromes_between_indices(string):
    english_letters = set("abcdefghijklmnopqrstuvwxyz")
    palindromes = set()
    for i in range(4, 8):
        for j in range(i + 1, 8):
            substr = string[i:j]
            if len(substr) >= 5 and all(letter in english_letters for letter in substr):
                palindromes.add(substr)
    return palindromes
