def all_even_ints_inclusive(my_list):
    # Find the index of the first even integer in the list
    for i in range(len(my_list)):
        if my_list[i] % 2 == 0:
            start = i
            break

    # Find the index of the last even integer in the list

    for j in range(len(my_list) - 1, -1, -1):
        if my_list[j] % 2 == 0:
            end = j
            break

    # Return the list of all even integers from index 74 to index 95, both inclusive

    return my_list[start:end+1]
