def all_even_ints_inclusive(my_list):
    # Get the range of even integers in the list
    even_int_range = [i for i in my_list if i % 2 == 0]
    # Return an empty list if no even integers exist in the specified range
    if not even_int_range:
        return []

    # Get the index of the first and last even integers in the list
    first_even_int = even_int_range[0]
    last_even_int = even_int_range[-1]

    # Return the list of all even integers from index 30 to index 87, both inclusive
    return even_int_range[first_even_int:last_even_int+1]
