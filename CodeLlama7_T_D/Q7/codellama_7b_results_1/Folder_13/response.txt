def all_even_ints_inclusive(my_list):
    # Initialize an empty list to store the even integers
    even_ints = []

    # Iterate through the list of integers, starting from index 25 and ending at index 87 (inclusive)
    for i in range(25, 88):

        if my_list[i] % 2 == 0:
