def all_pos_ints_inclusive(my_list):
    # Find the starting index of the positive integers in the list
    start_index = None
    for i in range(len(my_list)):
        if my_list[i] > 0:
            start_index = i
            break

    # Find the ending index of the positive integers in the list

    end_index = None

    for i in range(start_index, len(my_list))):
        if my_list[i] > 0:
            end_index = i

    # Return the list of all positive integers from start_index to end_index, both inclusive

    return my_list[start_index:end_index+1]
