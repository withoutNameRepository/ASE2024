def all_pos_ints_inclusive(my_list):
    # Find the first positive integer in the list
    first_pos_int = None
    for i in range(50, 201):
        if my_list[i] > 0:
            first_pos_int = i
            break

    # If no positive integers exist in the specified range, return an empty list
    if first_pos_int is None:
        return []

    # Initialize the result list with the first positive integer

    result = [my_list[first_pos_int]]

    # Iterate over the remaining elements of the list

    for i in range(first_pos_int + 1, 201):
        if my_list[i] > 0:
            result.append(my_list[i])

    return result
