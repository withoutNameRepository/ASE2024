def all_pos_ints_inclusive(my_list):
    # Initialize an empty list to store the positive integers
    pos_ints = []
    # Iterate through the list starting from index 66
    for i in range(66, len(my_list)):
        # Check if the element at current index is a positive integer
        if my_list[i] > 0:
