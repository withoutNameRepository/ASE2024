def is_prime(n):
    # Check if n is greater than 1
    if n > 1:
        # Loop through the numbers from 2 to n - 1
        for i in range(2, n):
            # Check if i divides n
            if n % i == 0:
                # Return False if a divisor is found
                return False

    # Return True if n is a prime number
    return True
