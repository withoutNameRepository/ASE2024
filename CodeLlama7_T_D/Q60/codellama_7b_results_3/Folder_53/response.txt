def all_left_right_truncatable_prime(x):
    # Define a lambda function to check if a number is prime or not
    is_prime = lambda n: (n > 1) and all(n % i for i in range(2, int(n ** 0.5) + 1))
    # Define a list to store all left-and-right-truncatable prime numbers less than or equal to x
    result = []
    # Iterate from 1 to x and check if each number is a left-and-right-truncatable prime number using the is_prime function
    for i in range(1, x + 1):
        if is_prime(i) and len(str(i)) >= 2:
            # If the number is a left-and-right-truncatable prime number, add it to the result list
            result.append(i)
    # Return the sorted result list in descending order
    return sorted(result, reverse=True)
