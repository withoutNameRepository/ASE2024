def all_left_right_truncatable_prime(x):
    # Create a list of all prime numbers less than or equal to x
    prime_list = [p for p in range(2, x + 1) if all(p % i != 0 for i in range(2, int(p ** 0.5) + 1))]
    # Create an empty list to store the left-and-right-truncatable prime numbers
    lr_prime_list = []
    # Iterate through each prime number in the list
    for prime in prime_list:
        # Check if the prime number is both left-truncatable and right-truncatable
        if all(str(prime)[i:] == str(prime)[:i] for i in range(1, len(str(prime)))):
            # If it is both left-truncatable and right-truncatable, add it to the list of left-and-right-truncatable prime numbers
            lr_prime_list.append(prime)
    # Return the sorted list of all left-and-right-truncatable prime numbers less than or equal to x
    return sorted(lr_prime_list, reverse=True)
