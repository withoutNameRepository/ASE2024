def is_prime(n):
    # Check if n is greater than 1
    if n > 1:
        # Iterate from 2 to the square root of n
        for i in range(2, int(n ** 0.5) + 1):
            # Check if i divides n
            if n % i == 0:
                # Return False if i divides n
                return False

    # Return True if n is a prime number
    return True

def is_left_right_truncatable_prime(n):
    # Check if n is a prime number
    if not is_prime(n):
        # Return False if n is not a prime number
        return False

    # Remove the leading leftmost and last rightmost digits from n
    leading_leftmost = int(str(n)[1:])
    last_rightmost = int(str(n)[:-1]))

    # Check if the number formed by removing the leading leftmost and last rightmost digits is a prime number
    if not is_prime(leading_leftmost) or not is_prime(last_rightmost):
        # Return False if the number formed by removing the leading leftmost and last rightmost digits is not a prime number
        return False

    # Return True if n is a left-and-right-truncatable prime number
    return True
