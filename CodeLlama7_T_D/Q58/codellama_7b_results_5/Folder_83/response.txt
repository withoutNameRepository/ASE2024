import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(matrix.shape[0] - 3 + 1):
        for j in range(matrix.shape[1] - 4 + 1):
            if all([matrix[i + k, j + l] == 47 for k in range(4) for l in range(4)])):
