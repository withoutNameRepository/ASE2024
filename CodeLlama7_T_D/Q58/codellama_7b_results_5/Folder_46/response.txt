import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(matrix.shape[0] - 143):
        for j in range(matrix.shape[1] - 143):
            submatrix = matrix[i:i+144, j:j+144]
            if np.count_nonzero(submatrix) == 144:
                count += 1
    return count
