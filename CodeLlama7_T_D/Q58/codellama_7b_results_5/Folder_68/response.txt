import numpy as np

def submatrix_with_n_numbers(matrix: np.ndarray, n: int) -> int:
    count = 0
    for i in range(matrix.shape[0] - n + 1):
        for j in range(matrix.shape[1] - n + 1):
            submatrix = matrix[i:i+n, j:j+n]
            if all([x == 74 for x in submatrix.flatten()]):
                count += 1

    return count
