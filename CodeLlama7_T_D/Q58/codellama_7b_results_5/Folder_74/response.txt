import numpy as np

def submatrix_with_n_numbers(matrix):
    count = 0
    for i in range(len(matrix) - 1):
        for j in range(len(matrix[i]) - 1):
            if matrix[i][j] == 48:
                for k in range(i, len(matrix)):
                    for l in range(j, len(matrix[k])):
                        if matrix[k][l] == 48:
                            count += 1
    return count
