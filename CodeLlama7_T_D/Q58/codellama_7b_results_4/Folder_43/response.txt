import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize the count variable
    count = 0

    # Loop over each submatrix of size 85x85 in the given matrix
    for i in range(matrix.shape[0] - 84):
        for j in range(matrix.shape[1] - 84):
            # Check if the current submatrix contains 85 integers each
            if np.count_nonzero(matrix[i:i+85, j:j+85]) == 85 * 85:
                # Increment the count variable
                count += 1

    # Return the count of all submatrices that contain 85 integers each
    return count
