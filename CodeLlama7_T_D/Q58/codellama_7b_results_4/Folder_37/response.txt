import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize the count variable
    count = 0
    # Loop over each submatrix of size (4, 4) in the given matrix
    for i in range(matrix.shape[0] - 3):
        for j in range(matrix.shape[1] - 3):
            # Check if the current submatrix contains 64 integers
            if np.count_nonzero(matrix[i:i+4, j:j+4]) == 64:
                # Increment the count variable
                count += 1

    # Return the count of all submatrices that contain 64 integers each
    return count
