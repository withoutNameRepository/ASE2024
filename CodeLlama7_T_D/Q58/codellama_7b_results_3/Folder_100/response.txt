# Function to count the number of integers in a submatrix
def count_integers(submatrix):
    # Initialize count variable
    count = 0

    # Loop over all rows and columns of the submatrix
    for i in range(submatrix.shape[0]):
        for j in range(submatrix.shape[1]):
            # Increment count if the current element is a 1
            if submatrix[i][j] == 1:
                count += 1

    # Return the number of integers in the submatrix
    return count
