import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize the count of submatrices with 2 integers each
    count = 0
    # Loop over all rows and columns of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Check if the element at position (i, j) is equal to 2
            if matrix[i][j] == 2:
                # Initialize a variable to keep track of the number of elements in the submatrix
                num_elements = 0
                # Loop over all rows and columns of the submatrix
                for k in range(i, i + 3):
                    for l in range(j, j + 3):
                        # Check if the element at position (k, l) is equal to 2
                        if matrix[k][l] == 2:
                            # Increment the number of elements in the submatrix
                            num_elements += 1
                # If the number of elements in the submatrix is 4, then increment the count by 1
                if num_elements == 4:
                    count += 1
    return count
