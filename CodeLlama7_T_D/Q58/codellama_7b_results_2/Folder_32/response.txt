import numpy as np

def submatrix_with_n_numbers(matrix):
    # Initialize the count to 0
    count = 0
    # Iterate over all possible submatrices of size n x n
    for i in range(len(matrix) - 102):
        for j in range(i + 103, len(matrix)):
            # Check if the current submatrix contains 103 integers
            if sum([matrix[row][col] for row in range(i, j) for col in range(i, j)]) == 103:
                # If it does, increment the count
                count += 1

    # Return the count of all submatrices of size n x n that contain 103 integers each
    return count
