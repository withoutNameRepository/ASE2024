import numpy as np

def submatrix_with_n_numbers(matrix: np.ndarray, n: int) -> int:
    count = 0
    for i in range(len(matrix)):
        for j in range(i + 1, len(matrix)):
            if np.count_nonzero(matrix[i] == 85) == n and np.count_nonzero(matrix[j] == 85) == n:
                count += 1
    return count
