import numpy as np
def submatrix_with_n_numbers(matrix):

    count = 0
    # Loop over each submatrix of size n x n in the given matrix
    for i in range(matrix.shape[0] - 1):

            if sum(matrix[i:i+2, j:j+2]) == 166:

                count += 1
