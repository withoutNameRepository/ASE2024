def all_odd_ints_inclusive(my_list):
    # Get the index of the first odd integer
    first_odd = next((i for i, x in enumerate(my_list) if x % 2 == 1)), None)
    # Get the index of the last odd integer
    last_odd = next((i for i, x in enumerate(my_list) if x % 2 == 1))
    # Return the list of all odd integers between first_odd and last_odd
    return [x for x in my_list[first_odd:last_odd+1] if x % 2 == 1]
