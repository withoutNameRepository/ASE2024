def all_odd_ints_inclusive(my_list):
    # Check if the list is valid
    if not my_list:
        return []

    # Get the range of indices to consider
    start = 22
    end = 50

    # Initialize an empty list to store the results
    result = []

    # Iterate over the indices in the specified range
    for i in range(start, end + 1):
        # Check if the current index contains an odd integer
        if my_list[i] % 2 != 0:
            # Add the current index to the list of results
            result.append(my_list[i])

    # Return the list of all odd integers found in the specified range
    return result
