def all_odd_ints_inclusive(my_list):
    # Create a new list to store the odd integers
    odd_ints = []
    # Loop through the list, starting from index 74
    for i in range(74, len(my_list)):
        # Check if the current element is an odd integer
        if my_list[i] % 2 != 0:
