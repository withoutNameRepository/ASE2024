def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each letter in the string
    for i in range(len(string)):
        # Get the character at index i
        char = string[i]
        # Check if the character is a letter (not whitespace or punctuation)
        if char.isalpha():
            # Get the index of the last occurrence of the character
            last_index = len(string) - 1 - string[::-1].find(char)
            # Check if the character occurs in the specified range
            if i <= last_index <= 7:
                # Create a palindrome by reversing the substring starting at index i and ending at the index of the last occurrence of the character
                palindrome = string[i:last_index+1][::-1] + string[i:last_index+1]
                # Add the palindrome to the set of palindromes
                palindromes.add(palindrome)
    return palindromes
