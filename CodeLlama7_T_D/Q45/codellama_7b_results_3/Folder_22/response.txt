def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(5, len(string)):
        # Check if the current character is a letter
        if string[i].isalpha():
            # Create a copy of the substring starting at index 5 and ending at index i
            substr = string[5:i]
            # If the length of the substring is greater than or equal to 3, check if it is a palindrome
            if len(substr) >= 3:
                # Check if the substring is a palindrome in a case-insensitive manner
                if substr == substr[::-1]:
                    # Add the substring to the set of palindromes
                    palindromes.add(substr)

    return palindromes
