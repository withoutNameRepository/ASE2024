def palindromes_between_indices(string):
    # Define the range of indices to consider
    start_index = 0
    end_index = 4
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over each substring in the given string that has a length of at least 4 characters
    for i in range(start_index, end_index + 1):
        for j in range(i + 1, end_index + 1):
            substring = string[i:j]
            # Check if the substring is a palindrome and has a length of at least 4 characters
            if substring == substring[::-1] and len(substring) >= 4:
                # Add the palindrome to the set
                palindromes.add(substring)
    return palindromes
