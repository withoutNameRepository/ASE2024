 def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(5, 7 + 1):
        # Get the current character and its lowercase equivalent
        char = string[i]
        lower_char = char.lower()
        # Check if the current character is a vowel or not
        if lower_char in 'aeiou':
            # If it is a vowel, add its lowercase equivalent to the palindrome set
            palindromes.add(lower_char)
    return palindromes
