def palindromes_between_indices(s):
    # Create a list of all English letters in the given string
    letters = []
    for c in s:
        if c.isalpha():
            letters.append(c)
    # Create a set to store all possible palindromes
    palindromes = set()
    # Iterate over all possible lengths of palindromes
    for length in range(3, len(letters) + 1):
        # Iterate over all possible starting indices of the palindrome
        for start in range(len(letters) - length + 2):
            # Check if the substring starting at 'start' and of length 'length' is a palindrome
            substr = letters[start:start + length]
            if substr == substr[::-1]:
                # Add the palindrome to the set
                palindromes.add("".join(substr))
    return palindromes
