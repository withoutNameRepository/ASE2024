def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each index of the string
    for i in range(len(string)):
        # Get the character at the current index
        char = string[i]
        # Check if the character is a letter (not a whitespace or a punctuation)
        if char.isalpha():
            # Loop through each index of the remaining part of the string
            for j in range(i + 1, len(string)):
                # Get the character at the current index
                char2 = string[j]
                # Check if the character is a letter (not a whitespace or a punctuation)
                if char2.isalpha():
                    # Create a palindrome by concatenating the two characters
                    palindrome = char + char2
                    # Check if the palindrome has at least 4 letters and is a case-insensitive match of the original string
                    if len(palindrome) >= 4 and palindrome.lower() == string.lower():
                        # Add the palindrome to the set of palindromes
                        palindromes.add(palindrome)
    # Return the set of palindromes
    return palindromes
