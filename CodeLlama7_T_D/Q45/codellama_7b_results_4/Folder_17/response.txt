def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each possible arrangement of the English letters in the given string that occur between index 2 to index 6, both inclusive
    for i in range(2, len(string) - 4 + 1):
        # Check if the substring starting at index 2 and ending at index i is a palindrome
        if string[2:i] == string[2:i][::-1]:
            # If it is, add it to the set of palindromes
            palindromes.add(string[2:i])
    # Return the set of palindromes
    return palindromes
