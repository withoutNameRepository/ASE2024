def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(3, len(string)):
        # Check if the character is a vowel or consonant
        if string[i] in "aeiouAEIOU":
            # Create a palindrome of length at least 5 by reversing the characters on either side of the current character
            for j in range(i-2, i+3):
                if j < 0 or j >= len(string):
                    continue

                if string[j] in "aeiouAEIOU":
                    palindrome = ""
                    for k in range(j-1, i+1):
                        if k < 0 or k >= len(string):
                            break

                        palindrome += string[k].lower()
                    palindromes.add(palindrome)
    return palindromes
