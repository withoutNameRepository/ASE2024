def palindromes_between_indices(string):
    # Create a list of all the English letters in the string
    english_letters = [char for char in string if char.isalpha()]
    # Filter out any letters that are not between index 3 and 8
    filtered_letters = [letter for letter in english_letters if 3 <= letter.lower().index(letter) <= 8]
    # Create a set of all palindromes of the length of at least 6 using the filtered letters
    palindromes = {word for word in filtered_letters if len(word) >= 6 and word == word[::-1]}
    return palindromes
