def palindromes_between_indices(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each possible arrangement of the English letters in the given string
    for i in range(1, len(string)):
        # Form a palindrome by reversing the substring starting from index i and ending at index 8 (inclusive)
        palindrome = string[i:8][::-1]
        # Check if the formed palindrome has length of at least 7
        if len(palindrome) >= 7:
            # Add the palindrome to the set of palindromes
            palindromes.add(palindrome)
    # Return the set of palindromes
    return palindromes
