def all_neg_ints_exclusive(my_list):
    # Find the index of the first negative integer in the list
    first_negative_index = next((i for i, x in enumerate(my_list) if x < 0)), None)
    # If there are no negative integers in the list, return an empty list
    if first_negative_index is None:
        return []

    # Find the index of the last negative integer in the list

    last_negative_index = len(my_list) - 1
    while my_list[last_negative_index] < 0 and last_negative_index > first_negative_index:
        last_negative_index -= 1

    # Return a list containing all the negative integers in the specified range, exclusive of the endpoints

    return my_list[first_negative_index:last_negative_index+1] if first_negative_index is not None and last_negative_index is not None else []
