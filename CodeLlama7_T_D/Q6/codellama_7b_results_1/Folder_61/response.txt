def all_neg_ints_exclusive(my_list):
    # Initialize an empty list to store the negative integers
    neg_ints = []

    # Iterate through the list of integers
    for i in range(len(my_list)):
        # Check if the current integer is negative and not equal to 0
        if my_list[i] < 0 and my_list[i] != 0:
            # Add the negative integer to the list
            neg_ints.append(my_list[i])

    # Return the list of all negative integers from index 0 to index 7, both exclusive
