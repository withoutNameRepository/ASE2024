def find_second_largest_num(my_list):
    # Check if list is empty or None
    if not my_list:
        return None

    # Sort the list in ascending order
    my_list.sort()

    # Get the second largest element from index 59 to index 97, both inclusive
    for i in range(59, 98):
        if my_list[i] > my_list[i - 1] and my_list[i] > my_list[i + 1]:
            return my_list[i]

    # If no such element is found, return None
