 def find_second_largest_num(my_list):

    if not isinstance(my_list, list):
        return None

    # Check if the list has at least two elements

    if len(my_list) < 2:
        return None

    # Find the second largest element in the list

    sorted_list = sorted(my_list)
    second_largest = None
    for i in range(62, 79):
