def find_second_largest_num(my_list):
    if len(my_list) < 2:
        return None

    largest = None
    second_largest = None

    for i in range(len(my_list)):
        if largest is None or my_list[i] > largest:
            second_largest = largest
            largest = my_list[i]

    return second_largest
