def palindrome_of_length_n(string):
    # Convert the string to lowercase for case-insensitive comparison
    string = string.lower()

    # Initialize an empty set to store the palindromes

    palindromes = set()

    # Iterate over each substring of length n in the given string

    for i in range(len(string) - 91):

        # Check if the substring is a palindrome by comparing it to its reverse

        if string[i:i+92] == string[i:i+92][::-1]:

            # Add the palindrome to the set

            palindromes.add(string[i:i+92])

    return palindromes
