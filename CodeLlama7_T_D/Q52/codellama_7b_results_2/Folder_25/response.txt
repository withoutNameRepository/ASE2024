 def palindrome_of_length_n(string):
    # Define the pattern for finding palindromes of length n
    pattern = r'(?i)(?=^.{90}$)\w*(\w)(\w)*\1'
    # Use re.findall to find all matches in the string
    return set(re.findall(pattern, string))
