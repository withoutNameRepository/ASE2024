def palindrome_of_length_n(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each substring of length n in the given string
    for i in range(len(string) - 2):
        # Check if the substring is a palindrome by reversing it and comparing with the original substring
        if string[i:i+3] == string[i:i+3][::-1]:
            # If it is a palindrome, add it to the set of palindromes
            palindromes.add(string[i:i+3])
    # Return the set of palindromes
    return palindromes
