def palindrome_of_length_n(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(len(string)):
        # Check if the character is a letter (i.e., not a space or other non-letter character)
        if string[i].isalpha():
            # Get the substring starting from the current character and ending with the character at index 2n - 1 (where n is the length of the palindrome)
            substr = string[i:2*i + 1]
            # Check if the substring is a palindrome
            if substr == substr[::-1]:
                # Add the palindrome to the set of palindromes
                palindromes.add(substr)
    # Return the set of palindromes
    return palindromes
