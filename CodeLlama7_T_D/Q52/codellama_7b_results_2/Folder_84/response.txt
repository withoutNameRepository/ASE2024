def palindrome_of_length_n(string):
    # Find all substrings of length n in the given string
    substrs = [string[i:i+n] for i in range(len(string)-n+1)]
    # Return an empty set if there are no substrings of length n
    if not substrs:
        return set()

    # Create a set of all palindromes found
    palindromes = set()
    for substr in substrs:
        if substr == substr[::-1]:
            palindromes.add(substr)

    # Return the set of all palindromes found
    return palindromes
