def palindrome_of_length_n(string):
# Check if the input string is a palindrome of length n
if len(string) != n:
return set()

# Convert the input string to lowercase for case-insensitive matching
lowercase_string = string.lower()

# Initialize an empty set to store the palindromes
palindromes = set()

# Iterate over the characters in the input string
for i in range(len(string)):
# Check if the character at position i is a palindrome of length n
if lowercase_string[i:i+n] == lowercase_string[i:i+n][::-1]:
# Add the palindrome to the set of palindromes
palindromes.add(lowercase_string[i:i+n])

return palindromes
