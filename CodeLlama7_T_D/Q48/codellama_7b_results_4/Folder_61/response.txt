def return_binary_or_hexa(tuple):
    # Initialize variables
    a = tuple[0]
    b = tuple[-1]
    sum = 0
    binary_string = ''
    hexadecimal_string = ''

    # Calculate the sum of all integers from a + 1 to b - 1 that are not present in the tuple between indices 0 and 8
    for i in range(a + 1, b):
        if i not in tuple:
            sum += i

    # Check if the sum is an odd number
    if sum % 2 != 0:

        # Return the binary representation string of the sum
        for digit in bin(sum)[2:]:
            binary_string += digit

        return binary_string

    else:

        # Return the hexadecimal representation string of the sum
        for digit in hex(sum)[2:]:
            hexadecimal_string += digit

        return hexadecimal_string
