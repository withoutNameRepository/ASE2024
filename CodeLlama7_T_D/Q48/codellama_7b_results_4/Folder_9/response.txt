def return_binary_or_hexa(tuple):
    # Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 70 and 200
    sum = 0
    for i in range(70, 200):
        if tuple[i] not in range(tuple[69] + 1, tuple[201] - 1)):
            sum += tuple[i]
    # Check if the sum is an odd number
    if sum % 2 == 1:
        return bin(sum)[2:]
    else:
        return hex(sum)[2:]
