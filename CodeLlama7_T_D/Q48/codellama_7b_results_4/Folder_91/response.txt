def return_binary_or_hexa(my_tuple):
# Check if all integers from a + 1 to b - 1 (both inclusive) are present in the specified range in the tuple
if all(i in my_tuple for i in range(my_tuple[0] + 1, my_tuple[6] - 1)):
return ""
# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 0 and 6
sum = 0
for i in range(my_tuple[0] + 1, my_tuple[6] - 1):
if i not in my_tuple:
sum += i
# Check if the sum is an odd number
if sum % 2 != 0:
return bin(sum)[2:]
else:
return hex(sum)[2:]
