def return_binary_or_hexa(my_tuple):
# Check if all integers from a + 1 to b - 1 (both inclusive) are present in the specified range in the tuple
if all(i in my_tuple[86:90] for i in range(my_tuple[86], my_tuple[90]+1)):
return ''
# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 86 and 90.
sum = 0
for i in range(my_tuple[86], my_tuple[90]+1):
if i not in my_tuple[86:90]:
sum += i
# Check if the sum is an odd number, if so return the binary representation string of the sum. Otherwise, return the hexadecimal representation string of the sum.
if sum % 2 == 1:
return ''.join(bin(sum)[2:])
else:
return ''.join(hex(sum)[2:]).upper()
