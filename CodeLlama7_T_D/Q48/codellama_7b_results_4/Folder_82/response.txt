def return_binary_or_hexa(tuple):
    # Initialize variables
    sum = 0
    binary = ''
    hexa = ''

    # Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 20 and 200
    for i in range(tuple[20], tuple[200]):
        if i not in tuple:
            sum += i

    # Determine whether the sum is an odd number or not
    if sum % 2 == 1:
        binary = bin(sum)[2:]

    else:

        hexa = hex(sum)[2:]

    # Return the binary or hexadecimal representation string of the sum, depending on whether it is an odd number or not
    if binary:
        return binary

    else:
        return hexa
