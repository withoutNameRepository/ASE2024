 def return_binary_or_hexa(my_tuple):
    a = my_tuple[43]
    b = my_tuple[70]
    sum_list = []

    for i in range(a + 1, b):
        if i not in my_tuple:
            sum_list.append(i)

    sum = sum(sum_list)

    if sum % 2 == 0:
        return hex(sum).replace('0x', '')

    else:
        return bin(sum).replace('0b', '')
