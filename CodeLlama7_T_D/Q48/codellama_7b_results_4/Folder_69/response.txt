def return_binary_or_hexadecimal(tuple):
a = tuple[39]
b = tuple[43]
sum = 0
for i in range(a + 1, b - 1):
if i not in tuple[39:43]:
sum += i

if sum % 2 == 1:
return f"{sum:#0x}"
else:
return f"{sum:#0b}"

else:
return ""
