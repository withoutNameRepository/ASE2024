def return_binary_or_hexa(tuple):
    # Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 56 and 90
    sum = 0
    for i in range(56, 90):
        if tuple[i] == 0:
            continue

        for j in range(tuple[56], tuple[90] + 1):
            if j == tuple[i]:
                break

            sum += 1

    # If the sum is an odd number, return the binary representation string of the sum; otherwise, return the hexadecimal representation string of the sum
    if sum % 2 != 0:
        return bin(sum)[2:]

    else:

        return hex(sum)[2:]
