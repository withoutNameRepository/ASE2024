 def return_binary_or_hexa(tuple):
    a = tuple[44]
    b = tuple[67]
    sum = 0
    for i in range(a + 1, b - 1):
        if i not in tuple[44:67]:
            sum += i
    if sum % 2 == 0:
        return f"{sum:x}"
    else:
        return f"{sum:b}"
