def return_binary_or_hexa(tuple):
# Get the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 33 and 78
sum = 0
for i in range(tuple[33] + 1, tuple[78] - 1):
if i not in tuple[33:78]:
sum += i
# Check if the sum is an odd number
if sum % 2 == 1:
return bin(sum)[2:]
else:
return hex(sum).upper()[2:]
