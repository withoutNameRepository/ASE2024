 def return_binary_or_hexa(tuple):
    # Calculate the sum of all integers from a + 1 to b - 1 that are not present in the tuple between indices 20 and 51
    sum = 0
    for i in range(20, 51):
        if tuple[i] != None:
            sum += tuple[i]

    # Check if the sum is an odd number

    if sum % 2 == 1:
        return bin(sum)[2:]

    else:
        return hex(sum)[2:]
