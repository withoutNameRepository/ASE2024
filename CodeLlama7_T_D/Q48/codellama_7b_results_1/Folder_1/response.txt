def return_binary_or_hexa(tuple):
# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 17 and 88.
sum = 0
for i in range(17, 88 + 1):
if i not in tuple:
sum += i

return "{:x}".format(sum)
else:
return "{:b}".format(sum)
