def return_binary_or_hexa(tuple):
# check if all integers from a + 1 to b - 1 (both inclusive) are present in the specified range in the tuple
if all(i in tuple[70:200] for i in range(tuple[69] + 1, tuple[201] - 1)):
return ""

# calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 70 and 200
sum = 0
for i in range(tuple[69] + 1, tuple[201] - 1):
if i not in tuple[70:200]:
sum += i

# check if the sum is an odd number and return the binary or hexadecimal representation string accordingly

if sum % 2 == 1:
return f"{sum:#b}"
else:
return f"{sum:#x}"
