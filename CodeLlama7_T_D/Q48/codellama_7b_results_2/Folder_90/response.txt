def return_binary_or_hexa(tuple):
# Get the indices of a and b in the tuple
a_idx = tuple.index(60)
b_idx = tuple.index(90)
# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 60 and 90
sum = 0
for i in range(a_idx + 1, b_idx):
if tuple[i] not in range(a_idx + 1, i):
sum += tuple[i]
# Return the binary or hexadecimal representation string of the sum
if sum % 2 == 0:
return f"{sum:x}"
else:
return f"{sum:b}"
