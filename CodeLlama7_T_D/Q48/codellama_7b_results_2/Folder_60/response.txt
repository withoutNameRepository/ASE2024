def return_binary_or_hexa(my_tuple):
# Get the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 38 and 52
sum = 0
for i in range(my_tuple[38], my_tuple[52]):
if i not in my_tuple:
sum += i
