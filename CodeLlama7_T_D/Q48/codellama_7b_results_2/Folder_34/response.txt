def return_binary_or_hexa(tuple):
# Check if all integers from a + 1 to b - 1 (both inclusive) are present in the specified range in the tuple
if len(set(tuple[60:200])) == len(range(tuple[60] + 1, tuple[200] - 1)))):
return ""
# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 60 and 200
sum = 0
for i in range(tuple[60] + 1, tuple[200] - 1)):
if i not in set(tuple[60:200])):
sum += i
# Return the binary or hexadecimal representation string of the sum depending on whether it is an odd number
if sum % 2 == 1:
return "{0}".format(bin(sum))
else:
return "{0}".format(hex(sum))
