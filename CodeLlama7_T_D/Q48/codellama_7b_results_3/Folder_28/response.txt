def return_binary_or_hexa(tuple):
# Initialize variables
a = tuple[86]
b = tuple[99]
sum = 0
# Calculate sum of integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 86 and 99
for i in range(a + 1, b):
if i not in tuple[86:99]:
sum += i
# Return binary or hexadecimal representation of sum depending on whether it is an odd number
if sum % 2 == 0:
return f"{sum:x}"
else:
return f"{sum:b}"
