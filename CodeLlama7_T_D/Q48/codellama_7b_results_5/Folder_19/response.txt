def return_binary_or_hexa(tuple):
# Check if all integers from a + 1 to b - 1 (both inclusive) are present in the specified range in the tuple
if all(x in tuple for x in range(tuple[1], tuple[4])):
return ''

# Calculate the sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 1 and 4
sum = sum(range(tuple[1] + 1, tuple[4]), tuple))

# Check if the sum is an odd number
if sum % 2 == 1:
return '{0}'.format(bin(sum)[2:])
else:
return '{0}'.format(hex(sum)[2:]).upper()
