def return_binary_or_hexa(tuple):
    # Initialize variables
    a = tuple[60]
    b = tuple[200]
    sum = 0
    binary_rep = ''
    hexa_rep = ''
    # Iterate through the range of integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 60 and 200
    for i in range(a + 1, b):
        if i not in tuple[60:200]:
            sum += i

    # Return the binary or hexadecimal representation string of the sum, depending on whether the sum is an odd number
    if sum % 2 == 1:
        return binary_rep

    else:
        return hexa_rep
