 def return_binary_or_hexa(tuple):
    # Calculate sum of all integers from a + 1 to b - 1 (both inclusive) that are not present in the tuple between indices 32 and 99
    sum = 0
    for i in range(32, 99 + 1):
        if tuple[i] == 0:
            sum += i - 32 + 1
    # If sum is an odd number, return the binary representation string of the sum; otherwise, return the hexadecimal representation string of the sum.
    if sum % 2 != 0:
        return bin(sum)[2:]
    else:
        return hex(sum)[2:]
