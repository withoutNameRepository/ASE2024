def gcf_three_nums(my_list):
    # Calculate the greatest common factor (gcf) of the elements at indices 90, 54 and 96
    gcf = my_list[90]
    for i in range(54, 96 + 1):
        if my_list[i] % gcf == 0:
            continue

        else:

            gcf = my_list[i] // (my_list[i] % gcf)
    return gcf
