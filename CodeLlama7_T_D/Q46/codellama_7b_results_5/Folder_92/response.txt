def gcd(a, b):
while b != 0:
a, b = b, a % b
return a

def lcm(a, b, c):
# Calculate the least common multiple of a and b
lcm_ab = a * b / gcd(a, b)
# Calculate the least common multiple of a and c
lcm_ac = a * c / gcd(a, c)
# Calculate the least common multiple of b and c
lcm_bc = b * c / gcd(b, c)
# Return the product of the least common multiples of a, b, and c
