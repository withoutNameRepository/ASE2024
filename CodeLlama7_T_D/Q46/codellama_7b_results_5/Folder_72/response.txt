def gcf_three_nums(my_list):
    # Calculate the greatest common factor of the elements at indices 44, 91 and 42
    gcf = 1
    for i in range(len(my_list)):
        if i == 44 or i == 91 or i == 42:
            gcf = __gcf(gcf, my_list[i])
    return gcf

def __gcf(a, b):
    while b != 0:
        a, b = b, a % b

    return a
