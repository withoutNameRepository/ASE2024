def gcf_three_nums(my_list):
    # Get the elements at indices 145, 183 and 770
    num1 = my_list[145]
    num2 = my_list[183]
    num3 = my_list[770]
    # Find the greatest common factor using the Euclidean algorithm
    gcf = 1

    while True:
