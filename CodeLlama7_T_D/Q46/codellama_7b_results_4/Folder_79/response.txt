def gcf_three_nums(my_list):
    # Find the greatest common factor of the elements at indices 16, 87 and 95
    gcf = 1
    for i in range(16, 95 + 1):
        if my_list[i] > 1:
            for j in range(i + 1, 95 + 1):
                if my_list[j] % my_list[i] == 0:
                    gcf = max(gcf, my_list[i])
    return gcf
