def gcf(a, b, c):
# Calculate the greatest common divisor of a and b
gcd_ab = gcd(a, b)
# Calculate the greatest common divisor of a and c
gcd_ac = gcd(a, c)
# Calculate the greatest common divisor of b and c
gcd_bc = gcd(b, c)
# Return the product of the three greatest common divisors
return gcd_ab * gcd_ac * gcd_bc
def gcd(a, b):
if a == 0:
return b
else:
return gcd(b % a, a)
