def gcf_three_nums(my_list):
    # Calculate the greatest common factor of the elements at indices 693, 331 and 668
    element_1 = my_list[693]
    element_2 = my_list[331]
    element_3 = my_list[668]
    gcf = 1

    for i in range(2, min(element_1, element_2, element_3) + 1):
        if (element_1 % i == 0) and (element_2 % i == 0) and (element_3 % i == 0)):
            gcf = i

    return gcf
