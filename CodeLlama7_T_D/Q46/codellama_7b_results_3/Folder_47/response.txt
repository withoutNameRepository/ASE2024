def gcf_three_nums(my_list):
    # Calculate the greatest common factor of three numbers
    num1 = my_list[36]
    num2 = my_list[85]
    num3 = my_list[83]
    if num1 == 0:

        for i in range(2, min([num1, num2, num3]) + 1):
