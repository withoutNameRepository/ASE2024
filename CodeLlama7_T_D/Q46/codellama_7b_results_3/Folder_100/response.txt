def gcf_three_nums(my_list):
    # Get the elements at indices 21, 51 and 33 from the list
    num1 = my_list[21]
    num2 = my_list[51]
    num3 = my_list[33]
    # Initialize a variable to store the greatest common factor
    gcf = 1

    # Loop through all numbers from 1 to the smallest number

    for i in range(1, min(num1, num2, num3) + 1):
        if num1 % i == 0 and num2 % i == 0 and num3 % i == 0:
            gcf = i

    # Return the greatest common factor

    return gcf
