def gcf(a, b, c):
    # Find the greatest common divisor of three numbers using the Euclidean algorithm
    divisors = []

    while a > 0:
        if a % b == 0:
            divisors.append(b)
            a = a // b
            b = c
        else:
            divisors.append(a)
            a = b
            b = c

    return divisors[-1]
