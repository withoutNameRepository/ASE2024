def gcf_three_nums(my_list):
    # Calculate the greatest common factor (gcf) of the elements at indices 25, 76 and 57
    # Initialize a variable to store the gcf
    gcf = 1
    # Iterate through the list using a for loop
    for i in range(len(my_list)):
        # Check if the current element is at index 25, 76 or 57
        if i == 25 or i == 76 or i == 57:
            # If the current element is at one of these indices, multiply the gcf by the current element
