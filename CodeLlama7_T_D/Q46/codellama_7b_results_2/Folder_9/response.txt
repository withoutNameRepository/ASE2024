def gcf_three_nums(my_list):
    # Calculate the greatest common factor of the elements at indices 66, 90 and 27
    gcf = 1
    for i in range(66, 90 + 1):
        if my_list[i] % gcf == 0:
            continue

        else:

            while True:

                if my_list[i] % gcf == 0 and my_list[27] % gcf == 0:

                gcf += 1
