def sum_ints_div_by_either_nums(my_list):
    # Initialize variables to keep track of the sum and the current index
    sum = 0
    i = 0
    
    # Loop through the list, starting from index 0
    while i < len(my_list):
        
        # If the current element is divisible by either -5 or -6, add it to the sum
        if my_list[i] % (-5) == 0 or my_list[i] % (-6) == 0:
            sum += my_list[i]
        
        # Increment the current index by 1
        i += 1
    
    # Return the sum of all integers divisible by either -5 or -6 from index 0 to index 6, both inclusive
    return sum
