def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    sum = 0
    start_index = 250
    end_index = 629
    # Iterate through the list and calculate the sum of all integers divisible by either 850 or 587 within the specified range
    for i in range(start_index, end_index + 1):
        if my_list[i] % 850 == 0 or my_list[i] % 587 == 0:
            sum += my_list[i]
    # Return the sum of all integers divisible by either 850 or 587 within the specified range, or 0 if no such integers exist
    return sum
