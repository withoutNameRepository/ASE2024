 def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    total = 0
    start_index = 138
    end_index = 424
    # Iterate over the list and calculate the sum of all integers divisible by either -863 or -329 within the specified range
    for i in range(start_index, end_index + 1):
        if my_list[i] % -863 == 0 or my_list[i] % -329 == 0:
            total += my_list[i]
    # Return the sum of all integers divisible by either -863 or -329 within the specified range, or 0 if no such integers exist in the specified range
    return total
