def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    start_index = 77
    end_index = 87
    total_sum = 0

    # Iterate through the list and calculate the sum of all integers divisible by either 23 or 57 from index 77 to index 87, both inclusive.
    for i in range(start_index, end_index + 1):
        if my_list[i] % 23 == 0 or my_list[i] % 57 == 0:
            total_sum += my_list[i]

    # Return the sum of all integers divisible by either 23 or 57 from index 77 to index 87, both inclusive. If no such integers exist in the specified range, the function should return 0.
    return total_sum
