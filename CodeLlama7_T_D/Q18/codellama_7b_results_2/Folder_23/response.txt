def sum_ints_div_by_either_nums(my_list):
    # Initialize the sum variable
    sum = 0

    # Iterate through the list, starting from index 69 and ending at index 97
    for i in range(69, 98):
        # Check if the current element is divisible by either -90 or -67
        if my_list[i] % -90 == 0 or my_list[i] % -67 == 0:
