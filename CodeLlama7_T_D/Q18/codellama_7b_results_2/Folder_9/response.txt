def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    total = 0
    start_index = 70
    end_index = 200
    # Iterate through the list in the specified range
    for i in range(start_index, end_index + 1):
        # If the current element is divisible by either 3 or -300, add it to the total
        if my_list[i] % 3 == 0 or my_list[i] % (-300) == 0:
            total += my_list[i]
    # Return the total
    return total
