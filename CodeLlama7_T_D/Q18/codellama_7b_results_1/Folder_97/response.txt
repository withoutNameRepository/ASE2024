def sum_ints_div_by_either_nums(my_list):
	# Initialize variables
	sum = 0
	start_index = 58
	end_index = 75
	# Iterate through the list and calculate the sum of all integers divisible by either 72 or 62
	for i in range(start_index, end_index + 1):
