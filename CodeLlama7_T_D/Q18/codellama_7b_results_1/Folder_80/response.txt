def sum_ints_div_by_either_nums(my_list):
    # Initialize variables for the sum and the divisible integers
    sum = 0
    divisible_integers = []
    # Loop through the list from index 14 to index 93, inclusive
    for i in range(14, 94):
        # Check if the current integer is divisible by either 71 or 81
        if my_list[i] % 71 == 0 or my_list[i] % 81 == 0:
            # If it is divisible, add it to the list of divisible integers

    return sum(divisible_integers)
