def sum_ints_div_by_either_nums(my_list):
    # Initialize variables
    sum = 0
    for i in range(20, 201):
        # Check if current element is divisible by either -20 or -200
        if my_list[i] % (-20) == 0 or my_list[i] % (-200) == 0:
            # If it is, add it to the sum
            sum += my_list[i]
