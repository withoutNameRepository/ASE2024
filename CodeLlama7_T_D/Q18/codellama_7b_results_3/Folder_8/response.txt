    # Initialize variables
    sum = 0
    lower_bound = 36
    upper_bound = 85

    # Iterate through the list and calculate the sum of all integers divisible by either -27 or -96
    for i in range(lower_bound, upper_bound + 1):
        if my_list[i] % (-27) == 0 or my_list[i] % (-96) == 0:
