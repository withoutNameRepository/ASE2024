    # Initialize variables
    total = 0
    start_index = 200
    end_index = 200
    # Iterate through the list and calculate the sum of all integers divisible by either 17 or -5
    for i in range(start_index, end_index + 1):
        if my_list[i] % 17 == 0 or my_list[i] % -5 == 0:
            total += my_list[i]
