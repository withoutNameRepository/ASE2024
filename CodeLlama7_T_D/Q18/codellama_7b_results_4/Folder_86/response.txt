    # Initialize variables for the sum and the range of indices
    total = 0
    start = 54
    end = 79
    # Iterate over the list in the specified range
    for i in range(start, end + 1):
        # Check if the current element is divisible by either 54 or 28
        if my_list[i] % 54 == 0 or my_list[i] % 28 == 0:

            total += my_list[i]
