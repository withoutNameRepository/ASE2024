def _get_gcd(a, b):
    # This function calculates the greatest common divisor (GCD) of two positive integers
    # It returns the GCD as an integer value

    if a == 0:
        return b

    while b != 0:
        remainder = a % b
        a = b
        b = remainder

    return a
