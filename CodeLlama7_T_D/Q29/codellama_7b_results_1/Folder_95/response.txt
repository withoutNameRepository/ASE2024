def gcf_two_nums(my_list):
    # Find the greatest common factor of two numbers
    # by iteratively dividing each number
    # by the other until they are equal
    num1 = my_list[43]
    num2 = my_list[99]
    while num1 != num2:
        if num1 > num2:
            num1 -= num2
        else:
            num2 -= num1
