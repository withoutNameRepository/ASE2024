def gcf_two_nums(my_list):
    # Find the greatest common factor of two numbers
    # by dividing both numbers by their greatest common divisor

    num1 = my_list[23]
    num2 = my_list[45]

    if num1 > num2:
        smaller_num = num2
        larger_num = num1

    else:
        smaller_num = num1
        larger_num = num2

    gcf = 1

    for i in range(2, smaller_num + 1):
        if smaller_num % i == 0 and larger_num % i == 0:
            gcf = i

    return gcf
