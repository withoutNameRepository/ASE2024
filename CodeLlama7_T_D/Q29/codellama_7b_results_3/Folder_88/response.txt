def gcf_two_nums(my_list):
    # Calculate the greatest common factor of two numbers at indices 27 and 34
    num1 = my_list[27]
    num2 = my_list[34]
    if num1 == num2:
        return num1

    while True:

        if num1 % 2 == 0 and num2 % 2 == 0:
            num1 /= 2
            num2 /= 2

        elif num1 % 3 == 0 and num2 % 3 == 0:
            num1 /= 3
            num2 /= 3

        else:

            break

    return num1
