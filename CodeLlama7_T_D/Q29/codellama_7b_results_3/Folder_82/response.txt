def gcf(a, b):
    if a == 0:
        return b

    while True:

        if a % 2 == 0 and b % 2 == 0:
            a /= 2
            b /= 2

        else:

            if a % 2 == 0:
                a /= 2

            elif b % 2 == 0:
                b /= 2

            else:

                a, b = b, a

    return a
