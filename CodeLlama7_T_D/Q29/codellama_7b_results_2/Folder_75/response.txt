def gcf_two_nums(my_list):
    # Calculate the greatest common factor of two numbers
    # by dividing both numbers until they are equal
    num1 = my_list[42]
    num2 = my_list[25]
    while num1 != num2:
        if num1 > num2:
            num1 -= num2
        else:
            num2 -= num1
