def gcf_two_nums(my_list):
    # Calculate the greatest common factor of two numbers
    num1 = my_list[33]
    num2 = my_list[72]
    if num1 == num2:
        return num1

    while num1 != 0 and num2 != 0:
        if num1 > num2:
            num1 -= num2
        else:
            num2 -= num1

    return num1 or num2
