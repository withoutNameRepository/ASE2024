def all_odd_ints_exclusive(my_list):
    # Check if the input is a list of integers
    if not (isinstance(my_list, list) and all(isinstance(x, int) for x in my_list))):
        raise ValueError("Input must be a list of integers")
    # Get the length of the input list
    n = len(my_list)
    # Check if the specified range is valid
    if not (20 <= 36 < n)):
        raise ValueError("Invalid range specified")
    # Initialize an empty list to store the odd integers
    odd_ints = []
    # Iterate over the input list and append the odd integers to the output list
    for i in range(20, 36):
        if my_list[i] % 2 != 0:
            odd_ints.append(my_list[i])
    return odd_ints
