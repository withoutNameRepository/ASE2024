def all_odd_ints_exclusive(my_list):
    # Find the first odd integer in the list, starting from index 50
    for i in range(50, len(my_list)):
        if my_list[i] % 2 != 0:
            first_odd = i
            break

        if my_list[i] % 2 != 0:

    else:
