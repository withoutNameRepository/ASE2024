def all_odd_ints_exclusive(my_list):
    # Initialize an empty list to store the odd integers
    odd_ints = []

    # Loop through the list of integers, starting from index 2 (exclusive) and ending at index 5 (exclusive)
    for i in range(2, 5):

        if my_list[i] % 2 != 0:
