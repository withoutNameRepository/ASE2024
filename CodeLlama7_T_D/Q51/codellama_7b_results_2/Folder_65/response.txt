def if_contains_anagrams(my_list):
    # Initialize variables
    anagram_count = 0
    three_or_more_length_count = 0
    case_insensitive_count = 0
    # Iterate through the list of strings
    for string in my_list:
        # Get a list of all possible anagrams
        anagram_strings = [string]
        # Check if there are any other strings in the list that are anagrams of this one
        for other_string in my_list:
            # If the string is not this one and it is an anagram, add it to the list of possible anagrams

                anagram_strings.append(other_string)

            three_or_more_length_count += 1
