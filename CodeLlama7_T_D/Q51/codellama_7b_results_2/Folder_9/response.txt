def is_anagram(str1, str2):
    # Convert both strings to lowercase
    str1 = str1.lower()
    str2 = str2.lower()

    # Sort the characters in both strings
    str1 = sorted(str1)
    str2 = sorted(str2)

    # Check if the two sorted strings are equal
    return str1 == str2
