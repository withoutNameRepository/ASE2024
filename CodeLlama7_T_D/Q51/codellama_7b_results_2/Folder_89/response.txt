def if_contains_anagrams(my_list):
    # Count the number of anagram pairs in the list
    anagram_pairs = 0
    for i in range(len(my_list) - 1):
        # Compare the current string with all subsequent strings
        for j in range(i + 1, len(my_list)):
            # If the two strings are anagrams of each other, increment the anagram pair count
            if sorted(my_list[i]) == sorted(my_list[j])):
                anagram_pairs += 1
    return anagram_pairs <= 73
