def if_contains_anagrams(my_list):
# Define a function that takes in a list of strings and returns true if there are at most 19 pairs of anagrams in the given list, otherwise false.
# Create a counter variable to keep track of the number of anagrams found in the list.

    current_string = my_list[i]
    current_length = len(current_string)

    if (current_length >= 3) and (current_string.isalpha()):
        # Create a copy of the current string that is converted to lowercase.
        lowercase_string = current_string.lower()
        # Iterate over the list of strings again, starting from the next element after the current string.
        for j in range(i + 1, len(my_list)):

            other_string = my_list[j]
            other_length = len(other_string)

            if (other_length >= 3) and (other_string.isalpha()):

                other_lowercase_string = other_string.lower()

# Return true if the number of anagrams found is at most 19, otherwise false.
