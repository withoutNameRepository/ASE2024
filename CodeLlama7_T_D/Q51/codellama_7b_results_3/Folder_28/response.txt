def if_contains_anagrams(strings):
    # Create a dictionary to store the counts of each letter
    letter_counts = {}

    # Iterate over each string in the list
    for string in strings:
        # Convert the string to lowercase
        string = string.lower()

        # Iterate over each character in the string
        for char in string:
            # If the character is not a letter, skip it
            if not char.isalpha():
                continue

            # Otherwise, increment the count of the letter
            else:
                if char in letter_counts:
                    letter_counts[char] += 1

                else:
                    letter_counts[char] = 1

    # Iterate over each pair of anagrams in the list
    for i in range(len(strings) - 1):
        for j in range(i + 1, len(strings))):
            # If the two strings are anagrams, return True
            if sorted(strings[i]) == sorted(strings[j]):
                return True

    # Otherwise, return False
    return False
