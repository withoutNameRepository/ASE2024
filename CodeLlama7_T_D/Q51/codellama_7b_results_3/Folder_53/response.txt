def if_contains_anagrams(strings):
    # Check if there are at most 8 pairs of anagrams in the list
    num_pairs = 0
    for i in range(len(strings)):
        for j in range(i + 1, len(strings)):
            if sorted(strings[i]) == sorted(strings[j]):
                num_pairs += 1
                if num_pairs > 8:
                    return False

    # Check if the anagrams are of the length of at least three

    for string in strings:
        if len(string) < 3:
            return False

    # Return True if no pairs of anagrams were found, or if there are at most 8 pairs of anagrams in the list

    return num_pairs <= 8
