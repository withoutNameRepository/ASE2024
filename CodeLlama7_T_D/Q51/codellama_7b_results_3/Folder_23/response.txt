def if_contains_anagrams(str_list):
    # Count the number of anagrams in the list
    num_anagrams = 0
    for i in range(len(str_list) - 1):
        for j in range(i + 1, len(str_list))):
            if sorted(str_list[i].lower()) == sorted(str_list[j].lower()):
                num_anagrams += 1
    # Return True if there are at most 49 pairs of anagrams in the list; otherwise, return False.
    if num_anagrams <= 49:
        return True
    else:
        return False
