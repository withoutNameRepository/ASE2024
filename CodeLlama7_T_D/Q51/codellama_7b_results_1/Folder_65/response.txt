def if_contains_anagrams(words):
    # Check if there are at most 72 pairs of anagrams in the list
    num_pairs = 0
    for i in range(len(words)):
        word1 = words[i].lower()
        for j in range(i + 1, len(words)):
            word2 = words[j].lower()
            if sorted(word1) == sorted(word2):
                num_pairs += 1
                if num_pairs > 72:
                    return False

    # If there are at most 72 pairs of anagrams in the list, return True
    return True
