def if_contains_anagrams(words):
    # Initialize variables
    anagram_pairs = 0
    minimum_length = 3

    # Check each word in the list
    for i in range(len(words)):
        # Get the current word
        word = words[i]

        # Check if the word is an anagram of itself
        if len(word) < minimum_length:
            continue

        # Check if the word is an anagram of any other word in the list
        for j in range(len(words)):
            # Skip the current word and any duplicates
            if i == j or words[i] == words[j]:
                continue

            # Check if the two words are anagrams of each other
            if sorted(word) == sorted(words[j]):
                # Increment the number of anagram pairs found
                anagram_pairs += 1

        # If there are more than five anagram pairs in the list, return False
        if anagram_pairs > 54:
            return False

    # If there are no more than five anagram pairs in the list, return True
    return True
