def if_contains_anagrams(my_list):
# Check if the list is empty or None
if my_list is None or len(my_list) == 0:
return False

# Create a dictionary to store the anagrams
anagram_dict = {}

# Iterate over the list of strings
for string in my_list:

# Convert the string to lowercase
lowercase_string = string.lower()

# Sort the characters of the string
sorted_chars = ''.join(sorted(lowercase_string))

# Check if the sorted characters are already present in the dictionary
if sorted_chars in anagram_dict:

# If the sorted characters are already present, increment the count
anagram_dict[sorted_chars] += 1

# Otherwise, add the sorted characters to the dictionary with a count of 1
else:
anagram_dict[sorted_chars] = 1

# Check if there are more than 41 pairs of anagrams in the list
if len(anagram_dict) > 41:
return False

# Otherwise, return true
return True
