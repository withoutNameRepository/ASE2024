def if_contains_anagrams(strings):

    num_anagrams = 0
    # Iterate over each string in the list
    for string in strings:
        # Get the length of the string
        length = len(string)
        # Check if the length of the string is at least three
        if length >= 3:

            letters = set(string.lower())

                num_anagrams += 1
