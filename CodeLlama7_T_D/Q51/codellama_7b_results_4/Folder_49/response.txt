def if_contains_anagrams(string_list):
    # Check if the list is empty
    if len(string_list) == 0:
        return False

    # Initialize a counter for the number of anagrams found
    num_anagrams = 0

    # Iterate over the strings in the list
    for string1 in string_list:
        # Convert the strings to lowercase before comparing them
        string1 = string1.lower()

        # Iterate over the remaining strings in the list
        for string2 in string_list[string_list.index(string1) + 1:]:
            # If the two strings are anagrams, increment the counter
            if sorted(string1) == sorted(string2):
                num_anagrams += 1

    # Return True if there are at most 10 pairs of anagrams in the list, otherwise return False
    return num_anagrams <= 10
