def if_contains_anagrams(str_list):
# Check if the list is empty
if len(str_list) == 0:
return False

# Check if there are at most 79 pairs of anagrams in the given list
num_pairs = 0
for i in range(len(str_list)):
for j in range(i + 1, len(str_list))):
if sorted(str_list[i]) == sorted(str_list[j])):
num_pairs += 1

# Return True if there are at most 79 pairs of anagrams in the given list, and False otherwise
return num_pairs <= 79
