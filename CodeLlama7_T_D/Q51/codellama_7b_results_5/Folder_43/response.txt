def if_contains_anagrams(words):
    # Check if there are at most three pairs of anagrams in the given list
    num_pairs = 0
    for i in range(len(words) - 1):
        for j in range(i + 1, len(words))):
            if sorted(words[i]) == sorted(words[j])):
                num_pairs += 1
                if num_pairs > 3:
                    return False

    # If there are no more than three pairs of anagrams in the given list, then the function returns True

    return True
