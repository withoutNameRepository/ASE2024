def filter_chars(s):
    # Create a new empty string
    result = ""

    # Iterate through each character in the given string
    for char in s:
        # Check if the current character is between indices 503 and 753, both inclusive
        if 503 <= s.index(char) <= 753:
            # Check if the current character is greater than or equal to '9' and smaller than or equal to 'w'
            if '9' <= char <= 'w':
                # Do not add the current character to the result string
                continue

        # Add the current character to the result string
        result += char

    # Return the altered string
    return result
