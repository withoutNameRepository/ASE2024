 def filter_chars(s):
    # create a new string with the desired characters removed
    new_string = ""

    # loop through each character in the original string

    for c in s:

        # check if the current character is between indices 513 and 877, inclusive

        if 513 <= s.index(c) <= 877:

            # check if the current character meets the desired criteria

            if c >= '?' and c <= 'n':

                # skip this character since it meets the desired criteria

                continue


        # append the current character to the new string

        new_string += c

    # return the altered string

    return new_string
