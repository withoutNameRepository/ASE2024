def filter_chars(my_string):
    # Create a list of characters in the given string
    char_list = list(my_string)
    # Initialize an empty list to store the filtered characters
    filtered_chars = []
    # Iterate through each character in the list
    for char in char_list:
        # Check if the current character is between indices 57 and 96, both inclusive, and is greater than or equal to character '<' and smaller than or equal to character 'w'
        if (char_list.index(char) >= 57 and char_list.index(char) <= 96) and (char >= '<' and char <= 'w'):
            # If the current character is in the desired range, remove all occurrences of that character from the string
            filtered_chars.append(char)
    # Join the filtered characters back into a string and return it
    return "".join(filtered_chars)
