 def filter_chars(s):
    # create a new empty string
    res = ""
    # loop through each character in the original string
    for c in s:
        # check if the current character is between indices 27 and 29, both inclusive
        if (27 <= ord(c) <= 29) and (';' <= c <= 'b'):
            # do not add the current character to the new string
            continue

        # add the current character to the new string
        res += c

    # return the altered string
    return res
