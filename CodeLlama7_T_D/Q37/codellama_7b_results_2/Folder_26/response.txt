def filter_chars(string):
    # Initialize a new empty string
    filtered = ""

    # Iterate over each character in the given string
    for char in string:
        # If the current character is between indices 20 and 62, both inclusive, and is greater than or equal to character 'i' and smaller than or equal to character 'k', then remove all occurrences of that character from the string
        if (20 <= len(filtered) + 1 <= 62) and ('i' <= char <= 'k'):
            filtered = filtered.replace(char, "")
        # Otherwise, add the current character to the filtered string
        else:
            filtered += char

    # Return the altered string

    return filtered
