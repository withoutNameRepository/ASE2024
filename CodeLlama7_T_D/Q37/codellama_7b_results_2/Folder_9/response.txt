def filter_chars(string):
    # Initialize an empty list to store the filtered characters
    filtered_characters = []

    # Iterate through each character in the given string
    for character in string:

        # If the current character is between indices 38 and 99, both inclusive, and is greater than or equal to character 'A' and smaller than or equal to character 'Q', add it to the filtered_characters list
        if (character.isalpha() and len(character) >= 1 and len(character) <= 100) and (38 <= ord(character) <= 99)):
            filtered_characters.append(character)

    # Join the filtered characters back into a single string
    filtered_string = "".join(filtered_characters)

    return filtered_string
