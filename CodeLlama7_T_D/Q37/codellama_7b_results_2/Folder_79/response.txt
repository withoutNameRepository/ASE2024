def filter_chars(s):
    # Create a new string with all characters removed
    new_str = ""

    # Iterate over each character in the original string
    for char in s:

        # If the current character is between indices 10 and 52, both inclusive, and is greater than or equal to character '@' and smaller than or equal to character 'T', then remove all occurrences of that character from the string
        if 10 <= char <= 52 and char >= '@' and char <= 'T':
            new_str += char

    # Return the altered string
    return new_str
