def filter_chars(string):
    # Create a list of characters in the given string
    char_list = list(string)

    # Loop through each character in the list
    for i, char in enumerate(char_list):
        # Check if the current character is between indices 73 and 74 (both inclusive), is greater than or equal to 'U', and is smaller than or equal to 'l'
        if i >= 73 and i <= 74 and char >= 'U' and char <= 'l':
            # Remove all occurrences of the current character from the string
            char_list = [c for c in char_list if c != char]

    # Return the altered string
    return "".join(char_list)
