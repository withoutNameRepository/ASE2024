def filter_chars(s):
    # Create a new string with the filtered characters
    filtered = ''

    # Iterate over each character in the original string
    for char in s:

        # Check if the current character should be filtered out
        if (char >= ':' and char <= 'r') and (char.isalpha() or char.isdigit()):
            continue

        # Add the character to the new string if it is not filtered out
        filtered += char

    # Return the altered string with the filtered characters removed
    return filtered
