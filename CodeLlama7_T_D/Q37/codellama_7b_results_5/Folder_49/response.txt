 def filter_chars(s):
    start = 42
    end = 97
    char_range = range(start, end + 1)
    filtered_string = ""

    for i in range(len(s)):
        if s[i].isalpha():
            if ord(s[i]) not in char_range:
                continue

            else:

                filtered_string += s[i]

        else:

            filtered_string += s[i]

    return filtered_string
