def filter_chars(s):
# Create a list of all characters in the string
char_list = list(s)
# Iterate over the list and remove any characters that do not meet the criteria
for i in range(len(char_list)):
if char_list[i] < '9' or char_list[i] > 's':
del char_list[i]
# Join the remaining characters back together into a string
return "".join(char_list)
