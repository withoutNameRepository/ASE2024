def filter_chars(s):
    # Create a new string buffer for the filtered characters
    buf = ""

    # Iterate over each character in the input string
    for c in s:
        # Check if the current character is between indices 22 and 85, both inclusive, and is greater than or equal to character '7' and smaller than or equal to character 'e'
        if (22 <= ord(c) <= 85) and (ord('7') <= ord(c) <= ord('e')):
            # If the current character meets the condition, skip it
            continue

        # Otherwise, append the current character to the buffer
        buf += c

    # Return the filtered string buffer
    return buf
