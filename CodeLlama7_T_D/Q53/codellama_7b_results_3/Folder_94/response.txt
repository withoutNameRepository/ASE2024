def palindrome_of_length_at_least_n(string, n=41):
    # Create a set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(len(string)):
        # Check if the character is an English letter
        if string[i].isalpha():
            # Get the length of the substring starting at index i and ending at the end of the string
            substr_length = len(string) - i
            # If the length of the substring is greater than or equal to n, check if it is a palindrome
            if substr_length >= n:
                # Check if the substring is a palindrome in a case-insensitive manner
                if string[i:].lower() == string[i:].lower()[::-1]:
                    # Add the palindrome to the set of palindromes
                    palindromes.add(string[i:])
    # Return the set of palindromes
    return palindromes
