def palindrome_of_length_at_least_n(string, n=13):
    # Create a set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(len(string)):
        # Check if the current character is a letter
        if string[i].isalpha():
            # Get the substring starting from the current character to the end of the string
            substr = string[i:].lower()
            # Loop through each palindrome in the substring
            for j in range(len(substr)):
                # Check if the current palindrome is at least n characters long
                if len(substr[:j+1]) >= n:
                    # Add the palindrome to the set
                    palindromes.add(substr[:j+1])
    # Return the set of palindromes
    return palindromes
