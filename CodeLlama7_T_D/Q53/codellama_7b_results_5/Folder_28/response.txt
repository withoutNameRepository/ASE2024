def palindrome_of_length_at_least_n(s, n):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each substring of length greater than or equal to n in the input string s
    for i in range(len(s) - n + 1):
        # Check if the substring is a palindrome in a case-insensitive manner
        if s[i:i+n].lower() == s[i:i+n][::-1].lower():
            # If it is a palindrome, add it to the set of palindromes
            palindromes.add(s[i:i+n])
    # Return the set of palindromes
    return palindromes
