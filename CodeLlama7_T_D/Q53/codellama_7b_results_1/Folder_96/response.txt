def palindrome_of_length_at_least_n(s, n):
    # Initialize an empty set to store the palindromes
    palindromes = set()

    # Iterate over each substring of length greater than or equal to 24 in the given string
    for i in range(len(s) - n + 1):
        # Check if the current substring is a palindrome
        if s[i:i+n] == s[i:i+n][::-1]:
            # If it is a palindrome, add it to the set of palindromes
            palindromes.add(s[i:i+n])

    # Return the set of palindromes found in the given string
