 def palindrome_of_length_at_least_n(string, n):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(len(string)):
        # Get the character at position i
        char = string[i]
        # If the character is a letter (not a whitespace or punctuation)
        if char.isalpha():
            # Loop through each character in the reversed substring starting from position i
            for j in range(len(string)-1, i-1, -1):
                # Get the character at position j
                rev_char = string[j]
                # If the character is a letter (not a whitespace or punctuation) and it matches the original character
                if rev_char.isalpha() and char.lower() == rev_char.lower():
                    # Add the palindrome to the set of palindromes
                    palindromes.add(string[i:j+1])
    # Return the set of palindromes
    return palindromes
