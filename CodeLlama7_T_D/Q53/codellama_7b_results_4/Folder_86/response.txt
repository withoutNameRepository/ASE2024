def palindrome_of_length_at_least_n(string, n):
    # Define a function to check if a string is a palindrome
    def is_palindrome(word):
        return word == word[::-1]
    # Create an empty set to store the palindromes
    palindromes = set()
    # Loop through each word in the given string
    for word in string.split():
        # Check if the word is a palindrome and its length is greater than or equal to n
        if is_palindrome(word) and len(word) >= n:
            # Add the palindrome to the set
            palindromes.add(word)
    # Return the set of all palindromes in the given string
    return palindromes
