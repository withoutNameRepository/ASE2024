def all_even_ints_exclusive(my_list):
    # Find the first even integer in the list
    for i in range(10, len(my_list)):
        if my_list[i] % 2 == 0:
            start = i
            break

    else:

        return []


# Find the last even integer in the list

for i in range(start, len(my_list)):

    if my_list[i] % 2 == 0:

        end = i


# Return a list of all even integers between start and end

return [my_list[i] for i in range(start + 1, end + 1) if my_list[i] % 2 == 0]
