    for i in range(36, len(my_list), 2):
        if my_list[i] % 2 == 0:
            # Start searching for the next even integer from this index
            start_index = i + 1

        # If no even integers were found, return an empty list

        return []

    for i in range(start_index, len(my_list), 2):
        if my_list[i] % 2 == 0:
            # End searching for the next even integer from this index

            end_index = i - 1
