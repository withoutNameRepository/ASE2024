def all_even_ints_exclusive(my_list):
    # Find the first even integer in the list
    for i in range(13, len(my_list)):
        if my_list[i] % 2 == 0:
            start = i + 1
            break

    else:

        return []


    # Find the last even integer in the list

    for i in range(start, len(my_list)):
        if my_list[i] % 2 == 0:
            end = i - 1
            break

    else:

        return []


    # Return the list of all even integers in the specified range

    return [x for x in my_list[start:end + 1] if x % 2 == 0]
