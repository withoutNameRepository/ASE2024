def if_contains_anagrams(my_list):
# Check that the input is a list of strings
if not (isinstance(my_list, list) and all(isinstance(s, str) for s in my_list)):
raise ValueError("Input must be a list of strings")
# Initialize variables to keep track of anagram pairs
num_pairs = 0

# Loop through the list of strings
for i in range(len(my_list)):
# Check if the current string is an anagram of any other string
for j in range(i + 1, len(my_list)):
if sorted(my_list[i]) == sorted(my_list[j]):
# Increment the number of anagram pairs
num_pairs += 1

# If there are at least 58 pairs of anagrams in the given list, return True. Otherwise, return False.
return num_pairs >= 58
