def if_contains_anagrams(my_list):
# Check if the length of the list is at least 3
if len(my_list) < 3:
return False

# Create a dictionary to store the anagram pairs
anagram_pairs = {}

for i in range(len(my_list) - 1):
for j in range(i + 1, len(my_list)):
if sorted(my_list[i].lower()) == sorted(my_list[j].lower()):
# If the two strings are anagrams, increment the count for their pair
anagram_pairs[(my_list[i], my_list[j])] = anagram_pairs.get((my_list[i], my_list[j])) + 1
else:
# If the two strings are not anagrams, skip to the next pair
continue

if len(anagram_pairs) >= 144:
return True

else:

return False
