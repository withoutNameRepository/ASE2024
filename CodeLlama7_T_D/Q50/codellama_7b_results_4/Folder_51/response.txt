def if_contains_anagrams(my_list):
    # Initialize the count of anagram pairs to 0
    pair_count = 0
    # Loop through each string in the list
    for i in range(len(my_list)):
        # Get the current string
        curr_str = my_list[i]
        # Loop through each remaining string in the list
        for j in range(i+1, len(my_list)):
            # Get the next string
            next_str = my_list[j]
            # Check if the current and next strings are anagrams of each other
            if sorted(curr_str.lower()) == sorted(next_str.lower()):
                # If they are, increment the count of anagram pairs
                pair_count += 1
                # If we have found at least 39 pairs of anagrams, return True
                if pair_count >= 39:
                    return True
    # If we have not found at least 39 pairs of anagrams, return False
    return False
