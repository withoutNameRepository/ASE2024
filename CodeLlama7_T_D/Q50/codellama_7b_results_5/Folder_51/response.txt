def if_contains_anagrams(strings):
    # count the number of anagram pairs in the list
    pair_count = 0
    for i in range(len(strings) - 1):
        for j in range(i + 1, len(strings)):
            # check if the two strings are anagrams
            if sorted(strings[i].lower()) == sorted(strings[j].lower()):
                pair_count += 1

    # return true if there are at least 39 pairs of anagrams in the list, otherwise return false
    return pair_count >= 39
