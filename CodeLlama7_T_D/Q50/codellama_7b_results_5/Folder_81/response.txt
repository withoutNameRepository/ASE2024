def if_contains_anagrams(str_list):
    # Get the length of the list
    len = len(str_list)
    # Initialize a counter for the number of pairs
    pair_count = 0
    # Loop through each string in the list
    for i in range(len):
        # Get the current string
        str1 = str_list[i]
        # Loop through each remaining string after the current one
        for j in range(i + 1, len):
            # Get the second string
            str2 = str_list[j]
            # Check if the strings are anagrams
            if sorted(str1.lower()) == sorted(str2.lower()):
                # Increment the pair counter
                pair_count += 1
    # Return whether there are at least 16 pairs of anagrams in the list
    return pair_count >= 16
