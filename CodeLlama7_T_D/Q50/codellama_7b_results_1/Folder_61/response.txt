def if_contains_anagrams(my_list):
    # create a dictionary to store the anagram counts
    anagram_counts = {}

    for word in my_list:
        # convert the word to lowercase
        word = word.lower()

        # check if the word is at least three characters long and if it contains at least two vowels
        if len(word) >= 3 and word.count('a') + word.count('e') + word.count('i') + word.count('o') + word.count('u') >= 2:
            # sort the characters in the word and create a string from it
            sorted_chars = ''.join(sorted(word))

            # check if the sorted string is already in the dictionary of anagram counts
            if sorted_chars in anagram_counts:
                # increment the count for this anagram
                anagram_counts[sorted_chars] += 1

            else:
                # add a new entry to the dictionary with a count of one
                anagram_counts[sorted_chars] = 1

    # check if there are at least eight pairs of anagrams in the list
    return len(anagram_counts) >= 8
