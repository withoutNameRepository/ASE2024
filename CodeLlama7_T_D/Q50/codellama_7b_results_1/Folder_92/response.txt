def if_contains_anagrams(my_list):
    # check if there are at least two pairs of anagrams in the list
    num_pairs = 0
    for i in range(len(my_list) - 1):
        for j in range(i + 1, len(my_list)):
            if sorted(my_list[i]) == sorted(my_list[j]):
                num_pairs += 1
    # check if there are at least two pairs of anagrams with a length of at least three
    num_long_anagrams = 0
    for i in range(len(my_list)):
        for j in range(i + 1, len(my_list)):
            if sorted(my_list[i]) == sorted(my_list[j]):
                if len(my_list[i]) >= 3:
                    num_long_anagrams += 1
    # return true if there are at least two pairs of anagrams with a length of at least three, otherwise return false
    if num_pairs >= 2 and num_long_anagrams >= 2:
        return True
    else:
        return False
