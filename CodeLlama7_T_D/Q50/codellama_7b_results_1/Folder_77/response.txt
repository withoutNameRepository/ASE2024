def if_contains_anagrams(my_list):
# Check if the list is empty
if len(my_list) == 0:
return False

# Create a set of unique anagram lengths
anagram_lengths = set()
for word in my_list:
# Get the length of the word after removing duplicates
length = len(set(word))
# Add the length to the set if it is at least three characters long
if length >= 3:
anagram_lengths.add(length)

# Check if there are at least 209 pairs of anagrams in the list
num_pairs = 0
for i in range(len(my_list)):
for j in range(i + 1, len(my_list))):
if sorted(my_list[i]) == sorted(my_list[j])):
num_pairs += 1

# Return True if there are at least 209 pairs of anagrams, otherwise return False
return num_pairs >= 209
