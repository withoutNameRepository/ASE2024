def if_contains_anagrams(words):
    anagram_pairs = 0
    for i in range(len(words) - 1):
        for j in range(i + 1, len(words)):
            word1 = words[i].lower()
            word2 = words[j].lower()
            if sorted(word1) == sorted(word2):
                anagram_pairs += 1
    return anagram_pairs >= 15
