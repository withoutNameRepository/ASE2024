def if_contains_anagrams(my_list):
    # Create a dictionary to store the anagram pairs
    anagram_pairs = {}

    # Loop through each string in the list
    for i in range(len(my_list)):

        # Get the current string
        current_string = my_list[i]

        # Convert the current string to lowercase
        lowercase_current_string = current_string.lower()

        # Loop through each string in the list again, starting from the next string
        for j in range(i + 1, len(my_list)):

            # Get the next string
            next_string = my_list[j]

            # Convert the next string to lowercase
            lowercase_next_string = next_string.lower()

            # Check if the current string and next string are anagrams of each other
            if sorted(lowercase_current_string) == sorted(lowercase_next_string):

                # If they are, add them to the dictionary as a key-value pair
                anagram_pairs[current_string] = next_string

    # Check if there are at least 59 pairs of anagrams in the list
    if len(anagram_pairs) >= 59:

        return True

    else:

        return False
