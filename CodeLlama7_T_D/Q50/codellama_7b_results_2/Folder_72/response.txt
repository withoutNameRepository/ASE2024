def if_contains_anagrams(my_list):
    # Get the number of elements in the list
    num_elements = len(my_list)
    # Initialize a counter for the number of anagram pairs found
    num_pairs = 0
    # Iterate over each element in the list
    for i in range(num_elements):
        # Get the current element
        element_i = my_list[i]
        # Iterate over each element in the list that comes after the current element
        for j in range(i + 1, num_elements):
            # Get the current element
            element_j = my_list[j]
            # Check if the elements are anagrams of each other
            if sorted(element_i) == sorted(element_j):
                # Increment the counter for the number of anagram pairs found
                num_pairs += 1
    # Return true if there are at least 55 pairs of anagrams in the list; otherwise, return false
    return num_pairs >= 55
