def if_contains_anagrams(my_list):
    # Initialize variables
    pair_count = 0
    anagram_length = 3

    # Loop through the list of strings
    for i in range(len(my_list) - 1)):
        # Loop through the remaining strings in the list

            if sorted(my_list[i]) == sorted(my_list[j])):
