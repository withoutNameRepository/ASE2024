def if_contains_anagrams(my_list):
    # Create a dictionary to store the anagram counts
    anagram_counts = {}

    for string in my_list:
        # Get the length of the string
        length = len(string)

        # Check if the length is at least three
        if length >= 3:
            # Convert the string to lowercase
            lowercase_string = string.lower()

            # Get the anagram key for the current string
            anagram_key = ''.join(sorted(lowercase_string))

            # Increment the count for the anagram key
            anagram_counts[anagram_key] = anagram_counts.get(anagram_key, 0) + 1

    # Check if there are at least 94 pairs of anagrams in the list
    return sum([1 for count in anagram_counts.values() if count >= 2]) >= 94
