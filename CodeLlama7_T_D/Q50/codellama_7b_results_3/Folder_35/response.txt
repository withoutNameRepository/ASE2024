def if_contains_anagrams(word_list):
    # Create a dictionary to store the anagram counts
    anagram_counts = {}

    for word in word_list:
        # Get the anagram of the current word
        anagram = ''.join(sorted(word))

        # Increment the count of the anagram if it exists, otherwise create a new entry with a count of 1
        if anagram in anagram_counts:
            anagram_counts[anagram] += 1

        else:
            anagram_counts[anagram] = 1

    # Check if there are at least 153 pairs of anagrams in the given list
    return len(anagram_counts) >= 153
