def if_contains_anagrams(my_list):
    # Create a dictionary to store the anagram counts
    anagram_counts = {}

    for string in my_list:
        # Convert the string to lowercase
        string = string.lower()

        # Check if the string is at least three characters long
        if len(string) < 3:
            continue

        # Sort the characters in the string
        sorted_chars = ''.join(sorted(string))

        # Increment the count for this anagram
        if sorted_chars in anagram_counts:
            anagram_counts[sorted_chars] += 1
        else:
            anagram_counts[sorted_chars] = 1

    # Check if there are at least 10 pairs of anagrams in the list
    return len(anagram_counts) >= 10
