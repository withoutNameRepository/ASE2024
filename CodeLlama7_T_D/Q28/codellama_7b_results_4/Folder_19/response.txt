def return_nth_smallest_ascii(string):
    # Initialize a dictionary to store the frequency of each character
    freq = {}

    # Iterate through the string and count the frequency of each character
    for char in string:
        if char not in freq:
            freq[char] = 1

        else:
            freq[char] += 1

    # Sort the dictionary by value (frequency) and then key (character)
    sorted_freq = sorted(freq.items(), key=lambda x: (x[1], x[0])))

    # Return the character with the 13th smallest ASCII value
    return chr(sorted_freq[12][0]))
