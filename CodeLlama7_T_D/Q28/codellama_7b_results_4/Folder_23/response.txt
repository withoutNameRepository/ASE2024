 def return_nth_smallest_ascii(string):
    # Create a list of all the characters in the string
    char_list = list(string)
    # Find the indices of the first and last characters in the string

    first_index = char_list.index('a')
    last_index = char_list.index('z')
    # Create a new list with only the characters between the first and last indices, inclusive
    sub_list = char_list[first_index:last_index+1]
    # Find the ASCII value of the nth smallest character in the sublist
    nth_smallest = sub_list[6]
    # Return the nth smallest character as a string
    return nth_smallest
return_nth_smallest_ascii("hello")
