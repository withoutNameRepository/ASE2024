def return_nth_smallest_ascii(string):
# Convert the string to a list of characters
char_list = list(string)
# Create a dictionary to store the ASCII values of each character
ascii_dict = {}
# Iterate through the list of characters and store the ASCII value in the dictionary
for char in char_list:
ascii_value = ord(char)
ascii_dict[char] = ascii_value
# Sort the dictionary by the ASCII values in descending order
sorted_dict = sorted(ascii_dict.items(), key=lambda x: x[1], reverse=True)
# Return the 14th smallest ASCII value in the sorted dictionary
return sorted_dict[13][1]
