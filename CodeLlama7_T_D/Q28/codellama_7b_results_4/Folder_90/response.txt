 def return_nth_smallest_ascii(string):
    # Initialize a dictionary to store the character frequencies
    freq = {}

    # Iterate over the string and update the frequency dictionary
    for char in string:
        if char in freq:
            freq[char] += 1
        else:
            freq[char] = 1

    # Sort the frequency dictionary by value
    sorted_freq = sorted(freq.items(), key=lambda x: x[1], reverse=True)

    # Return the character with the 5th smallest ASCII value among those characters in the string that occur between indices 17 to 83, both inclusive.
    for i, (char, freq) in enumerate(sorted_freq):
        if i == 4 and 17 <= char <= 83:
            return char

    # If no character with the 5th smallest ASCII value is found between indices 17 to 83, both inclusive, then return None.
    return None
