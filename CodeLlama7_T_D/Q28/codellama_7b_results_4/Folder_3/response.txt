def return_nth_smallest_ascii(s):
    # Initialize a dictionary to store the frequency of each character
    freq = {}

    # Iterate over the string and increment the frequency of each character
    for c in s:
        if c in freq:
            freq[c] += 1
        else:
            freq[c] = 1

    # Sort the dictionary by value (frequency) in descending order
    sorted_freq = sorted(freq.items(), key=lambda x: x[1], reverse=True)

    # Return the character with the 11th smallest ASCII value
    return chr(sorted_freq[10][0])
