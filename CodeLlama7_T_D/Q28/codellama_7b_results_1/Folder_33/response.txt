def return_nth_smallest_ascii(string):
# Split the string into a list of characters
characters = list(string)
# Create a dictionary to store the character frequencies
freq = {}

# Iterate over the characters in the string
for char in characters:

    # If the character is already in the frequency dictionary, increment its count by 1
    if char in freq:
        freq[char] += 1

    # Otherwise, add the character to the frequency dictionary with a count of 1
    else:
        freq[char] = 1

# Sort the frequency dictionary by value (descending)
sorted_freq = sorted(freq.items(), key=lambda x: x[1], reverse=True)

# Get the character with the 12th smallest ASCII value among those characters in the string that occur between indices 22 to 74, both inclusive.
smallest_ascii = sorted_freq[11][0]

return smallest_ascii
