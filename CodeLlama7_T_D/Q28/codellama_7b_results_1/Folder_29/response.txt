def return_nth_smallest_ascii(string):
    # convert the string to a list of characters
    char_list = list(string)
    # find the indices of the characters in the range 5-10
    indices = [i for i, x in enumerate(char_list) if 5 <= i <= 10]
    # sort the characters by their ASCII values
    sorted_chars = sorted(char_list, key=ord)
    # return the nth smallest ASCII value
    return sorted_chars[indices[5]]
