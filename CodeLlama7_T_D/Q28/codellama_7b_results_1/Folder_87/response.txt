def return_nth_smallest_ascii(string):
    # Create a dictionary to store the character counts
    char_counts = {}

    # Iterate over the characters in the string
    for char in string:
        # If the character is already in the dictionary, increment its count
        if char in char_counts:
            char_counts[char] += 1

        # Otherwise, add the character to the dictionary with a count of 1
        else:
            char_counts[char] = 1

    # Sort the dictionary by value (ascending)
    sorted_counts = sorted(char_counts.items(), key=lambda x: x[1])

    # Return the character with the 5th smallest ASCII value
    return chr(sorted_counts[4][0])
