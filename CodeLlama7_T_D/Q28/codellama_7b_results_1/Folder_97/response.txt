def return_nth_smallest_ascii(string):
    # Initialize a list to store the ASCII values of the characters in the string
    ascii_values = []

    # Iterate through each character in the string

    for char in string:

        # Append the ASCII value of the current character to the list

        ascii_values.append(ord(char))

    # Sort the list of ASCII values in ascending order

    ascii_values.sort()

    # Return the 6th smallest ASCII value among the characters that occur between indices 15 to 21, both inclusive

    return ascii_values[14:22]
