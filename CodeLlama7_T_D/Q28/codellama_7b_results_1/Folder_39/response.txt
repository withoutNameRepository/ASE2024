def return_nth_smallest_ascii(string):
    # Convert the string to a list of characters
    char_list = list(string)
    # Find the indices of the characters in the list that occur between indices 1 and 66, both inclusive
    indices = [i for i, x in enumerate(char_list[1:66]) if x in char_list]
    # Sort the indices in ascending order based on the ASCII value of the corresponding characters
    sorted_indices = sorted(indices, key=lambda x: ord(string[x+1]))
    # Return the 8th smallest ASCII value among the characters at the sorted indices
    return string[sorted_indices[7]]
