def return_nth_smallest_ascii(string):
    # Get the characters between indices 8 and 26
    substring = string[8:27]
    # Create a list of all the ASCII values in the substring
    ascii_values = [ord(char) for char in substring]
    # Sort the list in ascending order
    sorted_list = sorted(ascii_values)
    # Return the 6th smallest ASCII value in the sorted list
    return sorted_list[5]
