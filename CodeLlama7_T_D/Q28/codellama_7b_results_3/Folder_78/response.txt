def return_nth_smallest_ascii(string):
    # Extract the characters between indices 0 and 87
    sub_string = string[0:87]
    # Create a list of ASCII values for each character in the sub-string
    ascii_list = [ord(char) for char in sub_string]
    # Sort the list of ASCII values in ascending order
    sorted_ascii_list = sorted(ascii_list)
    # Return the 14th smallest ASCII value from the sorted list
