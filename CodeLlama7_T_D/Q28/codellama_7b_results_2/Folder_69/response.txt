def return_nth_smallest_ascii(s):
    # Count the frequency of each character in the string
    freq = {}
    for c in s:
        if c not in freq:
            freq[c] = 1
        else:
            freq[c] += 1
    # Find the character with the 7th smallest ASCII value
    ascii_values = []
    for c, f in freq.items():
        for i in range(f):
            ascii_values.append(ord(c))
    nth_smallest_ascii = sorted(ascii_values)[6]
    return chr(nth_smallest_ascii)
