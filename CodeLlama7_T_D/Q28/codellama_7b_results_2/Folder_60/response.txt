def return_nth_smallest_ascii(s):
    # Get the indices of the characters between 36 and 79
    indices = [i for i, c in enumerate(s) if 36 <= i <= 79]
    # Sort the characters by their ASCII values
    sorted_chars = sorted(s[i] for i in indices))
    # Return the character with the 12th smallest ASCII value
    return sorted_chars[12-1]
