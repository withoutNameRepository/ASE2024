def return_nth_smallest_ascii(s):
	# Create a list of all the characters in the string
	char_list = []
	for i in range(3, 27):
		char_list.append(s[i])
	# Sort the list of characters by their ASCII values
	sorted_list = sorted(char_list, key=ord)
