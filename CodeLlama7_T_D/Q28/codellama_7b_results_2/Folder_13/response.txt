def return_nth_smallest_ascii(s):
    # Create a dictionary to keep track of the character frequencies
    freq = {}
    for c in s:
        if c in freq:
            freq[c] += 1
        else:
            freq[c] = 1
    # Sort the frequency dictionary by value
    sorted_freq = sorted(freq.items(), key=lambda x: x[1], reverse=True)
    # Return the 4th smallest ASCII value among the characters in the string that occur between indices 47 to 80, both inclusive
    return sorted_freq[3][0]
