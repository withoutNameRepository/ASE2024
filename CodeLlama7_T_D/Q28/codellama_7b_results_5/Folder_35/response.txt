def return_nth_smallest_ascii(string):
    # Find the characters that occur between indices 50 to 75
    char_list = []
    for i in range(50, 76):
        if string[i] not in char_list:
            char_list.append(string[i])
    # Sort the list of characters based on their ASCII values
    sorted_char_list = sorted(char_list, key=ord)
    # Return the 7th smallest ASCII value among the characters
    return sorted_char_list[6]
