def return_nth_smallest_ascii(s):
    # Find the indices of the characters in the string
    indices = [i for i, c in enumerate(s) if c != ""]
    # Filter the indices to include only those between 17 and 86 (inclusive)
    filtered_indices = [index for index in indices if 17 <= index <= 86]
    # Sort the filtered indices by the ASCII value of the corresponding character
    sorted_indices = sorted(filtered_indices, key=lambda x: ord(s[x]))
    # Return the 15th smallest ASCII value among the characters in the string that occur between indices 17 and 86 (inclusive)
    return s[sorted_indices[14]]
