def return_nth_smallest_ascii(string):
    # Find the indices of the characters in the string
    char_indices = {char: i for i, char in enumerate(string)}
    # Get the ASCII values of all the characters in the string
    ascii_values = [ord(char) for char in string]
    # Sort the ASCII values in ascending order
    sorted_ascii_values = sorted(ascii_values)
    # Find the index of the 6th smallest ASCII value
    sixth_smallest_index = len(sorted_ascii_values) // 6
    # Return the character with the 6th smallest ASCII value
    return string[char_indices[sorted_ascii_values[sixth_smallest_index]]]]
