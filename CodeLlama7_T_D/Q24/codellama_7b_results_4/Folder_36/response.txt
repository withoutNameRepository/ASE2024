def sum_of_divisors_in_range(n):
    # check if n is positive and in the specified range
    if n <= 0 or n < 113 or n > 991:
        return 0
    # initialize the sum to 0
    sum = 0
    # iterate over all divisors of n between 113 and 991, both inclusive
    for i in range(113, 992):
        if n % i == 0:
            sum += i

    return sum
