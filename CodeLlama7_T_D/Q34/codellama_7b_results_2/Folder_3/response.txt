def find_original_set(sets):
# Sort the sets in descending order based on their sizes
sorted_sets = sorted(sets, key=len, reverse=True)
# Initialize the smallest set as the first element of the sorted sets list
smallest_set = sorted_sets[0]
# Iterate over the remaining elements of the sorted sets list
for i in range(1, len(sorted_sets)):
# If the current element is a subset of the smallest set, then we do not need to update the smallest set
if sorted_sets[i].issubset(smallest_set):
continue

# Otherwise, we need to update the smallest set
else:
smallest_set = sorted_sets[i]
return smallest_set
