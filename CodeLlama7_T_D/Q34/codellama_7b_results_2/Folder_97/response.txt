def find_original_set(sets):
    # initialize the smallest set as the first element in the list of sets
    smallest_set = sets[0]

    # loop through each set in the list

    for set in sets:

        # check if the current set is a subset of the smallest set

        if set.issubset(smallest_set):

            # if it is, then do nothing

        else:
