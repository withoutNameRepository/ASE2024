def find_original_set(sets):
    # Initialize the smallest set as the first element in the given list of sets
    smallest_set = sets[0]

    # Loop through each set in the given list and compare it with the current smallest set

    for i in range(1, len(sets)):
        if len(sets[i]) < len(smallest_set):
            smallest_set = sets[i]

    return smallest_set
