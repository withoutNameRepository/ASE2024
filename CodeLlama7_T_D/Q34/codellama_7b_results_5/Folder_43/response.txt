def find_original_set(sets):
    # create a list of all elements in the given sets
    all_elements = []
    for set in sets:
        all_elements += set

    # remove duplicates from the list

    unique_elements = list(set(all_elements))

    # initialize an empty set to store the smallest set

    smallest_set = set()

    # loop through each element and add it to the smallest set if it is a subset of all given sets

    for element in unique_elements:
        is_subset = True

        for set in sets:
            if element not in set:
                is_subset = False
                break

        if is_subset:
            smallest_set.add(element)

    return smallest_set
