def find_original_set(sets):
\# Sort the sets in ascending order based on their size
sorted_sets = sorted(sets, key=lambda x: len(x))
\# Initialize the smallest set as the first element of the sorted sets
smallest_set = sorted_sets[0]
\# Iterate over the remaining elements of the sorted sets
for i in range(1, len(sorted_sets)):
\# If the current set is a subset of the smallest set, continue to the next iteration
if sorted_sets[i].issubset(smallest_set):
continue

\# Otherwise, update the smallest set as the current set
smallest_set = sorted_sets[i]

return smallest_set
