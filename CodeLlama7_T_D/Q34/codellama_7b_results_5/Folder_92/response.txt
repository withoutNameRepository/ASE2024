def find_original_set(set1, set2, set3, set4, set5, set6, set7):
    # Create a list of all the given sets
    sets = [set1, set2, set3, set4, set5, set6, set7]
    # Initialize an empty set as the smallest set
    smallest_set = set()

    for i in range(len(sets)):
        # Check if each given set is a subset of the smallest set
        if all(item in smallest_set for item in sets[i]):
            # If it is, continue to the next set
            continue

        else:
            # If it's not, add the current set to the smallest set
            smallest_set = smallest_set.union(sets[i])

    return smallest_set
