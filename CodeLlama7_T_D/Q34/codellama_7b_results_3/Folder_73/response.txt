def find_original_set(sets):
    # Initialize the smallest set as the first element in the input list
    smallest_set = sets[0]

    # Iterate through each of the given arguments

    for i in range(1, len(sets)):

        # If the current set is smaller than the smallest set so far, update the smallest set

        if len(sets[i]) < len(smallest_set):
            smallest_set = sets[i]

    # Return the smallest set

    return smallest_set
