def find_original_set(sets):
    # Initialize an empty set
    original_set = set()

    # Iterate through each set in the input argument
    for s in sets:
        # Add each element of the current set to the original set
        for e in s:
            original_set.add(e)

    # Return the smallest set that contains all elements from the input sets
    return original_set
