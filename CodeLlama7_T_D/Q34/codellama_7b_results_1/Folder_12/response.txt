def find_original_set(sets):
    # Initialize the smallest set as the first set in the input
    smallest_set = sets[0]
    # Loop through each set in the input and compare it to the current smallest set
    for set in sets:
        # If the current set is smaller than the current smallest set, update the smallest set
        if len(set) < len(smallest_set):
            smallest_set = set

    # Return the smallest set as the original set
    return smallest_set
