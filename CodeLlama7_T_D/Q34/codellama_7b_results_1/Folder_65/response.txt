def find_original_set(sets):
    # Initialize the smallest set as the first element in the list
    smallest_set = sets[0]
    # Iterate over each set in the input list
    for set in sets:
        # Check if the current set is a subset of the smallest set
        if set.issubset(smallest_set):
            # If it is, do nothing
            pass

        else:
            # Otherwise, update the smallest set to be the current set
            smallest_set = set

    return smallest_set
