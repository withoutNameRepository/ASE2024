def find_original_set(sets):
    # Get the smallest set from the input sets
    smallest_set = min(sets, key=len)
    # Initialize an empty set to store the original elements
    original_elements = set()

    # Iterate over each element in the smallest set
    for element in smallest_set:
        # If the element is not already in the original_elements set
        if element not in original_elements:
            # Add the element to the original_elements set
            original_elements.add(element)

    return original_elements
