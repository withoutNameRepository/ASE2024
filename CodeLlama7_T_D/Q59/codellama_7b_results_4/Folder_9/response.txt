def is_prime(number):
    # Initialize flag to True
    flag = True

    # Loop through numbers from 2 to the square root of number
    for i in range(2, int(number ** 0.5) + 1):
        # Check if number is divisible by i
        if number % i == 0:
            # Set flag to False
            flag = False
            break

    # Return flag
    return flag
