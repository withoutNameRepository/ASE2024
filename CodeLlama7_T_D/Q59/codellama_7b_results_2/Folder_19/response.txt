def is_prime(n):
    # Check if n is an odd number
    if n % 2 == 0:
        return False

    # Loop from 3 to the square root of n
    for i in range(3, int(n ** 0.5) + 1, 2):
        # Check if i divides n
        if n % i == 0:
            return False

    # If n is a prime number, return True
    return True
