def isPrime(n):

    if n<2:

        return False

    for i in range(2,int(n**0.5)+1):

        if n%i==0:

            return False

    return True


def isLeftTruncatable(n,x):

    s = str(n)

    for i in range(len(s)):

        if int(s[:i])>x:

            return False

    return True
