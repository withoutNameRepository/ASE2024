def is_prime(n):
    # Check if n is greater than 1
    if n > 1:
        # Loop through all numbers from 2 to n-1
        for i in range(2, n):
            # If the result of dividing n by i is not equal to zero, then n is not prime
            if n % i == 0:
                return False

    return True
