def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True

def is_left_truncatable_prime(n):
    if str(n)[-1] == "0":
        return False

    for i in range(len(str(n)) - 2, -1, -1):
        if not is_prime(int(str(n)[:i + 1])):
            return False

    return True
