def is_prime(n):
    if n == 1 or n == 0:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True

def is_left_truncatable_prime(n):
    string = str(n)

    for i in range(len(string)):

        if not is_prime(int(string[:i] + string[i+1:])):
            return False

    return True
