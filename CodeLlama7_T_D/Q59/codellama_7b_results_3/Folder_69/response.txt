def is_prime(n):

# Check if n is a positive integer
if type(n) != int or n < 2:
return False

# Check if n is divisible by any number between 2 and sqrt(n)
for i in range(2, int(n ** 0.5) + 1):
if n % i == 0:
return False

return True
