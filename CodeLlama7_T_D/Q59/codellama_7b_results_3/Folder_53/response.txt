def is_prime(n):
if n < 2:
return False

for i in range(2, int(n ** 0.5) + 1):
if n % i == 0:
return False

return True

def left_truncatable_prime(n):
str_n = str(n)
for i in range(len(str_n)):
temp = int(str_n[:i] + str_n[i+1:]))
if temp == 0 or not is_prime(temp):
return False

return True
