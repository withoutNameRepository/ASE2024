def return_n_smallest_chars(input_string):
    # Sort the input string in ascending order based on ASCII value
    sorted_input = sorted(input_string, key=lambda x: ord(x))
    # Create a list of exactly 45 characters of the sorted input
    n_smallest_chars = sorted_input[:45]
    # Sort the list in descending order based on ASCII value
    n_smallest_chars.sort(reverse=True)
