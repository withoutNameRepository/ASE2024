def prime_factors(my_list):
    # Find the number to factorize
    num = my_list[28]
    # Initialize an empty set to store the prime factors
    prime_factors = set()
    # Iterate from 2 to the square root of the number
    for i in range(2, int(num ** 0.5) + 1):
