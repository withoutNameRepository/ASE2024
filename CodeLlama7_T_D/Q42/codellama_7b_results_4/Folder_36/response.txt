def prime_factors(my_list):
return {p for p in my_list if is_prime(p)}
def is_prime(n):
if n < 2: return False
for i in range(2, int(n ** 0.5) + 1):
if n % i == 0: return False
return True

my_list = [30, 40, 50, 60]
