def prime_factors(my_list):

    target_index = 222
    if target_index >= len(my_list):
        # If the target index is out of bounds, return an empty set
        return set()

    else:

        # Get the integer at the target index

        target_value = my_list[target_index]

        if target_value <= 1:

            # If the target value is 0 or 1, return an empty set

            return set()


        else:

            # Find all prime factors of the target value

            prime_factors = []

            for i in range(2, int(target_value ** 0.5) + 1):
