def prime_factors(my_list):
    # Check if input is a list
    if not isinstance(my_list, list):
        raise ValueError("Input must be a list")
    # Get the element at index 66
    element = my_list[66]
    # Check if the element is a prime number
    if not is_prime(element):
        raise ValueError("Element at index 66 must be a prime number")
    # Return the set of all prime factors of the element
