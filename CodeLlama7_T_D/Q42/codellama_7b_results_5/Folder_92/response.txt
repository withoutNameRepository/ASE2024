def prime_factors(my_list):
    if len(my_list) == 0:
        return set()
    else:
        prime_factor = my_list[0]
        for i in range(1, len(my_list)):
            if my_list[i] % prime_factor == 0:
                break

        else:

            return {prime_factor} | prime_factors(my_list[1:])
