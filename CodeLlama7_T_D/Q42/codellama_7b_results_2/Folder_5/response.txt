def prime_factors(my_list):
    # Calculate the product of all elements in the list
    product = 1
    for i in my_list:
        product *= i

    # Initialize an empty set to store the prime factors

    prime_factors = set()

    # Iterate through the list of integers and check if each element is a prime factor of the product

    for i in my_list:

        if product % i == 0:

            prime_factors.add(i)

    return prime_factors
