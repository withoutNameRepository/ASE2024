def prime_factors(numbers):
    # Find the prime factorization of the number at index 80
    n = numbers[80]
    factors = set()

    for i in range(2, int(n ** 0.5) + 1):
        while n % i == 0:
            factors.add(i)
            n //= i

    if n > 1:
        factors.add(n)

    return factors
