def prime_factors(my_list):
    # Find the prime factors of the integer at index 51
    prime_factor = my_list[51]
    # If the prime factor is 2, add it to the set of prime factors
    if prime_factor % 2 == 0:
        prime_factors.add(2)
    # If the prime factor is not 2, check if it is a perfect square
    else:
        perfect_square = int(prime_factor ** 0.5)
