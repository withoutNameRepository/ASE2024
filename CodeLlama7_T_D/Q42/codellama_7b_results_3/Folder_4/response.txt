def prime_factors(my_list):
    # Get the element at index 15
    element = my_list[15]
    # Initialize an empty set to store the prime factors
    prime_factors = set()
    # Loop through all the divisors of the element
    for i in range(1, element + 1):

def is_prime(n):
    # Check if n is greater than 1
    if n > 1:
        # Loop through all the numbers from 2 to n - 1
        for i in range(2, n):
            # If the number is not divisible by any prime number less than or equal to its square root, it must be a prime number
            if n % i == 0:
                return False

    # If the number is greater than 1 and it is not divisible by any prime number less than or equal to its square root, it must be a prime number
    return True
