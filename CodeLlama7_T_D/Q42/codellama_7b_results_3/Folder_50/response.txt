def prime_factors(my_list):
    # Get the integer at index 40 from the list
    num = my_list[40]
    # Initialize an empty set to store the prime factors
    prime_factor_set = set()
    # Iterate through the divisors of the number
    for i in range(1, num + 1):
        # If the current divisor is a prime number, add it to the set of prime factors
        if num % i == 0 and is_prime(i):
            prime_factor_set.add(i)
    return prime_factor_set

def is_prime(num):
    # Check if the number is less than or equal to 1
    if num <= 1:
        return False

    # Check if the number is divisible by 2 to n, where n is the square root of the number
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False

    # If none of the above conditions are true, then the number is prime
    return True
