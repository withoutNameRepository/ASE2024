def all_pos_ints_exclusive(my_list):
    # Find the first positive integer in the list
    first_pos_int = next((x for x in my_list if x > 0), None)
    # If no positive integers exist in the specified range, return an empty list
    if first_pos_int is None:
        return []

    # Find the last index of the list that contains a positive integer

    last_index = len(my_list) - 1
    while my_list[last_index] <= 0:
        last_index -= 1

    # Return the list of all positive integers from index 15 to index 99, both exclusive

    return [x for x in my_list[15:99] if x > 0]
