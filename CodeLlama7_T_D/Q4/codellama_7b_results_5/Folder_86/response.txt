def all_pos_ints_exclusive(my_list):
    # Find the first positive integer in the list
    first_pos_int = next((x for x in my_list if x > 0), None)
    # If no positive integers exist in the specified range, return an empty list
    if first_pos_int is None:
        return []

    # Find the last positive integer in the list
    last_pos_int = next((x for x in reversed(my_list) if x > 0), None)
    # If no positive integers exist in the specified range, return an empty list
    if last_pos_int is None:
        return []

    # Find the index of the first positive integer
    first_pos_int_index = my_list.index(first_pos_int)

    # Find the index of the last positive integer
    last_pos_int_index = len(my_list) - my_list[::-1].index(last_pos_int) - 1

    # Return the list of all positive integers from the specified range, both exclusive
    return my_list[first_pos_int_index + 1: last_pos_int_index]
