def all_pos_ints_exclusive(my_list):
    # Create a slice of the list starting from index 48 and ending at index 74, both exclusive
    my_slice = my_list[48:74]
    # Initialize an empty list to store the positive integers found in the slice
    pos_ints = []
    # Iterate over the elements of the slice
    for elem in my_slice:
        # Check if the element is a positive integer
        if isinstance(elem, int) and elem > 0:
            # If it is a positive integer, add it to the list of positive integers found in the slice
            pos_ints.append(elem)
    # Return the list of all positive integers found in the specified range, or an empty list if no such integers exist
    return pos_ints
