def all_pos_ints_exclusive(my_list):
    # Get the range of positive integers in the list
    pos_int_range = range(10, 100)
    # Create an empty list to store the result
    result = []
    # Loop through the range of positive integers
    for i in pos_int_range:
        # Check if the current index exists in the list and is a positive integer
        if i in my_list and my_list[i] > 0:
