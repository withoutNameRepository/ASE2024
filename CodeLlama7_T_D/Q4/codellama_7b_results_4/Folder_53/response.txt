def all_pos_ints_exclusive(my_list):
    # Initialize an empty list to store the positive integers
    pos_ints = []
    # Iterate over the input list
    for i in range(100, 200):
        # Check if the current element is a positive integer
        if my_list[i] > 0:

    return pos_ints
