def all_pos_ints_exclusive(my_list):
    # Find the first positive integer in the list
    first_pos_int = next((x for x in my_list if x > 0), None)
    # If no positive integers exist in the list, return an empty list
    if first_pos_int is None:
        return []

    # Initialize the output list with the first positive integer

    output = [first_pos_int]

    # Iterate over the rest of the list, starting from the index after the first positive integer

    for i in range(first_pos_int + 1, len(my_list)):
        if my_list[i] > 0:
            output.append(my_list[i])

    return output
