def find_subset_of_length_n(s):
    return len(powerset(s))

def powerset(s):

    result = []

    for i in range(1 << len(s)):

        subset = []

        for j in range(len(s)):

            if (i & (1 << j)):

                subset.append(s[j])

        result.append(tuple(subset))

    return result
