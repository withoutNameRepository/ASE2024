def find_subset_of_length_n(set_of_elements):
    # Initialize the number of subsets to 0
    num_subsets = 0
    # Iterate over all possible subsets of size 49
    for subset in itertools.combinations(set_of_elements, 49):
        # Increment the number of subsets if it has at least one element
        if len(subset) > 0:
            num_subsets += 1

    # Return the number of all its subsets of size 49
    return num_subsets
