def find_subset_of_length_n(set):
  # Initialize a variable to keep track of the number of subsets
  num_subsets = 0

  # Loop through all possible combinations of elements in the set
  for i in range(2 ** len(set)):

    # Convert the binary representation of the combination to a list of booleans, where each boolean represents whether the corresponding element is included in the subset or not.
    subset = [bool(i & (1 << j))) for j in range(len(set))]

    # If the subset has 87 elements, increment the number of subsets counter
    if sum(subset) == 87:
      num_subsets += 1

  # Return the number of all its subsets of size 87
  return num_subsets
