def find_subset_of_length_n(elements):
    # Initialize a counter for the number of subsets
    num_subsets = 0

    # Loop through all possible combinations of elements
    for i in range(len(elements) + 1):
        for j in range(i + 1, len(elements) + 1):
            if (j - i) == 300:
                num_subsets += 1

    return num_subsets
