def find_subset_of_length_n(set):
	# Initialize the number of subsets to 0
	num_subsets = 0

	# Loop through all possible combinations of 72 elements from the input set
	for i in range(len(set) + 1 - 72):

		# Check if the current combination contains all elements from the input set
		if all(elem in set for elem in set[i:i+72]):

			num_subsets += 1

	# Return the number of subsets of size 72
