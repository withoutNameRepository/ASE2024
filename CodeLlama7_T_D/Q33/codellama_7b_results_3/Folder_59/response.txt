 def return_vowels(my_string):
    # Initialize an empty list to store the vowel characters
    vowels = []
    # Loop through each character in the string
    for char in my_string:
        # Check if the current character is a vowel (a, e, i, o, u)
        if char in "aeiou":
            # If the current character is a vowel and it is greater than character Y and smaller than or equal to character s, add it to the list of vowels
            if char > "Y" and char <= "s":
                vowels.append(char)
    # Return the list of vowels that meet the condition
    return vowels
