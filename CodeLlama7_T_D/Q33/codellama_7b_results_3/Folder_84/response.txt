def return_vowels(my_string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 770
    end_index = 852
    found_vowels = []
    for i in range(start_index, end_index):
        char = my_string[i]
        if char.isalpha() and char.lower() in vowels:
            if char > 'b' and char <= 'i':
                found_vowels.append(char)
    return found_vowels
