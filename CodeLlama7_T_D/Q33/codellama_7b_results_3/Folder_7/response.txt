def return_vowels(my_string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 202
    end_index = 872
    filtered_list = []
    for char in my_string[start_index:end_index]:
        if char.isalpha():
            if char.lower() in vowels and char > '3' and char <= '8':
                filtered_list.append(char)
    return filtered_list
