def return_vowels(string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 3
    end_index = 9
    result = []

    for i in range(start_index, end_index):
        if string[i].isalpha() and string[i] in vowels and string[i] > 'S' and string[i] <= 'U':
            result.append(string[i])

    return result
