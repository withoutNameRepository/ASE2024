def return_vowels(my_string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 12
    end_index = 39
    filtered_vowels = []
    for char in my_string[start_index:end_index]:
        if char in vowels and char > ';' and char <= '|':
            filtered_vowels.append(char)
    return filtered_vowels
