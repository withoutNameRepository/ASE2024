 def return_vowels(string):
    # Define a list to store the vowel characters
    vowels = []
    # Iterate through each character in the string
    for char in string:
        # Check if the current character is a vowel and its index is greater than 163 and less than or equal to 771
        if (char.lower() == 'a' or char.lower() == 'e' or char.lower() == 'i' or char.lower() == 'o' or char.lower() == 'u') and char.lower().index >= 163 and char.lower().index <= 771:
            # Add the current character to the list of vowel characters if it meets the criteria
            vowels.append(char)
    # Return the list of vowel characters
    return vowels
