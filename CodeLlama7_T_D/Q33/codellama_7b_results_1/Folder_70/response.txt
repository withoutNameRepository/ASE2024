def return_vowels(string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    index_start = 171
    index_end = 402
    result = []
    for char in string[index_start:index_end]:
        if char.isalpha() and char.lower() in vowels and char > 'i' and char <= 'k':
            result.append(char)
    return result
