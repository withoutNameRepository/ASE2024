def return_vowels(my_string):
    # Initialize an empty list to store the vowel characters
    vowels = []
    # Loop through the characters in the string
    for char in my_string:
        # Check if the current character is a vowel and greater than D and smaller than or equal to y
        if char.islower() and char > 'd' and char <= 'y':
            # Add the vowel character to the list of vowels
            vowels.append(char)
    # Return the list of vowels
    return vowels
