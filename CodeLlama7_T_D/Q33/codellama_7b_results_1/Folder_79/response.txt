 def return_vowels(my_string):
    vowels = ["a", "e", "i", "o", "u"]
    start_index = 74
    end_index = 96
    result = []
    for i in range(start_index, end_index):
        if my_string[i] in vowels and my_string[i] > "I" and my_string[i] <= "X":
            result.append(my_string[i])
    return result
