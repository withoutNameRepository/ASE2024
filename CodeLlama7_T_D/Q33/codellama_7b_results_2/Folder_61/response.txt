def return_vowels(my_string):
    # Initialize an empty list to store the vowels
    vowels = []
    # Loop through each character in the string
    for char in my_string:
        # Check if the character is a vowel and is greater than 'C' and smaller than or equal to 'I'
        if char.isalpha() and (char > "C" and char <= "I"):
            # Add the vowel to the list
            vowels.append(char)
    # Return the list of vowels
    return vowels
