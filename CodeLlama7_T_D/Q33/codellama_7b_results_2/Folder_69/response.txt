def return_vowels(string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 641
    end_index = 872
    filtered_chars = []
    for i in range(start_index, end_index):
        if string[i] in vowels and string[i] > '>' and string[i] <= 'q':
            filtered_chars.append(string[i])
    return filtered_chars
