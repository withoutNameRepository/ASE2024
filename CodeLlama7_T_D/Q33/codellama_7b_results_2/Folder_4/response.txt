def return_vowels(my_string):
    vowels = ['a', 'e', 'i', 'o', 'u']
    start_index = 40
    end_index = 91
    result = []

    for i in range(start_index, end_index + 1):
        char = my_string[i]
        if char.islower() and char not in vowels:
            break

        elif char.isupper() and char in vowels:
            result.append(char)

    return result
