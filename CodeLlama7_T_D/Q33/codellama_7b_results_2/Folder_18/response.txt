def return_vowels(my_string):
    # Initialize an empty list to store the vowels
    vowels = []
    # Iterate through the characters in the string
    for char in my_string:
        # Check if the character is a vowel and greater than character M and smaller than or equal to character X
        if char.isalpha() and (char > 'M' and char <= 'X'):
            # Add the vowel to the list of vowels
            vowels.append(char)
    # Return the list of vowels found in the specified range
    return vowels
