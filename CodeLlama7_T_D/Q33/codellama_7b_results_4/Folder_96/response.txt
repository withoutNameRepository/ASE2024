def return_vowels(string):
    # Initialize an empty list to store the vowels
    vowels = []

    # Iterate through each character in the string
    for char in string:

        # Check if the current character is a vowel and its index is greater than 77 and smaller than or equal to 84
        if char.lower() in "aeiou" and char.lower().index(string) > 77 and char.lower().index(string) <= 84:

            # If the current character meets the above conditions, add it to the list of vowels
            vowels.append(char)

    # Return the list of vowels
    return vowels
