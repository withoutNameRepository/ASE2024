def return_vowels(my_string):
    # Initialize an empty list to store the vowel characters
    vowel_list = []
    # Iterate through the characters in the string
    for char in my_string:
        # Check if the character is a vowel and is greater than 'F' and smaller than or equal to 'h'
        if (char.lower() == 'a' or char.lower() == 'e' or char.lower() == 'i' or char.lower() == 'o' or char.lower() == 'u') and char > 'F' and char <= 'h':
            # If the condition is true, add the character to the list
            vowel_list.append(char)
    # Return the list of vowel characters that meet the specified conditions
    return vowel_list
