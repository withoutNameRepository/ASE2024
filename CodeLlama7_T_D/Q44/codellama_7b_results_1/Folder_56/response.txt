def composite_nums_between_indices(my_list):
    # Check if the input is a list of positive integers
    if not all(isinstance(x, int) and x > 0 for x in my_list)):
        raise ValueError("Invalid input. The function expects a list of positive integers.")
    # Initialize an empty set to store the composite numbers
    composite_nums = set()
    # Iterate over the list of integers
    for i in range(len(my_list)):
        # Check if the current number is prime
        if my_list[i] == 1 or any(my_list[i] % p == 0 for p in range(2, int(my_list[i] ** 0.5) + 1))):
            # If it is prime, move on to the next number
            continue
        else:
            # If it is not prime, add it to the set of composite numbers
            composite_nums.add(my_list[i])
    return composite_nums
