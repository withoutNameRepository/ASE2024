def is_composite(n):
    # Check if n is greater than 1
    if n > 1:
        # Iterate from 2 to the square root of n
        for i in range(2, int(n ** 0.5) + 1):
            # If i divides n with no remainder, return False
            if n % i == 0:
                return False

    # If n is a prime number greater than 1, return True
    return n > 1
