# Check if a number is composite by dividing it by all prime numbers less than or equal to its square root
def is_composite(n):
    for i in range(int(n**0.5), -1, -1):
        if n % i == 0:
            return True

    # If the number is not divisible by any prime numbers less than or equal to its square root, it is prime
    return False
