 def composite_nums_between_indices(my_list):
    # Initialize an empty set to store the composite numbers
    composite_nums = set()
    # Loop through the list of positive integers
    for i in range(len(my_list)):
        # Check if the current number is between index 661 and index 924 (both inclusive)
        if 661 <= i <= 924:
            # If the current number is a composite number, add it to the set of composite numbers
            if my_list[i] != 1 and my_list[i] % 2 != 0:
                composite_nums.add(my_list[i])
    # Return the set of all composite numbers between index 661 and index 924 (both inclusive), or an empty set if no such numbers exist
    return composite_nums
