def is_prime(n):

# Check if n is a positive integer
if not (isinstance(n, int) and n > 0):
raise ValueError("Input must be a positive integer")

# Return True if n is a prime number, False otherwise
for i in range(2, int(n ** 0.5) + 1):
if n % i == 0:
return False

return True
