# Define a function to check if a number is composite or not
def is_composite(n):
    # Check if n is divisible by any number between 1 and n-1
    for i in range(2, n):
        if n % i == 0:
            return False

    # If no number between 1 and n-1 divides n, it must be a prime number
    return True
