    # Create an empty set to store the composite numbers
    composite_nums = set()

    # Loop through the list of positive integers and check if each number is a composite number between index 0 and index 8, both inclusive
    for i in range(len(my_list)):

        # Check if the current number is a composite number
        if (num > 1) and (num % 2 != 0) or (num % 3 != 0):
