def is_composite(n):

    # If the number is less than 2, it is not composite
    if n < 2:
        return False

    # Check if n is divisible by any number between 2 and sqrt(n)
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return True

    return False
