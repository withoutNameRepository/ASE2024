# Define a function to check if a number is composite
def is_composite(n):
    # Check if n is divisible by any number between 2 and the square root of n
    for i in range(2, int(n ** 0.5) + 1)):
        if n % i == 0:
            return True

    # If no divisor is found, return False
    return False
