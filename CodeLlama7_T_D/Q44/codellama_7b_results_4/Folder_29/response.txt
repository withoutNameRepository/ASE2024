def composite_nums_between_indices(my_list):
    # Define the range of indices to check
    start_index = 52
    end_index = 71
    # Initialize an empty set to store the composite numbers
    composite_nums = set()

    # Iterate over the list and check if each element is a composite number between the specified range
    for i in range(start_index, end_index + 1):
        if my_list[i] > 1:
            for j in range(2, int(my_list[i]) ** 0.5 + 1):
                if my_list[i] % j == 0:
                    composite_nums.add(my_list[i])
                    break

    # Return the set of all composite numbers between the specified range
    return composite_nums
