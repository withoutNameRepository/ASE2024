    for i in range(30, 48+1):

# Check if a number is composite using the Euclidean algorithm
def is_composite(n):
    if n == 1 or n == 0:
        return False

    for i in range(2, int(math.sqrt(n)))+1):
        if n % i == 0:
            return True

    return False
