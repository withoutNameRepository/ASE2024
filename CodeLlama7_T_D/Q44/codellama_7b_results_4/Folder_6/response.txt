def composite_nums_between_indices(my_list):
    # Initialize an empty set to store the composite numbers
    composite_nums = set()

    # Iterate through the list and check if each element is a prime number
    for i in range(13, 68 + 1):
        is_prime = True

        # Check if the current element is not equal to 1, as 1 is neither prime nor composite
        if my_list[i] != 1:
            # Iterate through the list up to the current element and check if any of the elements divide the current element without leaving a remainder
            for j in range(2, i):
                if my_list[i] % my_list[j] == 0:
                    is_prime = False
                    break

        # If the current element is not prime, add it to the set of composite numbers
        if not is_prime:
            composite_nums.add(my_list[i])

    return composite_nums
