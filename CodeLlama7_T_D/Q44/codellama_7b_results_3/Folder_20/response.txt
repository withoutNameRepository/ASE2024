def is_composite(n):
    # Check if 'n' is greater than 1
    if n > 1:
        # Iterate from 2 to the square root of 'n'
        for i in range(2, int(n ** 0.5) + 1):
            # If 'i' divides 'n', return False
            if n % i == 0:
                return False

    # If we reach this point, then 'n' is prime
    return True
