def composite_nums_between_indices(my_list):
    # Check if the input is a list of positive integers
    if not all(isinstance(x, int) and x > 0 for x in my_list)):
        raise ValueError("Input must be a list of positive integers")
    # Initialize an empty set to store the composite numbers
    composite_nums = set()
    # Iterate over the list of integers and check if each number is composite
    for i in range(len(my_list)):
        num = my_list[i]
        # Check if the current number is composite by dividing it by all numbers less than or equal to its square root
        for j in range(i+1):
            if num % my_list[j] == 0:
                composite_nums.add(num)
                break
    # Return the set of all composite numbers in the given list that are between index 0 to index 9, both inclusive
    return composite_nums
