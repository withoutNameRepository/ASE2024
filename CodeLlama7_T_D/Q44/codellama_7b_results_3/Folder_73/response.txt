def composite_nums_between_indices(my_list):
    # Check if the input is a list of positive integers
    if not all(isinstance(x, int) and x > 0 for x in my_list)):
        raise ValueError("Input must be a list of positive integers")

    composite_nums = set()
    # Iterate over the input list
    for i in range(28, 41):
        # Check if the current element is prime or not
        if my_list[i] == 1:
            # If it is prime, check if its next number is composite or not
            if i + 1 < len(my_list) and my_list[i + 1] != 1:
                # If the next number is composite, add it to the set of composite numbers
                composite_nums.add(my_list[i + 1])
