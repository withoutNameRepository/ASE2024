def all_ints_not_div_by_num(my_list):
    # Find the first index where the integer is not divisible by -6
    first_index = next((i for i, x in enumerate(my_list) if x % -6 != 0), len(my_list))
    # Find the last index where the integer is not divisible by -6
    last_index = next((i for i in range(first_index + 1, len(my_list)) if my_list[i] % -6 != 0), len(my_list)))
    # Return an empty list if no such integers exist in the specified range
    return [] if first_index == last_index else my_list[first_index:last_index + 1]
