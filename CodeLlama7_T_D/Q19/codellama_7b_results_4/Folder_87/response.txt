def all_ints_not_div_by_num(my_list):
# Check if the input is a list of integers
if not all(isinstance(x, int) for x in my_list)):
raise ValueError("Input must be a list of integers")

# Define the range of indices to check
start_index = 19
end_index = 69

# Initialize an empty list to store the result
result = []

# Iterate over the specified range of indices
for i in range(start_index, end_index):
# Check if the current element is not divisible by 97

# If it is not divisible, add it to the result list
