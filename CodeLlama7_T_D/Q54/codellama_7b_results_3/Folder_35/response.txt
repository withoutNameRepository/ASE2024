def all_right_truncatable_prime(tuple):

    for i in range(2, x):
        is_prime = True
        for j in range(2, int(i ** 0.5) + 1):

                is_prime = False
                break

            prime_numbers.append(i)

    right_truncatable_prime_numbers = []

    for i in prime_numbers:

        while temp > 0:
