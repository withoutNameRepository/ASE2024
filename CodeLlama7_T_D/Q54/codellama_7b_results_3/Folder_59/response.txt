# Function to check if a number is a prime number
def is_prime(n):
	# Check if n is greater than 1
	if n > 1:
		# Iterate from 2 to n-1
		for i in range(2, n):
			# Check if i divides n
			if n % i == 0:
				# If i divides n, return False
				return False

	# If n is a prime number, return True
	return True
