def is_prime(n):
    # Check if n is less than or equal to 1
    if n <= 1:
        # If it is, return False
        return False

    # Check if n is divisible by 2 to n-1
    for i in range(2, n):
        # If it is divisible, return False
        if n % i == 0:
            return False

    # If none of the above conditions are true, return True
    return True
