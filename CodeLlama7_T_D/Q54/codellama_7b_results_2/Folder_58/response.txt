# Check if an integer is a prime number
def is_prime(n):

    # If the integer is not greater than 1, it is not prime
    if n <= 1:
        return False

    # Check if the integer is divisible by any integer between 2 and the square root of the integer
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    # If the integer is not divisible by any integer between 2 and the square root of the integer, it is prime
    return True
