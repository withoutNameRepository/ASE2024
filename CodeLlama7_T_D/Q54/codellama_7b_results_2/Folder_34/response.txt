def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True


def is_right_truncatable_prime(n):
    string_of_n = str(n)
    last_digit = int(string_of_n[-1])

    for i in range(2, last_digit + 1):
        if n % i == 0:
            return False

    return True
