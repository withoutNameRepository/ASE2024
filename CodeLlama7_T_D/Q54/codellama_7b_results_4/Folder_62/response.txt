def is_prime_number(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True


def is_right_truncatable_prime_number(n):
    length_of_n = len(str(n))
    for i in range(length_of_n - 1, -1, -1):
        if n % 10 ** i != 0:
            return False

    return True
