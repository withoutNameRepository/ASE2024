def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True


def is_right_truncatable_prime(n):
    for i in range(2, len(str(n)))]:
        if str(n)[-i] != str(n // 10 ** i))[-i]:
            return False

    return True
