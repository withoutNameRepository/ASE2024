def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True


def is_right_truncatable_prime(n):
    if n < 2:
        return False

    for i in range(len(str(n))-1, -1, -1):
        temp = n // 10**i

        if temp == 0 or is_right_truncatable_prime(temp):
            continue

        else:

            return False

    return True
