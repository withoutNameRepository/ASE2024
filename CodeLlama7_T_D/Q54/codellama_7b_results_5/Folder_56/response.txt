def is_prime(n):
if n < 2:
return False

for i in range(2, int(n ** 0.5) + 1):
if n % i == 0:
return False

return True

def is_right_truncatable_prime(n):
while n > 0:
last_digit = n % 10
n //= 10
if last_digit != 1 and last_digit != 3 and last_digit != 7 and last_digit != 9:
return False

return True
