def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True

def is_right_truncatable_prime(n):
    string_representation_of_n = str(n)
    for i in range(len(string_representation_of_n), 1, -1):
        if not is_prime(int(string_representation_of_n[:i])):
