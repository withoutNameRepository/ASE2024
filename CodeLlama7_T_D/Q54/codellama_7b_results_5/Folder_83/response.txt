def is_prime(n):
    if n < 2:
        return False

    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    return True

def is_right_truncatable_prime(n):
    if n < 10:
        return False

    while n > 0:
        last_digit = n % 10
        n //= 10
        if last_digit == 0:
            return False

    return True
