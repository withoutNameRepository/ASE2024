def sum_in_range(my_list):
    # Initialize a variable to store the sum
    sum = 0

    # Iterate through the given list of integers
    for i in my_list:
        # Check if the current integer is between 3 and 9, both inclusive
        if 3 <= i <= 9:
            # If it is, add it to the sum
            sum += i

    # Return the sum of all integers from the given list that are between 3 and 9, both inclusive. If there are no such integers, return 0.
    return sum
