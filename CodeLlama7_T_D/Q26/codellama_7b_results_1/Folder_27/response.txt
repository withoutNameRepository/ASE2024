def sum_in_range(my_list):
    # Initialize sum to 0
    sum = 0
    # Loop through the list of integers
    for i in my_list:
        # Check if the current integer is between 68 and 99, both inclusive
        if (i >= 68) and (i <= 99):
            # If it is, add it to the sum
            sum += i

    # Return the sum
