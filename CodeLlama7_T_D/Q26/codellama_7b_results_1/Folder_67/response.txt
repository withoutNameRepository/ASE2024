def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate over the list of integers
    for num in my_list:
        # Check if the number is between 31 and 50, both inclusive
        if 31 <= num <= 50:
            # If it is, add it to the total
            total += num

    # Return the total sum
    return total
