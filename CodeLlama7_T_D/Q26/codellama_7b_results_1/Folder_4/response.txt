def sum_in_range(my_list):
    # Initialize the sum to 0
    sum = 0
    # Loop through the list of integers
    for num in my_list:
        # Check if the current integer is between 32 and 44, both inclusive
        if 32 <= num <= 44:
            # If it is, add it to the sum
            sum += num

    # Return the sum
    return sum
