def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate over the list of integers
    for num in my_list:

        # Check if the current integer is between 113 and 392, inclusive
        if 113 <= num <= 392:

            # If it is, add it to the sum
            total += num

    # Return the sum
    return total
