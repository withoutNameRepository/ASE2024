def sum_in_range(my_list):
    # Initialize a variable to store the sum
    sum = 0

    # Iterate over the list of integers
    for num in my_list:
        # Check if the current number is between 36 and 52, both inclusive
        if (num >= 36) and (num <= 52):
            # If it is, add it to the sum
            sum += num

    # Return the sum
    return sum
