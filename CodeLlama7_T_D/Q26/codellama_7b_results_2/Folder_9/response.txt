def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate through the list of integers
    for num in my_list:
        # Check if the current integer is between 4 and 5, both inclusive
        if 4 <= num <= 5:
            # Add the current integer to the sum
            total += num

    # Return the sum of all integers from the given list that are between 4 and 5, both inclusive
    return total
