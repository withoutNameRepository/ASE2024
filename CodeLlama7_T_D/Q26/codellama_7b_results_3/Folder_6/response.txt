def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Loop through the list of integers
    for num in my_list:

        # Check if the current integer is between 335 and 611, inclusive
        if 335 <= num <= 611:

            # If it is, add it to the sum
            total += num

    # Return the sum
    return total
