def sum_in_range(my_list):
    # Initialize the sum to 0
    total = 0

    # Iterate through the list of integers
    for num in my_list:
        # Check if the current number is between 139 and 767
        if 139 <= num <= 767:
            # If it is, add it to the sum
            total += num

    # Return the sum
    return total
