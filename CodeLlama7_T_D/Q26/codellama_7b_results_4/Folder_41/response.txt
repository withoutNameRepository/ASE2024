def sum_in_range(my_list):
    # Initialize the sum to 0
    sum = 0
    # Iterate through the list of integers
    for i in my_list:
        # Check if the current integer is between 2 and 4, both inclusive
        if 2 <= i <= 4:
            # Add the current integer to the sum
            sum += i

    return sum
