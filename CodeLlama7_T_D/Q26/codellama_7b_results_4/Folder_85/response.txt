def sum_in_range(my_list):
    # Initialize the sum variable
    total = 0

    # Loop through the list and check if each element is between 6 and 7
    for i in my_list:
        if (i >= 6) and (i <= 7):
            # If the element is between 6 and 7, add it to the total sum
            total += i

    # Return the total sum of all elements between 6 and 7, or 0 if no such elements were found
    return total
