def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Loop through the list of integers
    for i in my_list:
        # Check if the current integer is between 514 and 816, inclusive
        if 514 <= i <= 816:
            # If it is, add it to the total sum
            total += i

    # Return the total sum of integers between 514 and 816, inclusive
