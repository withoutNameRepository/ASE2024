def sum_in_range(my_list):
    # Initialize the sum to 0
    sum = 0
    # Iterate over the list of integers
    for num in my_list:
        # Check if the current number is between -9 and -7, both inclusive
        if -9 <= num <= -7:
            # If it is, add it to the sum
            sum += num

    # Return the sum
    return sum
