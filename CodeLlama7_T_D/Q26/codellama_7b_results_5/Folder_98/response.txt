def sum_in_range(my_list):
    # Initialize the sum to 0
    total = 0
    # Iterate through the list of integers
    for num in my_list:
        # Check if the number is between 3 and 8 (inclusive)
        if 3 <= num <= 8:
            # If it is, add it to the sum
            total += num

    # Return the sum
    return total
