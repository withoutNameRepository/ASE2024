def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Loop through the list of integers
    for num in my_list:
        # Check if the current number is between 7 and 8, both inclusive
        if 7 <= num <= 8:
            # If it is, add it to the sum
            total += num

    # Return the sum of all integers from the given list that are between 7 and 8, both inclusive. If there are no such integers, return 0.
    return total
