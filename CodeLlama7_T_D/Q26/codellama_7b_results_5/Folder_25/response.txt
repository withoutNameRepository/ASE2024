def sum_in_range(my_list):
    # Initialize sum to 0
    sum = 0
    # Loop through the list of integers
    for i in my_list:
        # Check if the integer is between -10 and 48, both inclusive
        if -10 <= i <= 48:
            # If it is, add it to the sum
            sum += i

    # Return the sum
    return sum
