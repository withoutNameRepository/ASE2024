def sum_in_range(my_list):
    # Initialize sum to 0
    sum = 0
    # Iterate over the list of integers
    for num in my_list:
        # Check if the number is between 3 and 7, both inclusive
        if 3 <= num <= 7:
            # If it is, add it to the sum
            sum += num

    # Return the sum
    return sum
