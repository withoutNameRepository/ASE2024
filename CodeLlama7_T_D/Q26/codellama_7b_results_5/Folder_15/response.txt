def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0
    # Loop through each element in the list
    for i in my_list:
        # Check if the element is between 6 and 9, both inclusive
        if 6 <= i <= 9:
            # If it is, add it to the sum
            total += i

    # Return the sum
    return total
