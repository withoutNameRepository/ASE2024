def sum_in_range(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate over the list of integers
    for num in my_list:
        # Check if the number is between 335 and 611, both inclusive
        if 335 <= num <= 611:
            # If it is, add it to the total
            total += num

    # Return the sum of all integers in the given list that are between 335 and 611, both inclusive
    return total
