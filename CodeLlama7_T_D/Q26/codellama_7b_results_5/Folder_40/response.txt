def sum_in_range(my_list):
    # Initialize the sum to 0
    sum = 0
    # Loop through the list of integers
    for i in my_list:
        # Check if the integer is between -9 and -7, both inclusive
        if -9 <= i <= -7:

            sum += i

    # Return the sum
    return sum
