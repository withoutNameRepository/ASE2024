def palindromes_of_specific_lengths(string):
    # Get the substring of interest
    substring = string[15:94]
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the substring
    for i in range(len(substring)):
        # Get the character and its opposite
        char = substring[i]
        opp_char = substring[len(substring) - 1 - i]
        # Check if the character is a letter
        if char.isalpha():
            # Check if the character is the same as its opposite
            if char == opp_char:
                # Add the palindrome to the set
                palindromes.add(substring[i:len(substring) - i + 1])
