def palindromes_of_specific_lengths(string):
    # Define the substring range
    start = 15
    end = 86
    # Create an empty set to store the palindromes
    palindromes = set()
    # Iterate over the characters in the substring
    for i in range(start, end + 1):
        # Check if the character at index i is a letter
        if string[i].isalpha():
            # Create a copy of the substring starting from index i and ending at the end of the string
            substr = string[i:]
            # Iterate over the characters in the copy of the substring
            for j in range(len(substr)):
                # Check if the character at index j is a letter
                if substr[j].isalpha():
                    # Create a palindrome by reversing the characters in the substring starting from index i and ending at index j
                    palindrome = substr[i:j+1][::-1]
                    # Check if the palindrome has the desired length
                    if len(palindrome) >= 51 and len(palindrome) <= 60:
                        # Add the palindrome to the set of palindromes
                        palindromes.add(palindrome)
