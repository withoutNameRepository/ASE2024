def palindromes_of_specific_lengths(string):
# Case-insensitive search for palindromes in the substring from index 31 to index 74, both inclusive
palindromes = []
for i in range(31, 75):
    for j in range(i + 1, 75):
        substr = string[i:j]
        # Check if the substring is a palindrome
        if substr == substr[::-1]:
            palindromes.append(substr)
# Filter out palindromes that are not of lengths between 23 and 39 (both inclusive)
palindromes = [p for p in palindromes if len(p) in range(23, 40)]
return set(palindromes)
