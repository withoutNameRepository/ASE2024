def palindromes_of_specific_lengths(string):
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Loop through each character in the string
    for i in range(11, 84):
        # Check if the substring starting at index i and ending at index i + 37 (or 60) forms a palindrome
        if string[i:i+37] == string[i:i+37][::-1]:
            # If it does, add it to the set of palindromes
            palindromes.add(string[i:i+37])
    return palindromes
