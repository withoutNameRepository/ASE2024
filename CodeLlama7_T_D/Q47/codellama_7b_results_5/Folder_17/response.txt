def palindromes_of_specific_lengths(string):

index_range = range(22, 95 + 1)
length_range = range(52, 55 + 1)
# Initialize an empty set to store the palindromes
palindromes = set()
# Iterate over each index in the index range

# Get the substring starting at the current index
substring = string[i:]
# Check if the substring is a palindrome of length 52 to 55

# If it is, add it to the set of palindromes
