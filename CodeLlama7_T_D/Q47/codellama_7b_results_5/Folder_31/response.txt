def palindromes_of_specific_lengths(string):
    # Define the substring range for finding palindromes
    start_index = 11
    end_index = 83
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over each character in the string within the specified range
    for i in range(start_index, end_index + 1):
        # Check if the current character is a palindrome with the required length
        if string[i - 1:i] == string[i - 1:i][::-1]:
            # If it is a palindrome, add it to the set of palindromes
            palindromes.add(string[i - 1:i])
    return palindromes
