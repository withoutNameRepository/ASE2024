def palindromes_of_specific_lengths(string):
    # Define the range of lengths to consider
    length_range = range(3, 7)
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over each substring in the specified index range
    for i in range(0, len(string), 7):
        # Get the current substring
        substring = string[i:i+7]
        # Iterate over each length in the defined range
        for length in length_range:
            # Get the palindrome of the specified length from the current substring
            palindrome = substring[:length][::-1] + substring[length:]
            # Check if the palindrome occurs in the substring
            if palindrome in substring:
                # Add the palindrome to the set of palindromes
                palindromes.add(palindrome)
    # Return the set of palindromes found in the specified index range
    return palindromes
