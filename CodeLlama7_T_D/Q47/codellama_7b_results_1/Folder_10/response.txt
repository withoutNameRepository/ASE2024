def palindromes_of_specific_lengths(string):
    # Define the range of lengths to consider
    lengths = range(16, 40)
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over each length in the specified range
    for length in lengths:
        # Define the starting and ending indices for the substring
        start_index = 16
        end_index = 60
        # Iterate over each character in the substring
        for i in range(start_index, end_index):
            # Check if the current character is a palindrome of the specified length
            if string[i - (length // 2)] == string[i + (length // 2)]:
                # Add the palindrome to the set
                palindromes.add(string[i - (length // 2)] + string[i] + string[i + (length // 2)])
    # Return the set of palindromes
    return palindromes
