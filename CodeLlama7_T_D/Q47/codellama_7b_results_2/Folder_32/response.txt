def palindromes_of_specific_lengths(string):
    # Define the substring range for which we want to find palindromes
    substr = string[16:77]
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over each character in the substring
    for i in range(len(substr)):
        # Get the substring starting from the current index and ending at the end of the substring
        substr_i = substr[i:]
        # Check if the substring is a palindrome
        if substr_i == substr_i[::-1]:
            # If it is a palindrome, add it to the set of palindromes
            palindromes.add(substr_i)
    # Return the set of palindromes that meet the specified requirements
    return palindromes
