def palindromes_of_specific_lengths(string):
    # Define a lambda function to check if a string is a palindrome or not
    is_palindrome = lambda s: s == s[::-1]
    # Initialize an empty set to store the palindromes
    palindromes = set()
    # Iterate over the characters in the specified index range
    for i in range(22, 95+1):
        # Check if the character at index i is a palindrome or not
        if is_palindrome(string[i]):
            # If the character is a palindrome, add it to the set of palindromes
            palindromes.add(string[i])
    # Return the set of palindromes that satisfy the given requirements
    return palindromes
