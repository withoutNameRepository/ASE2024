import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []

    # Loop over each row and column in the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Extract a submatrix from the original matrix
            submatrix = matrix[i:, j:]

            # Check if the sum of all integers in the submatrix is equal to 94
            if np.sum(submatrix) == 94:
                # If the condition is met, add the submatrix to the list of submatrices
                submatrices.append(submatrix)

    # Return the list of all submatrices that meet the condition
    return submatrices
