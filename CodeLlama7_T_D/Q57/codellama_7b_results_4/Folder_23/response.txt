import numpy as np
def submatrix_with_particular_sum(matrix):

    submatrices = []
    # Loop over each row and column of the matrix
    for i in range(matrix.shape[0]):

            # Calculate the sum of the elements in the current submatrix
            sum = 0
            for k in range(i, i + 3):
                for l in range(j, j + 3):
                    sum += matrix[k][l]
            if sum == 59:
                # Add the current submatrix to the list of submatrices with sum equal to 59
                submatrices.append([i, i + 2], [j, j + 2]))
