import numpy as np
def submatrix_with_particular_sum(matrix):
    # Find all rows and columns with sum equal to 80
    rows, cols = matrix.shape
    row_indices, col_indices = [], []
    for i in range(rows):
        if np.sum(matrix[i]) == 80:
            row_indices.append(i)
    for j in range(cols):
        if np.sum(matrix[:, j]) == 80:
            col_indices.append(j)
    # Find all submatrices with sum equal to 80
    submatrices = []
    for i in row_indices:
        for j in col_indices:
            submatrix = matrix[i:i+1, j:j+1]
            if np.sum(submatrix) == 80:
                submatrices.append(submatrix)
    # Return the list of all submatrices with sum equal to 80
    return submatrices
