import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []

    # Loop over each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Calculate the sum of all integers in the submatrix
            submatrix_sum = np.sum(matrix[i:, j:])

            # If the sum is equal to -157, add the submatrix to the list of submatrices
            if submatrix_sum == -157:
                submatrices.append(matrix[i:, j:])

    # Return the list of all submatrices with a sum equal to -157
    return submatrices
