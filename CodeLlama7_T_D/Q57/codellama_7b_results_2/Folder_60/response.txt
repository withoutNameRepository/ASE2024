import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices
    submatrices = []

    # Loop through each row of the matrix
    for i in range(len(matrix)):
        # Loop through each column of the matrix
        for j in range(len(matrix[i])):
            # Initialize a variable to store the sum of the current submatrix
            submatrix_sum = 0

            # Loop through each row and column of the current submatrix
            for k in range(len(matrix[i][j])):
                submatrix_sum += matrix[i][j][k]

            # If the sum of the current submatrix is equal to 94, add it to the list of submatrices
            if submatrix_sum == 94:
                submatrices.append(matrix[i][j])

    # Return the list of all submatrices that have a sum of 94
    return submatrices
