import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []

    # Iterate over each row and column in the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Extract the submatrix that starts at (i,j)
            submatrix = matrix[i:i+2, j:j+2]

            # Check if the sum of all integers in the submatrix equals -25
            if np.sum(submatrix) == -25:
                # If yes, add the submatrix to the list of submatrices
                submatrices.append(submatrix)

    # Return the list of submatrices that satisfy the condition
    return submatrices
