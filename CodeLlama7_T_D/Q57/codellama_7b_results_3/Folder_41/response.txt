import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices
    submatrices = []

    # Iterate over each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Calculate the sum of the elements in the submatrix
            submatrix_sum = np.sum(matrix[i:,j]) + np.sum(matrix[i,j+1:])

            # Check if the sum equals -26
            if submatrix_sum == -26:
                # Add the submatrix to the list of submatrices
                submatrices.append(matrix[i:,j] + matrix[i,j+1:])

    # Return the list of all submatrices that have a sum of -26
    return submatrices
