import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize an empty list to store the submatrices
    submatrices = []

    # Loop through each row of the matrix
    for i in range(matrix.shape[0]):

        # Loop through each column of the matrix
        for j in range(matrix.shape[1]):

            # Initialize a variable to store the sum of the submatrix
            submatrix_sum = 0

            # Loop through each element of the submatrix
            for row in range(i, i + 2):
                for col in range(j, j + 2):
                    submatrix_sum += matrix[row][col]

                    # If the sum of the elements in the submatrix equals -43, add the submatrix to the list of submatrices
                    if submatrix_sum == -43:
                        submatrices.append(matrix[i:i + 2, j:j + 2])

            # If no submatrix was found with a sum of -43, return an empty list
            if not submatrices:
                return []

    # Return the list of all submatrices that have a sum of -43
    return submatrices
