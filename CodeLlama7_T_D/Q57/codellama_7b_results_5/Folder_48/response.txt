import numpy as np

def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []

    # Iterate over each row and column in the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Create a copy of the matrix with only one element selected
            submatrix = matrix[i:i+1, j:j+1].copy()

            # Check if the sum of all integers in the submatrix is equal to -261
            if np.sum(submatrix) == -261:
                # If it is, add it to the list of submatrices
                submatrices.append(submatrix)

    # Return the list of submatrices that meet the condition
    return submatrices
