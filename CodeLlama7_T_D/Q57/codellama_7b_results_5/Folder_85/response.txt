import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices with sum -87
    submatrices = []
    # Loop over all possible starting indices of a submatrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Calculate the sum of all elements in the submatrix
            sum_submatrix = 0
            for k in range(i, i + 2):
                for l in range(j, j + 2):
                    sum_submatrix += matrix[k][l]
            # If the sum is equal to -87, add the submatrix to the list of submatrices with sum -87
            if sum_submatrix == -87:
                submatrices.append(matrix[i:i + 2, j:j + 2])
    # Return the list of all submatrices of the given matrix such that the sum of all integers in each submatrix equals -87
    return submatrices
