import numpy as np
def submatrix_with_particular_sum(matrix):

    submatrices = []
    # Loop over each row of the matrix
    for i in range(matrix.shape[0]):

            submatrix_sum = np.sum(matrix[i:, j]) + np.sum(matrix[i, j+1:]) - matrix[i, j]

            if submatrix_sum == -37:
