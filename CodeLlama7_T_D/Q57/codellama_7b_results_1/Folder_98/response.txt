import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []
    # Loop over each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Create a copy of the original matrix
            submatrix = np.copy(matrix)
            # Remove the current element from the submatrix
            submatrix[i,j] = 0
            # Check if the sum of all elements in the submatrix is 0
            if np.sum(submatrix) == 0:
                # Add the submatrix to the list of submatrices
                submatrices.append(submatrix)
    # Return the list of submatrices that meet the condition
    return submatrices
