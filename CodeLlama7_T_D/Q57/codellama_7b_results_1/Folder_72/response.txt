import numpy as np
def submatrix_with_particular_sum(matrix):
    # Initialize the list of submatrices
    submatrices = []

    # Loop through each row and column of the matrix
    for i in range(matrix.shape[0]):
        for j in range(matrix.shape[1]):
            # Create a copy of the original matrix
            submatrix = np.copy(matrix)

            # Remove all elements from the submatrix that are not in the specified row and column
            submatrix = submatrix[i:i+1, j:j+1]

            # Check if the sum of all elements in the submatrix is equal to 41
            if np.sum(submatrix) == 41:
                # Add the submatrix to the list of submatrices
                submatrices.append(submatrix)

    # Return the list of submatrices that have a sum of all elements equal to 41, or an empty list if no such submatrix is found
    return submatrices
