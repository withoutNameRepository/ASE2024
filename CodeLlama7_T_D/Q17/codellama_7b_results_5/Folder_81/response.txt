def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the results
    result = []

    # Iterate through the list of integers starting from index 10 to index 100, both inclusive
    for i in range(10, len(my_list)):

        # Check if the current integer is divisible by both 10 and 100
        if my_list[i] % 10 == 0 and my_list[i] % 100 == 0:

            # If it is, add it to the result list
            result.append(my_list[i])

    # Return the result list
    return result
