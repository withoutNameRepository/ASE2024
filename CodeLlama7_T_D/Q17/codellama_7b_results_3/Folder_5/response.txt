def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the divisible integers
    divisible_ints = []

    # Loop through the specified range of the list
    for i in range(len(my_list))[:2]:

        # Check if the current integer is divisible by both -6 and 4
        if my_list[i] % (-6) == 0 and my_list[i] % 4 == 0:

            divisible_ints.append(my_list[i])

    # Return the list of all integers divisible by both -6 and 4 from index 0 to index 1, both inclusive
