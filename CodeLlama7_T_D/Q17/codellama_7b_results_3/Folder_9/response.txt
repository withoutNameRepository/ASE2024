def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the divisible integers
    divisible_ints = []

    # Iterate through the list and check if each integer is divisible by both 3 and -300
    for i in range(70, 201):
        if my_list[i] % 3 == 0 and my_list[i] % -300 == 0:
            # If the integer is divisible by both 3 and -300, add it to the list of divisible integers
            divisible_ints.append(my_list[i])

    # Return the list of all integers divisible by both 3 and -300 from index 70 to index 200, both inclusive
    return divisible_ints
