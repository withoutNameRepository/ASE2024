def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the divisible integers
    divisible_ints = []

    # Iterate through the list and check if each integer is divisible by both 31 and 11
    for i in range(61, 89):
        if my_list[i] % 31 == 0 and my_list[i] % 11 == 0:
            # If the integer is divisible by both numbers, add it to the list of divisible integers
            divisible_ints.append(my_list[i])

    # Return the list of all integers divisible by both 31 and 11 from index 61 to index 88, both inclusive
    return divisible_ints
