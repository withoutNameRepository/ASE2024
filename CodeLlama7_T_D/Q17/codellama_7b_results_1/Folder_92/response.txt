def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the result
    result = []

    # Iterate over the list and check if each element is divisible by both 1 and -1
    for i in range(len(my_list)):
        if my_list[i] % 1 == 0 and my_list[i] % -1 == 0:
            result.append(my_list[i])

    # Return the list of all integers divisible by both 1 and -1 from index 0 to index 0, both inclusive
    return result
