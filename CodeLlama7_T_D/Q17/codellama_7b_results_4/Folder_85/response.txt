def all_ints_div_by_both_two_nums(my_list):
    # Initialize an empty list to store the result
    result = []

    # Iterate through the list and check if each integer is divisible by both -9 and -3
    for i in range(1, 9):
        if my_list[i] % (-9) == 0 and my_list[i] % (-3) == 0:
            # If the integer is divisible by both numbers, append it to the result list
