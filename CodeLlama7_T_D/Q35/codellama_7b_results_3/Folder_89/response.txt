def remove_repeat_chars(s):
    # Create a dictionary to keep track of character frequencies
    freq = {}

    # Iterate over the string and increment the frequency count for each character
    for c in s:
        if c not in freq:
            freq[c] = 0

        freq[c] += 1

    # Remove all characters that occur more than once between indices 55 and 84, both exclusive
    for i in range(55, 84):
        c = s[i]

        if freq[c] > 1:
            s = s[:i] + s[i+1:]

            break


    return s
