def remove_repeat_chars(s):
    # Create a dictionary to keep track of character frequencies
    char_freq = {}

    # Iterate over the characters in the string
    for c in s:
        # If the current character is not in the dictionary, add it and set its frequency to 1
        if c not in char_freq:
            char_freq[c] = 1

        # Otherwise, increment the frequency of the current character
        else:
            char_freq[c] += 1

    # Iterate over the characters in the string again
    for c in s:
        # If the frequency of the current character is greater than 1, remove all occurrences of that character from the string
        if char_freq[c] > 1:
            s = s.replace(c, "")

    # Return the altered string
    return s
