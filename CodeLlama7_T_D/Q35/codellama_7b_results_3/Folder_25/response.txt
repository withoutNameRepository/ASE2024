def remove_repeat_chars(s):
    # Create a new string that is the same length as the original string
    new_string = ""

    # Iterate through each character in the original string
    for char in s:
        # If the current character has not been seen before, add it to the new string
        if new_string.find(char) == -1:
            new_string += char

    return new_string
