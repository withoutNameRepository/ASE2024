def remove_repeat_chars(s):
    # Create a dictionary to store the frequency of each character
    char_freq = {}
    for i in range(len(s)):
        # Update the frequency of each character
        if s[i] in char_freq:
            char_freq[s[i]] += 1
        else:
            char_freq[s[i]] = 1
    # Remove characters that occur more than once between indices 48 and 76
    for i in range(len(s)):
        if s[i] in char_freq and char_freq[s[i]] > 1:
            # Find the first occurrence of the character
            first_occurrence = i

            # Update the frequency of the character
            char_freq[s[i]] -= 1

            # Remove all occurrences of the character from the string
            while s[first_occurrence] == s[i]:
                s = s[:first_occurrence] + s[first_occurrence+1:]
                first_occurrence += 1

    return s
