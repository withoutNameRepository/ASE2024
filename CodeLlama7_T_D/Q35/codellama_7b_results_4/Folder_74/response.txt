def remove_repeat_chars(s):
    # create a new string with the characters from the original string
    # that are not repeated between indices 69 and 97
    new_str = ""
    for i in range(len(s)):
        if (i < 69 or i > 97) and s[i] not in new_str:
            new_str += s[i]
    return new_str
