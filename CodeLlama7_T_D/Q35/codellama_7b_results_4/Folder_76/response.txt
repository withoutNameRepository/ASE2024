def remove_repeat_chars(s):
# Split the string into a list of characters
char_list = list(s)
# Create an empty list to store the filtered characters
filtered_char_list = []
# Iterate over each character in the list
for char in char_list:
# If the current character is not already in the filtered list, add it
if char not in filtered_char_list:
filtered_char_list.append(char)
# Join the filtered list of characters back into a string
altered_string = "".join(filtered_char_list)
return altered_string
