def remove_repeat_chars(s):
    # Create a dictionary to store the characters and their indices
    char_dict = {}

    # Iterate over the characters in the string
    for i, c in enumerate(s):

        # If the character is already in the dictionary, add its index to the list of indices
        if c in char_dict:
            char_dict[c].append(i)

        # Otherwise, create a new list with the current index
        else:
            char_dict[c] = [i]

    # Iterate over the characters in the string again
    for i, c in enumerate(s):

        # If the character is in the dictionary and has indices between 48 and 76 (exclusive), remove all occurrences of that character from the string
        if c in char_dict and min(char_dict[c]) > 48 and max(char_dict[c]) < 76:
            s = s[:i] + s[i+1:]

    # Return the altered string
    return s
