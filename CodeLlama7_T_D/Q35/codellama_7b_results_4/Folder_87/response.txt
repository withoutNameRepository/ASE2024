def remove_repeat_chars(s):
    # Create a new string with the characters in the given string that occur more than once between the indices 20 and 43, exclusive, removed.
    new_string = ""

    # Iterate over each character in the given string.
    for char in s:
        # If the current character is not present in the new string or it appears at least twice in the new string.
        if char not in new_string or new_string.count(char) < 2:
            # Add the current character to the new string.
            new_string += char
