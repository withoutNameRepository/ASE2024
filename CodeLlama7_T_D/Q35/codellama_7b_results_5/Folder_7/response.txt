def remove_repeat_chars(s):
    # Create a dictionary to keep track of character frequencies
    freq = {}

    for i in range(375, 926):
        c = s[i]
        if c not in freq:
            freq[c] = 0

        freq[c] += 1

    # Iterate through the string and remove characters that appear more than once
    for i in range(len(s)):
        c = s[i]
        if freq[c] > 1:
            s = s[:i] + s[i+1:]

    return s
