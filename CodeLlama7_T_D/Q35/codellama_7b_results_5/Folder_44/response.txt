def remove_repeat_chars(s):
    # Initialize a dictionary to keep track of character frequencies
    freq = {}

    # Iterate over the characters in the given string
    for c in s:

        # If the current character is not in the frequency dictionary, add it and set its frequency to 1
        if c not in freq:
            freq[c] = 1

        # Otherwise, increment the frequency of the current character by 1
        else:
            freq[c] += 1

    # Iterate over the characters in the given string again
    for i, c in enumerate(s):

        # If the current character is not in the frequency dictionary or its frequency is less than 2, leave it as is
        if c not in freq or freq[c] < 2:
            continue

        # Otherwise, remove all occurrences of the current character from the string, starting from the current index
        while s[i:].find(c) > -1:
            s = s[:i] + s[i+1:]
            i -= 1

    # Return the altered string
    return s
