def remove_repeat_chars(s):
# Split the string into a list of characters
char_list = list(s)
# Create a dictionary to keep track of each character's frequency
freq_dict = {}
# Iterate through the list of characters and update the frequency dictionary
for char in char_list:
if char not in freq_dict:
freq_dict[char] = 1
else:
freq_dict[char] += 1
# Remove any character that appears more than once between indices 32 and 44, exclusive
for i in range(32, 44):
if freq_dict[char_list[i]] > 1:
char_list = char_list[:i] + char_list[i+1:]
# Join the list of characters back into a string and return it
return "".join(char_list)
