def remove_repeat_chars(s):
    # create a new string with the characters in order
    new_str = ""

    # loop through each character in the original string
    for char in s:

        # if the current character is not already in the new string
        if char not in new_str:

            # add the current character to the new string
            new_str += char

        # if the current character is already in the new string
        else:

            # find the first occurrence of the current character in the new string
            index = new_str.index(char)

            # remove all occurrences of the current character from the new string, starting from the first occurrence
            new_str = new_str[:index] + new_str[index+1:]

    return new_str
