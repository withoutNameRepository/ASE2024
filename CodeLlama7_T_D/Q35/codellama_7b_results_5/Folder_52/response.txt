def remove_repeat_chars(s):
# Split the string into a list of characters
char_list = list(s)
# Create a dictionary to keep track of each character's frequency
freq_dict = {}
# Iterate over the characters in the string
for char in char_list:
# If the character is not already in the dictionary, add it with frequency 1
if char not in freq_dict:
freq_dict[char] = 1
# Otherwise, increment the frequency of the character
else:
freq_dict[char] += 1
# Iterate over the characters in the string again
for char in char_list:
# If the frequency of the character is greater than 1, remove all occurrences of that character from the string
if freq_dict[char] > 1:
s = s.replace(char, "")
return s
