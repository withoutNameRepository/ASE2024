def remove_repeat_chars(s):
    # create a dictionary to keep track of each character's frequency
    freq = {}
    for i in range(3, len(s)):
        if s[i] not in freq:
            freq[s[i]] = 1
        else:
            freq[s[i]] += 1
    # loop through the string and remove any character that appears more than once
    for i in range(3, len(s)):
        if freq[s[i]] > 1:
            s = s[:i] + s[i+1:]
    return s
