def remove_repeat_chars(s):
    # Create a set of unique characters in the string
    unique_chars = set(s)
    # Initialize an empty string for the result
    result = ""

    # Iterate over each character in the string
    for char in s:
        # If the character is not in the set of unique characters, add it to the result string
        if char not in unique_chars:
            result += char

        # If the character is already in the set of unique characters, do not add it to the result string
        else:
            continue

    # Return the altered string
    return result
