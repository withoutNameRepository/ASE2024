def remove_repeat_chars(s):
    # Create a dictionary to keep track of characters and their counts
    char_counts = {}

    # Iterate over the string and count the occurrences of each character
    for c in s:
        if c not in char_counts:
            char_counts[c] = 1

        else:

            char_counts[c] += 1


    # Iterate over the characters in the string and remove any duplicates that are outside of the specified indices
    for i, c in enumerate(s):

        if char_counts[c] > 1 and i < 56 or i > 90:
            s = s.replace(c, '', 1)

    return s
