def remove_repeat_chars(s):
    # Create a dictionary to keep track of characters and their indices
    char_indices = {}
    # Iterate through the string and update the dictionary with character and its index
    for i, c in enumerate(s):
        if c not in char_indices:
            char_indices[c] = []
        char_indices[c].append(i)
    # Remove all occurrences of characters that occur more than once between the indices 44 and 67, both exclusive
    for c in char_indices:
        if len(char_indices[c]) > 1:
            indices = sorted(char_indices[c])
            # Remove all occurrences of the character starting from the end index
            for i in range(len(indices) - 1, -1, -1):
                s = s[:indices[i]] + s[indices[i] + 1:]
    return s
