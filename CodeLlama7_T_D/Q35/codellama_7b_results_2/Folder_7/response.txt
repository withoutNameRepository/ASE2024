def remove_repeat_chars(string):
    # Create a list of all characters in the given string
    char_list = list(string)

    # Iterate over each character in the string and check if it occurs more than once between indices 375 and 926
    for i, char in enumerate(char_list):
        if char_list.count(char) > 1 and i >= 375 and i < 926:
            # If the character occurs more than once, remove all occurrences of it from the string
            while char_list.count(char) > 0:
                char_list.remove(char)

    # Join the list of characters back into a string and return it
    return "".join(char_list)
