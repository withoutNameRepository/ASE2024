def remove_repeat_chars(s):
    # Create a dictionary to keep track of characters
    char_dict = {}

    # Iterate through the string and update the dictionary
    for c in s:
        if c not in char_dict:
            char_dict[c] = 1

        else:

            char_dict[c] += 1



    # Create a new string to store the altered characters
    new_s = ""

    # Iterate through the dictionary and append characters that occur only once
    for c in s:
        if char_dict[c] == 1:
            new_s += c


    return new_s
