def remove_repeat_chars(string):
    # Create a dictionary to store the frequency of each character
    char_freq = {}
    for char in string:
        if char in char_freq:
            char_freq[char] += 1
        else:
            char_freq[char] = 1
    # Create a new string with the characters that occur only once
    new_string = ""
    for char, freq in char_freq.items():
        if freq == 1:
            new_string += char

    return new_string
