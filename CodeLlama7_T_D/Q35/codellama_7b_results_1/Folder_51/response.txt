def remove_repeat_chars(s):
# Split the string into a list of characters
char_list = list(s)
# Create a dictionary to store the characters and their counts
counts = {}

# Iterate over the list of characters, updating the dictionary with the character's count
for char in char_list:
if char not in counts:
counts[char] = 1
else:
counts[char] += 1

# Iterate over the keys of the dictionary, which are the unique characters in the string
for char in counts.keys():

# If the character's count is greater than 1, remove all occurrences of that character from the string
if counts[char] > 1:
s = s.replace(char, "")

# Return the altered string
return s
