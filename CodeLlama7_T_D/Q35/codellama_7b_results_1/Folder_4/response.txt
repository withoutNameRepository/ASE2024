def remove_repeat_chars(string):
    # Create a dictionary to keep track of the repeated characters
    repeated_chars = {}

    # Iterate over the string and check if each character is repeated
    for i in range(49, 74):
        char = string[i]
        if char in repeated_chars:
            # If the character is repeated, remove all occurrences from the string
            string = string.replace(char, "")

        else:
            # If the character is not repeated, add it to the dictionary
            repeated_chars[char] = True

    return string
