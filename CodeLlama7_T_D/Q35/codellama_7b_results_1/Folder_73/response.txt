def remove_repeat_chars(s):
    # create a set of unique characters in the string
    unique_chars = set(s)
    # initialize an empty string to store the altered string
    altered_string = ""

    # iterate over each character in the string
    for char in s:
        # if the character is not in the set of unique characters, append it to the altered string
        if char not in unique_chars:
            altered_string += char

        # if the character is in the set of unique characters and its index is between 10 and 76 (exclusive), remove all occurrences of that character from the string
        elif char in unique_chars and s.index(char) > 10 and s.index(char) < 76:
            altered_string += char

    # return the altered string
    return altered_string
