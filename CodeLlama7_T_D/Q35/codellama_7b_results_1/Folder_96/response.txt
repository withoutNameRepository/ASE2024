def remove_repeat_chars(s):
# Find the indices of the first and last occurrence of each character in the string
indices = {}
for i, c in enumerate(s):
if c not in indices:
indices[c] = (i, i)
else:
indices[c] = (min(indices[c][0], i), max(indices[c][1], i))
# Remove all occurrences of each character that occurs more than once between the indices 60 and 200, exclusive
for c in s:
if indices[c][0] > 59 and indices[c][1] < 199 and indices[c][1] - indices[c][0] > 1:
s = s[:indices[c][0]] + s[indices[c][1]+1:]
return s
