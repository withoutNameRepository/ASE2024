def remove_repeat_chars(string):
    # Create a dictionary to store the characters and their indices
    char_indices = {}

    # Iterate through the string and update the dictionary
    for i, char in enumerate(string):
        if char not in char_indices:
            char_indices[char] = []
        char_indices[char].append(i)

    # Remove the characters that occur more than once
    for char, indices in char_indices.items():
        for index in sorted(indices)[1:]:
            string = string[:index] + string[index+1:]

    return string
