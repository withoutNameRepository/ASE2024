def remove_repeat_chars(s):
    # Create a set of unique characters in the string
    unique_chars = set(s)
    # Initialize an empty string to store the altered string
    altered_string = ""

    for char in s:
        if char not in unique_chars:
            # If the character is repeated, remove all occurrences of that character from the string
            while s.count(char) > 1:
                s = s.replace(char, "")

            # Add the character to the set of unique characters
            unique_chars.add(char)

            # Add the character to the altered string
            altered_string += char

    return altered_string
