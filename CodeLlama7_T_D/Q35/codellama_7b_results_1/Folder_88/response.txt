def remove_repeat_chars(s):
    # Initialize an empty list to store the characters of the input string
    char_list = []

    # Iterate over each character in the input string
    for c in s:

        # If the current character is not already in the character list, add it
        if c not in char_list:
            char_list.append(c)

# Create a new empty string to store the modified string
modified_string = ""

# Iterate over each character in the character list
for c in char_list:

    # If the current character is not already in the modified string, add it and its all occurrences from the input string
    if s.count(c) > 1 and c not in modified_string:
        modified_string += c * (s.count(c) - 1)

# Return the modified string
return modified_string
