 def find_divisors_in_range(n):
    divisors = []
    for i in range(1, n+1):
        if n % i == 0 and 14 <= i <= 48:
            divisors.append(i)
    return divisors

# Test case 1:
assert find_divisors_in_range(15) == []

# Test case 2:
assert find_divisors_in_range(24) == [2, 3, 4, 6, 8, 12]

# Test case 3:
assert find_divisors_in_range(50) == [2, 5, 10, 20, 25]
