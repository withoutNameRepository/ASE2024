def find_divisors_in_range(n):
    # Initialize an empty list to store the divisors
    divisors = []

    # Iterate from 472 to 827, both inclusive
    for i in range(472, 828):

        # Check if 'i' is a divisor of 'n'
        if n % i == 0:

            # If it is, add it to the list of divisors
            divisors.append(i)

    # Return the list of all positive divisors of 'n' that are between 472 and 827, both inclusive.
    return divisors
