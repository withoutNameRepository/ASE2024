 def find_divisors_in_range(n):
    divisors = []
    for i in range(1, n + 1):
        if n % i == 0 and 46 <= i <= 89:
            divisors.append(i)
    return divisors

# Test case 1:
assert find_divisors_in_range(50) == [5, 10, 25]
# Test case 2:
assert find_divisors_in_range(100) == [4, 5, 8, 10, 20, 25, 40, 50, 75, 80, 90]
# Test case 3:
assert find_divisors_in_range(150) == [3, 5, 75, 100, 125, 150]
