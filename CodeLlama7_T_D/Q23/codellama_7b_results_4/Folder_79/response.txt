 def find_divisors_in_range(n):
    divisors = []
    for i in range(1, 8 + 1):
        if n % i == 0:
            divisors.append(i)
    return divisors

### Example usage of the function:

assert find_divisors_in_range(12) == [1, 2, 3, 4, 6, 12]

assert find_divisors_in_range(100) == [1, 2, 4, 5, 10, 20, 25, 50, 100]

assert find_divisors_in_range(81) == [3, 9, 27, 81]
