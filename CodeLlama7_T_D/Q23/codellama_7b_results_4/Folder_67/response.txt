def find_divisors_in_range(n):
    # Initialize an empty list to store the divisors
    divisors = []

    # Loop through all positive integers between 33 and 57, inclusive
    for i in range(33, 58):

        # Check if 'i' is a divisor of 'n'
        if n % i == 0:

            # If it is a divisor, add it to the list
            divisors.append(i)

    # Return the list of divisors in the specified range
    return divisors
