def find_divisors_in_range(n):
    # Initialize an empty list to store the divisors
    divisors = []

    # Iterate from 1 to n
    for i in range(1, n + 1):
        # Check if i is a divisor of n
        if n % i == 0:
            # If i is a divisor of n, check if it is between 43 and 46
            if 43 <= i <= 46:
                # If i is between 43 and 46, add it to the list of divisors
                divisors.append(i)

    # Return the list of divisors that are between 43 and 46
    return divisors
