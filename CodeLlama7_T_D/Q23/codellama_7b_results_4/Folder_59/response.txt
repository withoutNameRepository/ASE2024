def find_divisors_in_range(n):
    # Initialize an empty list to store the divisors
    divisors = []

    # Iterate from 3 to n - 1 (both inclusive)
    for i in range(3, n):

        # Check if i is a divisor of n
        if n % i == 0:

            # Add i to the list of divisors
            divisors.append(i)

    # Return the list of divisors
    return divisors
