 def find_divisors_in_range(n):
    divisors = []
    for i in range(1, n+1):
        if n % i == 0 and 11 <= i <= 41:
            divisors.append(i)
    return divisors

# Test case 1:
assert find_divisors_in_range(24) == [2, 3, 4, 6, 8, 12, 18]
# Test case 2:
assert find_divisors_in_range(100) == [2, 4, 5, 10, 20, 25, 50]
# Test case 3:
assert find_divisors_in_range(1000) == []
