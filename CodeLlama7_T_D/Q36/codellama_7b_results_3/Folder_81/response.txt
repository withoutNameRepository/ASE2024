def filter_chars(my_string):
    # Create a new string with the filtered characters
    filtered_string = ""

    # Iterate over each character in the input string
    for char in my_string:

        # If the current character is between indices 51 and 76 (both exclusive) and it is greater than '5' and smaller than 'f', remove all occurrences of that character within the input string while keeping the order of all characters intact.
        if 51 < my_string.index(char) < 76:
            if char > '5' and char < 'f':
                filtered_string += char
