def filter_chars(my_string):
    # Initialize an empty list to store the filtered characters
    filtered_chars = []

    # Iterate over the characters in the string
    for char in my_string:
        # If the current character is between 23 and 45 (exclusive) and is greater than 3 and smaller than I, then remove all occurrences of that character from the string
        if char not in filtered_chars and ord(char) > 23 and ord(char) < 45 and ord(char) > ord('3') and ord(char) < ord('I'):
            # Remove all occurrences of the current character from the string
            my_string = my_string.replace(char, '')

    # Return the altered string with all filtered characters removed
    return my_string
