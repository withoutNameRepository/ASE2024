def filter_chars(s):
    # Create a list of all characters in the string
    char_list = list(s)

    # Iterate through each character in the string
    for i in range(len(char_list)):
        # Check if the current character is between 54 and 92 (exclusive)
        if 53 < ord(char_list[i]) < 93:
            # If the current character is greater than 'J' and smaller than 'X', remove all occurrences of that character in the string
            char_list = [c for c in char_list if not (57 < ord(c) < 88 and c == char_list[i])]
    # Return the altered string
    return "".join(char_list)
