def filter_chars(input_string):
    # Initialize an empty string to store the filtered output
    filtered_output = ""
    
    # Iterate over each character in the input string
    for char in input_string:
        # If the current character is between 221 and 419 (exclusive)
        if 220 < ord(char) < 420:
            # And it is greater than the '-' character and smaller than the 'v' character
            if ord(char) > ord('-') and ord(char) < ord('v'):
                # Then remove all occurrences of this character in the string
                filtered_output = filtered_output.replace(char, "")
    return filtered_output
