def filter_chars(s):
    # Create a new string with the filtered characters
    new_str = ""

    # Iterate through the characters in the given string
    for char in s:

        # Check if the character is between 46 and 68 (exclusive)
        if 46 < ord(char) < 68:

            # Check if the character is greater than 'H' and smaller than 's'
            if char > 'H' and char < 's':

                # Skip this character in the new string
                continue

        # Append the character to the new string
        new_str += char

    # Return the altered string
    return new_str
