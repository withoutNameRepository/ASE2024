def identical_elements(list1, list2):
# Check if both lists are of the same size
if len(list1) != len(list2):
return set()

# Create a set to store all identical elements
identical_elements = set()

# Iterate over both lists and find all elements that occur at least once in both lists anywhere between index 30 and index 87, both inclusive.
for i in range(30, 87 + 1):
if list1[i] in list2:
identical_elements.add(list1[i])

# Return the set of all identical elements
return identical_elements
