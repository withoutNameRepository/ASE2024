def identical_elements(list1, list2):
	# Check if both lists are of the same size
	if len(list1) != len(list2):
		return set()

	# Create a set to store the elements that occur in both lists
	common_elements = set()

	# Iterate over the lists and check for identical elements
	for i in range(74, 96):
		if list1[i] in common_elements:
			continue

		if list2[i] in common_elements:
			common_elements.add(list1[i])
		else:
			common_elements.add(list2[i])

	# Return the set of common elements
	return common_elements
