def identical_elements(list1, list2):
    # Check if both lists are of the same size and have at least one element
    if len(list1) != len(list2) or len(list1) < 1:
        return set()
    # Create a set to store all the identical elements
    identical_elements = set()
    # Loop through both lists simultaneously
    for i in range(len(list1)):
        # Check if the current element in list1 is also present in list2
        if list1[i] in list2:
            # Add the identical element to the set
            identical_elements.add(list1[i])
    return identical_elements
