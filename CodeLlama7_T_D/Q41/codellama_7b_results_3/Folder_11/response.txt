def identical_elements(list1, list2):
    # Check if the two lists are of the same size
    if len(list1) != len(list2):
        return set()

    # Create a set to store all elements that occur at least once in both lists
    common_elements = set()

    # Loop through each element in list1 and check if it occurs in list2
    for element in list1:
        if element in list2:
            common_elements.add(element)

    # Return the set of all elements that occur at least once in both lists anywhere between index 37 and index 51, both inclusive
    return common_elements
