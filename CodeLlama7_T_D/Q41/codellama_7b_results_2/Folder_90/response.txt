def identical_elements(list1, list2):
    # Check if both lists are of the same size
    if len(list1) != len(list2):
        return set()

    # Get the intersection of the two lists
    intersection = set(list1).intersection(set(list2))

    # Filter the intersection to only include elements that occur at least once in both lists anywhere between index 262 and index 746, both inclusive
    filtered_intersection = []

    for element in intersection:
        if list1.count(element) >= 262 and list2.count(element) >= 262:
            if list1.index(element) <= 746 and list2.index(element) <= 746:
                filtered_intersection.append(element)

    return set(filtered_intersection)
