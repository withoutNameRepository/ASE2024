def find_primes_between_indices(my_list):
    # Initialize a list to store the prime numbers
    prime_numbers = []

    # Iterate through the list of integers, starting from index 60
    for i in range(60, len(my_list)):
        # Check if the current element is a prime number
        if is_prime(my_list[i]):

            prime_numbers.append(my_list[i])

    prime_numbers.sort(reverse=True)

# Define a function to check if a number is prime or not
def is_prime(n):
    # Check if n is less than 2, as we know that all positive numbers greater than 1 are either prime or composite.
    if n < 2:
        return False

    # Check if n is divisible by any number between 2 and the square root of n. If it is, then it is not prime.
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False

    # If we reach this point, then the number must be prime.
    return True
