def find_primes_between_indices(my_list):
    # Initialize an empty list to store the prime numbers
    prime_numbers = []

    # Loop through the list and check if each element is a prime number
    for i in range(len(my_list)):
        # Check if the current element is divisible by any of the previous elements
        is_prime = True
        for j in range(i):
            if my_list[i] % my_list[j] == 0:
                is_prime = False
                break

        # If the current element is a prime number, add it to the list of prime numbers
        if is_prime:
            prime_numbers.append(my_list[i])

    # Return the sorted list of prime numbers in ascending order
    return sorted(prime_numbers)
