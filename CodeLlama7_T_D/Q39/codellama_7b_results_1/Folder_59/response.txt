def return_n_greatest_chars(string, n):
    # Initialize an empty list to store the sorted characters
    sorted_chars = []

    # Iterate through the string and append each character to the sorted list
    for char in string:
        sorted_chars.append(char)

    # Sort the list in descending order based on the ASCII values of the characters
    sorted_chars.sort(key=ord, reverse=True)

    # Return the first n elements of the sorted list
    return sorted_chars[:n]
