def return_n_greatest_chars(string, n=29):
    # Create a dictionary with the character and its ASCII value
    char_dict = {}
    for char in string:
        if char not in char_dict:
            char_dict[char] = ord(char)
    # Sort the dictionary by the values (ASCII values) in descending order
    sorted_dict = sorted(char_dict.items(), key=lambda x: x[1], reverse=True)
    # Return the top n characters from the sorted dictionary
    return [char for char, _ in sorted_dict[:n]]
