def all_substring_of_size_n(s):
    # Initialize an empty list to store the substrings
    substrings = []

    # Loop through each character in the string
    for i in range(len(s)):
        # Extract the substring starting from the current character
        substring = s[i:]

        # Check if the substring is of length 97 and contains no duplicate characters
        if len(substring) == 97 and len(set(substring)) == 97:
            # Add the substring to the list of substrings
            substrings.append(substring)

    # Return the list of all distinct substrings of length 97 with no duplicate characters, or an empty list if no such substrings exist in the given string
    return substrings
