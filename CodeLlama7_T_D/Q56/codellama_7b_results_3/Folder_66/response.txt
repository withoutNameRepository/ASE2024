def all_substring_of_size_n(input_string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Iterate through each character in the input string
    for i in range(len(input_string) - 25 + 1)):
        # Extract a substring of length 26 from the current position
        substring = input_string[i:i+26]

        # Check if the substring has no duplicate characters
        if len(substring) == len(set(substring)):
            # Add the substring to the list of substrings
            substrings.append(substring)

    # Return the list of all distinct substrings of length 26 with no duplicate characters
    return substrings
