def all_substring_of_size_n(string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Iterate over the characters in the string
    for i in range(len(string) - 105 + 1):
        # Extract a substring of length 105 from the current position
        substring = string[i:i+105]

        # Check if the substring has no duplicate characters
        if len(set(substring)) == len(substring):
            # If yes, add it to the list of substrings
            substrings.append(substring)

    # Return the list of substrings
    return substrings
