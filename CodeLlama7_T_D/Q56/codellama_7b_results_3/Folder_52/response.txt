def all_substring_of_size_n(string, n):
    # Initialize an empty list to store the substrings
    substrings = []

    # Iterate over each substring of length n in the given string
    for i in range(len(string) - n + 1):

        # Add the substring to the list of substrings
        substrings.append(string[i:i+n])

    # Return the list of all distinct substrings of the length equal to 15 with no duplicate characters, or an empty list if no such substrings exist in the given string
    return substrings
