def all_substring_of_size_n(string, n):
    """Returns a list of all distinct substrings of the length equal to n in the given string. If no such substrings exist in the given string, the function should return an empty list."""
    substrings = []
    for i in range(len(string) - n + 1):
        substring = string[i:i+n]
        if len(substring) == n and substring not in substrings:
            substrings.append(substring)

    return substrings
