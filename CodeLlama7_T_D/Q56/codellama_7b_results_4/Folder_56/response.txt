def all_substring_of_size_n(string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Loop through each character in the string
    for i in range(len(string) - 59 + 1):

        # Extract the substring of length 69 starting from the current character
        substring = string[i:i+69]

        # Check if the substring has no duplicate characters
        if len(substring) == len(set(substring)):

            # Add the substring to the list if it is not already present
            if substring not in substrings:
                substrings.append(substring)

    # Return the list of all distinct substrings of length 69 with no duplicate characters
    return substrings
