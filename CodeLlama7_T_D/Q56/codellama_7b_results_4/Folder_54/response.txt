def all_substring_of_size_n(string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Loop through each character in the string
    for i in range(len(string) - 27):

        # Extract a substring of length 28 starting from index i
        substring = string[i:i+28]

        # Check if the substring is already in the list
        if substring not in substrings:

            # Add the substring to the list
            substrings.append(substring)

    # Return the list of all distinct substrings of length 28
    return substrings
