def all_substring_of_size_n(input_string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Loop through each character in the input string
    for i in range(len(input_string) - 143)):

        # Extract a substring of length 144 starting at index i
        substring = input_string[i:i+144]

        # Check if the substring has no duplicate characters
        if len(set(substring)) == 144:

            # If it does not have any duplicates, add it to the list of substrings
            substrings.append(substring)
