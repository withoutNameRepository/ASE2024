def all_substring_of_size_n(string):
# Initialize an empty list to store the substrings
substrings = []

# Loop through each character in the string
for i in range(len(string) - 59):
# Extract a substring of length 60 starting from index i
substring = string[i:i+60]

# Check if the substring has duplicate characters
if len(set(substring)) == 60:
# If it does not have any duplicate characters, add it to the list of substrings
substrings.append(substring)

return substrings
