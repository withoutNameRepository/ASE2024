def all_substring_of_size_n(string):
# Initialize an empty list to store the substrings
substrings = []
# Loop through each character in the string
for i in range(len(string) - 149 + 1):
# Extract a substring of length 149 starting at position i
substring = string[i:i+149]
# Check if the substring is distinct and contains no duplicate characters
if all(c not in substring for c in set(substring)) and len(set(substring)) == len(substring):
# If the substring is distinct and contains no duplicate characters, add it to the list of substrings
substrings.append(substring)
return substrings
