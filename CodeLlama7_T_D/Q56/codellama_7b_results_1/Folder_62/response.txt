def all_substring_of_size_n(string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Iterate through each character in the string
    for i in range(len(string) - 2):

        # Extract a substring of size 32 starting from the current character
        substring = string[i:i + 32]

        # If the substring is not already in the list, add it to the list
        if substring not in substrings:
            substrings.append(substring)

    # Return the list of all distinct substrings of size 32 with no duplicate characters
    return substrings
