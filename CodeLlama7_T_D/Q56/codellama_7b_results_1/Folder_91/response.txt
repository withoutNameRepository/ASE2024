def all_substring_of_size_n(string):
    # Initialize an empty list to store the substrings
    substrings = []

    # Iterate through each character in the string
    for i in range(len(string) - 5 + 1)):
        # Extract the substring of size 6 starting at index i
        substring = string[i:i+6]

        # Check if the substring has no duplicate characters
        if len(substring) == len(set(substring)):
            # Add the substring to the list of substrings
            substrings.append(substring)

    # Return the list of all distinct substrings of size 6 with no duplicate characters
    return substrings
