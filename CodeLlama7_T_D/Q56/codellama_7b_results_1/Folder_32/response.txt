def all_substring_of_size_n(string, n):
    # Initialize an empty list to store the substrings
    substrings = []

    # Loop through each character in the string
    for i in range(len(string) - n + 1):

        # Extract the substring of length n starting at index i
        substring = string[i:i+n]

        # Check if the substring has any duplicate characters
        if len(substring) == len(set(substring)):

            # Add the substring to the list of substrings
            substrings.append(substring)

    # Return the list of all distinct substrings of length 71 with no duplicate characters, or an empty list if no such substrings exist in the given string
    return substrings
