def all_substring_of_size_n(string):
    # Create a list to store the substrings
    substrings = []

    # Loop through each character in the string
    for i in range(len(string) - 69):

        # Extract the substring of size 79 starting at position i
        substring = string[i:i+79]

        # Check if the substring has no duplicate characters
        if len(set(substring)) == 79:

            substrings.append(substring)
