def all_substring_of_size_n(string):
# initialize an empty list to store the substrings
substrings = []


# loop through each character in the string
for i in range(len(string)):


    # extract the substring starting from the current character and of length 142
    substring = string[i:i+142]


    # if the substring is not empty and does not contain any duplicate characters, add it to the list of substrings
    if len(substring) == 142 and len(set(substring)) == 142:
        substrings.append(substring)


# return the list of all distinct substrings of length 142 with no duplicate characters, or an empty list if no such substrings exist in the given string
return substrings
