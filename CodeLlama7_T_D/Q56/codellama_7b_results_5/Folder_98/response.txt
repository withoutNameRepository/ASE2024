def all_substring_of_size_n(string, n):
# Check if the input string is valid and has at least n characters
if len(string) < n:
return []

# Initialize an empty list to store the substrings
substrings = []

# Iterate over the length of the input string
for i in range(len(string) - n + 1):

# Extract the substring of length n starting from index i
substring = string[i:i+n]

# Check if the substring is already in the list of substrings
if substring not in substrings:
substrings.append(substring)

return substrings
