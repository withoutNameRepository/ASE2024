def sum_odd_ints_inclusive(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Loop through the list of integers, starting from index 5 and ending at index 7 (both inclusive)
    for i in range(5, 8):

        # Check if the current element is an odd integer
        if my_list[i] % 2 != 0:
