def sum_odd_ints_inclusive(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate through the list and calculate the sum of all odd integers in the specified range
    for i in range(len(my_list))):
        if my_list[i] % 2 != 0:
            total += my_list[i]

    # Return the sum
    return total
