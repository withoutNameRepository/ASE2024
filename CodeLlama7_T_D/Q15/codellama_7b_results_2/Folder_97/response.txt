def sum_odd_ints_inclusive(my_list):
    # Initialize sum to 0
    sum = 0
    # Iterate through list from index 40 to index 80, both inclusive
    for i in range(40, 81):
        # If current element is an odd integer, add it to sum
        if my_list[i] % 2 != 0:
