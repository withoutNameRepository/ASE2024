def sum_odd_ints_inclusive(my_list):
    # Initialize sum to 0
    sum = 0
    # Iterate through the list from index 29 to index 51, both inclusive
    for i in range(29, 52):
        # Check if the current element is an odd integer
        if my_list[i] % 2 != 0:
