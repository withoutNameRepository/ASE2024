def lists_with_product_equal_n(my_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Loop through the elements of the given list
    for i in range(len(my_list)):
        # Initialize a variable to store the product of the elements in the current sublist
        product = 1

        # Loop through the elements of the remaining part of the given list after the current element
        for j in range(i+1, len(my_list)):
            # Calculate the product of the elements in the current sublist
            product *= my_list[j]

            # If the product equals 851, add the sublist to the list of sublists and break out of the inner loop
            if product == 851:
                sublists.append(my_list[i:j+1])
                break

        # If no sublist was added, continue looping through the remaining elements of the given list
        else:
            continue

        # Break out of the outer loop if a sublist was added
        break

    # Return the list of sublists
    return sublists
