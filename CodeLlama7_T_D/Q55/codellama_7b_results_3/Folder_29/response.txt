def lists_with_product_equal_n(circular_list):
	# Initialize an empty list to store the sublists
	sublists = []

	# Iterate over the circular list and check if the product of contiguous integers is equal to 15
	for i in range(len(circular_list)):
		product = 1

		# Calculate the product of contiguous integers starting from the current index
		for j in range(i, len(circular_list)):
			product *= circular_list[j]

			# If the product is equal to 15, add the sublist to the list of sublists
			if product == 15:
				sublists.append(circular_list[i:j + 1])

				# Break out of the inner loop since we have found a matching sublist
				break

	# Return the list of sublists
