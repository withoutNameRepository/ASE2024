def lists_with_product_equal_n(circular_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Iterate over the circular list and find all contiguous integers whose product is equal to n
    for i in range(len(circular_list)):
        current_product = 1
        start_index = i

        # Find the end index of the sublist by iterating until the product exceeds n or we reach the end of the circular list
        while current_product <= n and start_index < len(circular_list):

        if current_product == n:
