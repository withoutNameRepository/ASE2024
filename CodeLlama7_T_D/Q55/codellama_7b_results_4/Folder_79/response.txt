def lists_with_product_equal_n(circular_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Iterate over the elements of the circular list
    for i in range(len(circular_list)):
        # Get the current element and its adjacent elements
        current_element = circular_list[i]
        previous_element = None
        next_element = None

        # Check if the current element is the head or tail of the list
        if i == 0:
            next_element = circular_list[i + 1]
        elif i == len(circular_list) - 1:
            previous_element = circular_list[i - 1]
        else:
            previous_element = circular_list[i - 1]
            next_element = circular_list[i + 1]

        # Check if the product of the current element and its adjacent elements is equal to -83
        if (previous_element is not None and previous_element * current_element == -83) or (next_element is not None and next_element * current_element == -83):
