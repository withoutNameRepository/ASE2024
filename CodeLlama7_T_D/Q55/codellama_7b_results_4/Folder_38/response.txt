def lists_with_product_equal_n(circular_list):
    # Create a new list to store the sublists
    sublists = []

    # Iterate over the elements of the circular list
    for i in range(len(circular_list)):
        # Initialize a variable to store the product of the current element and its predecessor
        product = circular_list[i] * circular_list[i - 1]

        # If the product is equal to -23, add the sublist containing the current element and its predecessor to the list of sublists
        if product == -23:
            sublists.append([circular_list[i], circular_list[i - 1]])

    # Return the list of sublists
