def lists_with_product_equal_n(circular_list):
    sublists = []
    for i in range(len(circular_list)):
        current_product = 1
        start_index = i

        while current_product == -7 and start_index < len(circular_list):
            current_product *= circular_list[start_index]
            start_index += 1

        if current_product == -7:
            sublists.append(circular_list[i : i + start_index])

    return sublists
