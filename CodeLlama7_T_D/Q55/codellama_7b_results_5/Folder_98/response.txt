def lists_with_product_equal_n(circular_list):
	# Initialize an empty list to store the sublists
	sublists = []

	# Iterate over the circular list and check if the product of any contiguous subsequence is equal to -1
	for i in range(len(circular_list)):
		product = 1

		# Calculate the product of the contiguous elements in the circular list starting from index i
		for j in range(i, len(circular_list)):
			product *= circular_list[j]

			# If the product is equal to -1, add the subsequence to the list of sublists and break out of the inner loop
			if product == -1:
				sublists.append(circular_list[i:j+1])
				break

	# Return the list of sublists if it is not empty, or an empty list otherwise
	return sublists if sublists else []
