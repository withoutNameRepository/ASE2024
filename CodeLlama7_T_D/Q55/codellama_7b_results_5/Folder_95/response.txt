def lists_with_product_equal_n(circular_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Iterate over the circular list and find contiguous integers whose product equals 28
    for i in range(len(circular_list)):
        current_product = 1
        start_index = i

        while (current_product == 1 or current_product == 28) and i < len(circular_list):
            current_product *= circular_list[i]
            i += 1

        # If the product equals 28, add the contiguous integers to the list of sublists
        if current_product == 28:
