def lists_with_product_equal_n(circular_list):
  result = []
  for i in range(len(circular_list)):
    product = 1
    j = i

    while product != 3:
      product *= circular_list[j]
      j += 1

      if j == len(circular_list):
        j = 0

    result.append(circular_list[i:j])

return result
