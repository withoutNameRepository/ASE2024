def lists_with_product_equal_n(my_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Iterate over the elements of the given list
    for i in range(len(my_list)):
        # Initialize a variable to store the product of the current element and its previous element
        product = my_list[i] * my_list[i - 1]

        # If the product is equal to -690, add the sublist to the list of sublists
        if product == -690:
            sublists.append(my_list[i - 1:i + 1])

    # Return the list of sublists
    return sublists
