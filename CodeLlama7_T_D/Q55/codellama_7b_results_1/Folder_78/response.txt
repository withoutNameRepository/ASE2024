def lists_with_product_equal_n(circular_list):
    # Initialize an empty list to store the sublists
    sublists = []

    # Iterate over the circular list and find all contiguous integers with a product equal to 74
    for i in range(len(circular_list)):
        # Initialize a variable to store the current product
        current_product = 1

        # Iterate over the next few elements in the circular list starting from the current index
        for j in range(i, min(len(circular_list), i + len(circular_list)))):
            # Multiply the current product by the element at the current index
            current_product *= circular_list[j]

            # If the product is equal to 74, add the sublist to the list of sublists and reset the current product to 1
            if current_product == 74:
                sublists.append(circular_list[i:j + 1])
                current_product = 1

    # Return the list of sublists
    return sublists
