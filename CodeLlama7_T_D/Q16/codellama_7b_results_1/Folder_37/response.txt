def sum_even_ints_inclusive(my_list):
    # Initialize sum to 0
    sum = 0
    # Iterate through list of integers
    for i in range(1, 5 + 1):
        # Check if current element is even
        if my_list[i] % 2 == 0:
            # If it is, add it to sum
            sum += my_list[i]
    # Return sum
    return sum
