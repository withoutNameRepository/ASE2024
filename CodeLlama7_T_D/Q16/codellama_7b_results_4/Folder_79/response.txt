def sum_even_ints_inclusive(my_list):
    # Initialize a variable to store the sum
    total = 0
    # Iterate through the list from index 4 to index 4, both inclusive
    for i in range(4, 5):
        # Check if the current element is even
        if my_list[i] % 2 == 0:
