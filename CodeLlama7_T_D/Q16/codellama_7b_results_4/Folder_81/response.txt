def sum_even_ints_inclusive(my_list):
    # Initialize a variable to store the sum
    total = 0

    # Iterate through the list of integers, starting from index 10 and ending at index 100 (inclusive)
    for i in range(10, 101):
        # Check if the current element is an even integer
        if my_list[i] % 2 == 0:
            # If it is, add it to the total sum
            total += my_list[i]

    # Return the total sum of all even integers in the specified range, or 0 if no even integers exist
    return total
