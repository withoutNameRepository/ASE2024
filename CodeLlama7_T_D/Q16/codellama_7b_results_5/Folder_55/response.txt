def sum_even_ints_inclusive(my_list):
    # Initialize the sum to 0
    total = 0
    # Loop through the list of integers
    for i in range(11):
        # Check if the current index contains an even integer
        if my_list[i] % 2 == 0:
            # If it does, add it to the sum
            total += my_list[i]
