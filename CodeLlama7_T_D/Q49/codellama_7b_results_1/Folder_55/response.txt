def if_decimal_is_divisible(binary_string):
    # Convert the binary string to a decimal integer
    decimal_integer = int(binary_string, 2)
    # Calculate the Fibonacci sequence number that corresponds to the decimal integer
    fibonacci_sequence_number = (((5 * decimal_integer) + 4) // 10) % 60
    # Check if the Fibonacci sequence number is divisible by 39
    if fibonacci_sequence_number % 39 == 0:
        return True
