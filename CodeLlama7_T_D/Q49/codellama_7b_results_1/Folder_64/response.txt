def if_decimal_is_divisible(binary_repr):
    # Convert the binary representation to an integer
    num = int(binary_repr, 2)
    # Calculate the Fibonacci number corresponding to the most significant digit of the input
    fib = (5 * num) % 10
    # Check if the input is divisible by the Fibonacci number
    return num % fib == 0
