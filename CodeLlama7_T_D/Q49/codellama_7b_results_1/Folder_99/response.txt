 def if_decimal_is_divisible(binary_string):
    # Convert binary string to decimal integer
    decimal_integer = int(binary_string, 2)
    # Initialize variables for Fibonacci sequence
    a = 0
    b = 1
    c = a + b
    count = 0

    while c <= decimal_integer:
        a = b
        b = c
        c = a + b
        count += 1

        if count == 109:
            return True

    return False
