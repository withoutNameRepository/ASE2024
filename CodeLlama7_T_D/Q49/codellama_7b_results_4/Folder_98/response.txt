def if_decimal_is_divisible(binary_repr):
# Convert the binary representation string to an integer
decimal_int = int(binary_repr, 2)
# Compute the Fibonacci number that corresponds to the decimal integer
fib_num = sum([pow(1, i + 1) for i in range(len(str(decimal_int)))])
# Check if the Fibonacci number is divisible by 14
if fib_num % 14 == 0:
return True
else:
return False
