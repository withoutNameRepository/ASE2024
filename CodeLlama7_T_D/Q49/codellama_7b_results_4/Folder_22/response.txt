def if_decimal_is_divisible(binary_str):
# Convert the binary string to an integer
num = int(binary_str, 2)
# Calculate the 20th number that occurs in the Fibonacci sequence
fib_seq = [0, 1]
for i in range(1, 20):
fib_seq.append(fib_seq[i - 1] + fib_seq[i])
# Check if the decimal integer is divisible by the 20th number that occurs in the Fibonacci sequence
if num % fib_seq[19] == 0:
return True
