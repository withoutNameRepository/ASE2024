def if_decimal_is_divisible(binary_repr):
	# Convert binary representation to decimal integer
	decimal_int = int(binary_repr, 2)
	# Calculate the 24th number in the Fibonacci sequence
	fib24 = fibonacci_sequence(24)
	# Check if the decimal integer is divisible by the 24th number in the Fibonacci sequence
	if decimal_int % fib24 == 0:
		return True
	else:
		return False

def fibonacci_sequence(n):
	a, b = 0, 1
	for _ in range(n - 1):
		a, b = b, a + b
	return a
