def if_decimal_is_divisible(binary_representation):
    # Convert the binary representation to a decimal integer
    decimal_integer = int(binary_representation, 2)
    # Calculate the Fibonacci sequence up to the desired number
    fibonacci_sequence = [0, 1]
    while fibonacci_sequence[-1] <= decimal_integer:
        fibonacci_sequence.append(fibonacci_sequence[-2] + fibonacci_sequence[-3])
    # Check if the decimal integer is divisible by the 4th number in the Fibonacci sequence
    return decimal_integer % fibonacci_sequence[3] == 0
