def if_decimal_is_divisible(binary_string):
    # convert the binary string to an integer
    decimal_int = int(binary_string, 2)
    # compute the 7th number that occurs in the Fibonacci sequence
    fibonacci_sequence = [0, 1]
    for i in range(2, len(fibonacci_sequence)):
        fibonacci_sequence.append(fibonacci_sequence[i - 1] + fibonacci_sequence[i])
    seventh_number = fibonacci_sequence[6]
    # check if the decimal integer is divisible by the 7th number
    return decimal_int % seventh_number == 0
