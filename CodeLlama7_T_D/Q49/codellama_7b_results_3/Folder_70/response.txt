def if_decimal_is_divisible(binary_repr):
    # Convert the binary representation to an integer
    num = int(binary_repr, 2)
    # Find the first occurrence of 137 in the Fibonacci sequence
    for i in range(len(str(num)) - len(str(num)) % 6)):
        if str(num)[i:i + 6] == '137':
            return True

    # If no occurrence of 137 is found, return False
