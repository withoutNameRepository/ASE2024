def if_decimal_is_divisible(binary_string):
    # convert the binary string to a decimal integer
    decimal_int = int(binary_string, 2)
    # calculate the Fibonacci sequence up to the 105th term
    fib_seq = [0, 1]
    for i in range(2, 106):
