def if_decimal_is_divisible(binary_string):
# Convert the binary string to an integer
integer = int(binary_string, 2)
# Get the most significant digit of the integer
msd = integer // 10 ** (len(str(integer))) - 1
# Check if the msd is divisible by 99
if msd % 99 == 0:
return True
else:
return False
