def if_decimal_is_divisible(binary_string):
    # Convert the binary string to an integer
    decimal_int = int(binary_string, 2)
    # Calculate the number of digits in the binary string
    num_digits = len(binary_string)
    # Initialize variables for the Fibonacci sequence
    fib_1, fib_2 = 0, 1
    # Check if the decimal integer is divisible by the 48th number in the Fibonacci sequence
    for i in range(num_digits):
        if (decimal_int // pow(fib_1 + fib_2, i)) % 10 == 0:
            return True

    # If the loop completes without returning True, then the decimal integer is not divisible by the 48th number in the Fibonacci sequence
    return False
