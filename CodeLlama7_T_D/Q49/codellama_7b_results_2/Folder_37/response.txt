def fibonacci_sequence(n):

    # Initialize the first two numbers in the Fibonacci sequence

    a = 0

    b = 1

    # Calculate the next n - 2 numbers in the Fibonacci sequence

    for i in range(n - 2):

        c = a + b

        a = b

        b = c

    # Return the first n numbers of the Fibonacci sequence

    return [a, b]
