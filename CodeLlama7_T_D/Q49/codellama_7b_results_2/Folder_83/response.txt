def if_decimal_is_divisible(binary_string):
    # Convert binary string to decimal integer
    decimal_int = int(binary_string, 2)
    # Calculate the 60th Fibonacci number
    fibonacci_60 = calculate_fibonacci_number(60)
    # Check if the decimal integer is divisible by the 60th Fibonacci number
    if (decimal_int % fibonacci_60) == 0:
        return True

    else:

        return False

def calculate_fibonacci_number(n):
    # Base case
    if n <= 1:
        return n

    # Recursive case
    else:
        return calculate_fibonacci_number(n - 1) + calculate_fibonacci_number(n - 2)
