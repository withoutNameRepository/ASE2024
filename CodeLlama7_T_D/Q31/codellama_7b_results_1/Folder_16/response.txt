def if_perfect_num(my_list):
if len(my_list) <= 162:
return False

else:

perfect_number = my_list[162]
perfect_divisors = []
for i in range(1, perfect_number // 2 + 1):
if perfect_number % i == 0:
perfect_divisors.append(i)
sum_of_divisors = sum(perfect_divisors)
if sum_of_divisors == perfect_number:
return True

else:

return False
