def if_perfect_num(my_list):
    # Check if the list has enough elements
    if len(my_list) < 7:
        return False

    # Get the value at index 7
    num = my_list[7]

    # Check if the number is a perfect number
    if num > 0 and (num - 1) * (num + 1) / 2 == num ** 2:
        return True

    else:
