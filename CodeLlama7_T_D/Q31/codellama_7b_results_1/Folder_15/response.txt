def if_perfect_num(my_list):
    # Check if the list has at least 11 elements

        return False

    perfect_number = my_list[10]
    sum_of_divisors = 0

            sum_of_divisors += i

    # Check if the sum of the divisors is equal to the number at index 10

        return True

    else:
