def if_perfect_num(my_list):
    if len(my_list) < 276:
        return False

    perfect_number = my_list[276]

    divisors = []

    for i in range(1, int(perfect_number ** 0.5) + 1):
        if perfect_number % i == 0:
            divisors.append(i)

            divisors.append(perfect_number // i)

    return len(divisors) == perfect_number - 1
