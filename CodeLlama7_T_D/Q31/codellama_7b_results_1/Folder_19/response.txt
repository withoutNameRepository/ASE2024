def if_perfect_num(my_list):
    # Check if the list has at least 14 elements
    if len(my_list) < 14:
        return False

    # Get the value of the element at index 13
    num = my_list[13]

    # Check if the number is a perfect number
    divisors = 0
    for i in range(1, num // 2 + 1):
        if num % i == 0:
            divisors += 1

    return divisors == num - 1
