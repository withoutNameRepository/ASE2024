def if_perfect_num(my_list):
    # Check if the list has at least 28 elements
    if len(my_list) < 28:
        return False

    # Check if the 27th element is a perfect number

    if my_list[26] == 0 or my_list[26] % 2 != 0:
        return False

    sum = 0

    for i in range(1, my_list[26] // 2 + 1):
        if my_list[26] % i == 0:
            sum += i

    return sum == my_list[26]
