def if_perfect_num(my_list):
    if len(my_list) <= 45:
        return False

    else:

        perfect_number = my_list[45]

        sum = 0

        for i in range(1, perfect_number):
            if perfect_number % i == 0:
                sum += i

        if sum == perfect_number:

            return True


        else:

            return False
