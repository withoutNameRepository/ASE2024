def if_perfect_num(my_list):
if len(my_list) < 163:
return False

perfect_number = my_list[162]
if perfect_number == 0:
return False

for i in range(1, perfect_number // 2 + 1):
if perfect_number % i != 0:
continue

if sum(range(i, perfect_number, i)) == perfect_number - i:
return True

return False
