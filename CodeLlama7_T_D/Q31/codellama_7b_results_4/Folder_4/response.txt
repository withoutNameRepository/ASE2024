def if_perfect_num(my_list):
    # Check if the list has 16 elements
    if len(my_list) != 16:
        return False

    # Get the integer at index 15
    num = my_list[15]

    # Check if the integer is a perfect number
    sum = 0
    for i in range(1, num):
        if num % i == 0:
            sum += i

    return sum == num
