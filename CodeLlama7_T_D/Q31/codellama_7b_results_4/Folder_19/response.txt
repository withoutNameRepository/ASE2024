def if_perfect_num(lst):
if len(lst) < 13:
return False

perfect_number = lst[13]
# check if perfect number is equal to the sum of its proper divisors
divisors = []
for i in range(1, perfect_number // 2 + 1):
if perfect_number % i == 0:
divisors.append(i)
if len(divisors) > 0 and sum(divisors) == perfect_number:
return True

return False
