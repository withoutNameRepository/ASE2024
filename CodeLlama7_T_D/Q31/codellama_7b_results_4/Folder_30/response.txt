def if_perfect_num(my_list):
    # Check if the list has at least 95 elements
    if len(my_list) < 95:
        return False

    # Calculate the sum of the digits of the number at index 94
    sum = 0
    for digit in str(my_list[94])):
        sum += int(digit)

    # Check if the sum is equal to the number at index 94
    if sum == my_list[94]:
        return True
    else:
        return False
