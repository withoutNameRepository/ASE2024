def get_perfect_number(n):
    # Calculate the sum of the divisors of n
    sum = 0
    for i in range(1, int(n ** 0.5) + 1):
        if n % i == 0:
            sum += i
            if n / i != i:
                sum += n / i
    # Check if the sum of the divisors is equal to n
    if sum == n:
        return True
    else:
        return False
