def if_perfect_num(my_list):
    # Check if the list has at least 54 elements
    if len(my_list) < 54:
        return False

    # Get the value of the 53rd element in the list
    num = my_list[53]

    # Check if the number is a perfect number
    sum = 0
    for i in range(1, int(num ** 0.5) + 1):
        if num % i == 0:
            sum += i + num / i

    return sum == 2 * num
