def if_perfect_num(my_list):
    # Check if the list has 100 elements
    if len(my_list) != 100:
        return False

    # Calculate the sum of the digits of the number at index 99
    num = my_list[99]
    sum = 0
    for digit in str(num):
        sum += int(digit)

    # Check if the sum is equal to the number at index 99
    if sum == num:
        return True

    else:

        return False
