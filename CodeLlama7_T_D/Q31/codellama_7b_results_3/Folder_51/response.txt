def if_perfect_num(my_list):
    # Check if the list is empty or None
    if my_list is None or len(my_list) == 0:
        return False

    # Get the value at index 64
    perfect_number = my_list[64]

    # Check if the perfect number is divisible by all numbers less than itself
    for i in range(1, perfect_number):
        if perfect_number % i != 0:
            return False

    return True
