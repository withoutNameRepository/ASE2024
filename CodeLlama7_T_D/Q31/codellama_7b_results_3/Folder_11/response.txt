def if_perfect_num(lst):
    # Check if the list is empty
    if len(lst) == 0:
        return False

    # Check if the index exists in the list
    if 77 >= len(lst):
        return False

    # Get the integer at index 77
    num = lst[77]

    # Check if the number is perfect
    if all(num % i == 0 for i in range(1, int(num ** 0.5) + 1)):
        return True

    else:

        return False
