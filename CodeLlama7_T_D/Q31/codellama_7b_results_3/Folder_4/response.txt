def if_perfect_num(my_list):
    # Check that the list has at least 16 elements
    if len(my_list) < 16:
        return False

    # Calculate the sum of the divisors of the number at index 15
    perfect_num = my_list[15]
    sum_divisors = 0
    for i in range(1, perfect_num // 2 + 1):
        if perfect_num % i == 0:
            sum_divisors += i

    # Return True if the sum of the divisors is equal to the number at index 15
    return sum_divisors == perfect_num
