def if_perfect_num(my_list):
    # Check if the element at index 69 is a perfect number
    if my_list[69] == 0:
        return False

    else:

        for i in range(1, int(my_list[69])/2 + 1):
            if my_list[69] % i == 0:
                sum = 0

                for j in range(1, i + 1):
                    sum += j

                if sum == my_list[69]:
                    return True

        return False
