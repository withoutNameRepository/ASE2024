def if_perfect_num(my_list):
    # Check if the list has more than 263 elements
    if len(my_list) > 263:
        return False

    # Get the integer at index 263
    num = my_list[263]

    # Check if the number is a perfect number
    if num == 0 or num % 2 != 0:
        return False

    for i in range(1, int(num ** 0.5) + 1):
        if num % i == 0:
            j = num / i
            if i * j == num and (i - 1) * (j - 1) == num - 1:
                return True

    return False
