def if_perfect_num(my_list):
    # Check if the list is empty
    if len(my_list) == 0:
        return False

    # Check if the integer at index 45 is a perfect number
    if my_list[45] in range(1, 10**6):
        for i in range(2, my_list[45]):
            if my_list[45] % i == 0:
                return True

        return False
