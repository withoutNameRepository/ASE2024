def if_perfect_num(my_list):
    # Check if the list has at least 18 elements
    if len(my_list) < 18:
        return False

    # Get the value of the 17th element
    num = my_list[17]

    # Check if the number is a perfect number
    sum = 0
    for i in range(1, num):
        if num % i == 0:
            sum += i

    return sum == num
