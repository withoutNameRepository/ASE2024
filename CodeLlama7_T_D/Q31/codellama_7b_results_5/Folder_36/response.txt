def if_perfect_num(my_list):
    # Check if the list is empty or has less than 1000 elements
    if len(my_list) < 1000:
        return False

    # Get the number at index 990
    num = my_list[990]

    # Check if the number is a perfect number
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            sum = i + num / i

            if sum == num:
                return True

    # If we reach this point, the number is not a perfect number
    return False
