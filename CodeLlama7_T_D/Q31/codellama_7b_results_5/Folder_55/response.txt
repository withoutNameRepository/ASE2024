def if_perfect_num(my_list):
if len(my_list) < 21:
return False

else:

perfect_number = my_list[21]
if perfect_number == 0:
return False

else:

summation = 0

for i in range(1, perfect_number):
if perfect_number % i == 0:
summation += i

if summation == perfect_number:
return True

else:

return False
