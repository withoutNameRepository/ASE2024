def if_perfect_num(my_list):
    # Check if the list has at least 9 elements
    if len(my_list) < 9:
        return False

    # Get the number at index 8
    num = my_list[8]

    # Check if the number is a perfect number
    perfect_nums = [6, 28, 496, 8128, 33550336]
    if num in perfect_nums:
        return True

    # If the number is not a perfect number, return False
    return False
