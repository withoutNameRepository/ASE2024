def find_second_smallest_num(my_list):
    # Check if list is empty or has only one element
    if len(my_list) < 2:
        return None

    # Get the second smallest element from index 28 to index 32, both inclusive
    second_smallest = None
    for i in range(28, 32 + 1):
        if my_list[i] < second_smallest or second_smallest is None:
