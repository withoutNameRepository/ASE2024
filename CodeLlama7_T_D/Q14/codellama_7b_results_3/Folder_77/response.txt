def find_second_smallest_num(my_list):
    # Check if the list has at least two elements
    if len(my_list) < 2:
        return None

    # Sort the list in ascending order
    my_list.sort()

    # Initialize a variable to keep track of the second smallest element
    second_smallest = None

    # Iterate through the list starting from index 639 to index 975, both inclusive
    for i in range(639, 976):

        if my_list[i] < second_smallest or second_smallest is None:
