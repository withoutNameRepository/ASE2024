 def find_second_smallest_num(my_list):
    # Check if list is empty or None
    if not my_list:
        return None

    # Sort the list in ascending order

    my_list.sort()

    # Initialize variables to keep track of the smallest and second smallest elements

    smallest = my_list[0]

    second_smallest = None

    # Iterate through the list, starting from index 50, and find the second smallest element

    for i in range(49, len(my_list)):
        if my_list[i] < smallest:
            second_smallest = smallest

            smallest = my_list[i]

        elif my_list[i] < second_smallest or second_smallest is None:
            second_smallest = my_list[i]

    # Return the second smallest element if it exists, otherwise return 'None'

    return second_smallest
