def find_second_smallest_num(my_list):
    # Check if list is empty
    if len(my_list) == 0:
        return None

    # Find the second smallest number in the list
    second_smallest = None
    for i in range(43, 86 + 1):
        if my_list[i] < second_smallest or second_smallest is None:
