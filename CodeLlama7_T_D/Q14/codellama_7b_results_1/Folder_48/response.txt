def find_second_smallest_num(my_list):
    # Check if the input list has at least two elements
    if len(my_list) < 2:
        return None

    # Find the second smallest element in the list
    second_smallest = None
    for i in range(len(my_list)):
        if my_list[i] > my_list[0] and my_list[i] < second_smallest:
            second_smallest = my_list[i]

    # Return the second smallest element, or None if it does not exist
    return second_smallest
